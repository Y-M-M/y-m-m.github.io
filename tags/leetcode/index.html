    leetcode hot100 32 最长有效括号

    题目

    给你一个只包含 ‘(’ 和 ‘)’ 的字符串，找出最长有效（格式正确且连续）括号 子串 的长度。
    左右括号匹配，即每个左括号都有对应的右括号将其闭合的字符串是格式正确的，比如 “(()())"。

    示例

    示例 1：

    输入：s = “(()”
    输出：2
    解释：最长有效括号子串是 “()”

    示例 2：

    输入：s = “)()())”
    输出：4
    解释：最长有效括号子串是 “()()”

    示例 3：

    输入：s = "”
    输出：0

    提示

    • 0 <= s.length <= 3 * 104
    • s[i] 为 ‘(’ 或 ‘)’

    题解

    方法一

    • 分析
      • 以pos结尾的子串若是有效子串，则必满足如下条件
        • pos - 1 为’)‘时，则其前面可有任意个有效子串，再往前的字符为’(’
        • pos - 1为’(‘时，构成长度为2的有效子串
      • 在此基础上还需注意：有效子串拼接后仍为有效子串
    • 提交结果
      • 通过
    class Solution {
public:
    int findlen(const string& s, vector<int>& nowlen, int pos, int prepos)
    {
        if (s[prepos] == '(')
        {
            return pos - prepos + 1;
        }
        else
        {
            if (nowlen[prepos] == 0)
            {
                return 0;
            }
            else
            {
                int newprepos = prepos - nowlen[prepos];
                if (newprepos < 0) return 0;
                return findlen(s, nowlen, pos, newprepos);
            }
        }
        return 0;
    }

    void combine(vector<int>& nowlen, int pos)
    {
        int prepos = pos - nowlen[pos];
        if (prepos >= 0 && nowlen[prepos] > 0)
        {
            nowlen[pos] += nowlen[prepos];
            combine(nowlen, pos);
        }
    }

    int longestValidParentheses(string s) {
        int n = s.length();
        if (!n) return 0;
        vector<int> nowlen(n);
        nowlen[0] = 0;
        int ans = 0;
        for (int i = 1; i < n; i++)
        {
            if (s[i] == '(') nowlen[i] = 0;
            else
            {
                nowlen[i] = findlen(s, nowlen, i, i - 1);
                combine(nowlen, i);
            }
            ans = max(ans, nowlen[i]);
        }

        return ans;
    }
};

    方法二

    • 分析
      • 栈
      • 栈底一开始存入-1，之后
        • 对于 ‘(’，存入其下标
        • 对于 ‘)’，将栈顶元素出栈
          • 若出栈后栈中还有元素，说明刚刚出栈的元素与’)‘构成有效子串
          • 若出栈后栈中没有元素，说明刚刚出栈的元素不能与’)‘构成有效子串，此时应将当前’)‘的下标入栈，继续进行后续操作
      • 提交结果
        • 通过
    class Solution {
public:
    int longestValidParentheses(string s) {
        stack<int> invalidpos;
        invalidpos.push(-1);
        int ans = 0;
        for (int i = 0; i < s.length(); i++)
        {
            if (s[i] == '(')
            {
                invalidpos.push(i);
            }
            else
            {
                invalidpos.pop();
                if (invalidpos.size())
                {
                    ans = max(i - invalidpos.top(), ans);
                }
                else
                {
                    invalidpos.push(i);
                }
            }
        }
        return ans;
    }
};

    方法三

    • 分析
      • 从左往右遍历，只要保证左括号数目始终不超过右括号数目，等左右括号数目相等时，即可获得最长有效子串
      • 这种方法会漏掉一种情况，当左括号数目始终多于右括号时，就无法求的最长子串的长度
      • 从右往左遍历一次，重复此过程，即可解决这个问题
    • 提交结果
      • 通过
    class Solution {
public:
    int longestValidParentheses(string s) {
        int left = 0, right = 0;
        int n = s.length();
        int ans = 0;
        for (int i = 0; i < n; i++)
        {
            if (s[i] == '(') left++;
            else right++;
            if (left == right) ans = max(left * 2, ans);
            else if (right > left) left = 0, right = 0;
        }
        left = 0, right = 0;
        for (int i = n - 1; i >= 0; i--)
        {
            if (s[i] == '(') left++;
            else right++;
            if (left == right) ans = max(left * 2, ans);
            else if (left > right) left = 0, right = 0;
        }
        return ans;
    }
};

    leetcode hot100 416 分割等和子集

    题目

    给你一个 只包含正整数 的 非空 数组 nums 。请你判断是否可以将这个数组分割成两个子集，使得两个子集的元素和相等。

    示例

    示例一

    输入：nums = [1,5,11,5]
    输出：true
    解释：数组可以分割成 [1, 5, 5] 和 [11] 。

    示例二

    输入：nums = [1,2,3,5]
    输出：false
    解释：数组不能分割成两个元素和相等的子集。

    提示

    • 1 <= nums.length <= 200
    • 1 <= nums[i] <= 100

    题解

    方法一

    • 分析
      • 动态规划
        • 计算所有数字的和sum，然后可将问题转化为求数组中能不能找出和为sum/2的元素(元素不重复)，即0-1背包问题
      • 提交结果
        • 通过
    class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int sum = 0;
        int n = nums.size();
        for (int num : nums)
        {
            sum += num;
        }
        if (sum % 2) return false;
        int target = sum / 2;
        cout << target << endl;
        vector<vector<bool>> dp(n + 1, vector<bool>(target + 1, false));
        dp[0][0] = true;
        for (int i = 1; i <= n; i++)
        {
            if (nums[i - 1] > target) return false;
            else if (nums[i - 1] == target) return true;
            for (int j = 0; j <= target; j++)
            {
                if (nums[i - 1] > j) 
                    dp[i][j] = dp[i - 1][j];
                else
                {
                    dp[i][j] = dp[i - 1][j] || dp[i - 1][j - nums[i - 1]];
                }
            }
        }
        return dp[n][target];
    }
};

    leetcode hot100 64 最小路径和

    题目

    给定一个包含非负整数的 m x n 网格 grid ，请找出一条从左上角到右下角的路径，使得路径上的数字总和为最小。
    说明：每次只能向下或者向右移动一步。

    示例

    示例一


    输入：grid = [[1,3,1],[1,5,1],[4,2,1]]
    输出：7
    解释：因为路径 1→3→1→1→1 的总和最小。

    示例二

    输入：grid = [[1,2,3],[4,5,6]]
    输出：12

    提示

    • m == grid.length
    • n == grid[i].length
    • 1 <= m, n <= 200
    • 0 <= grid[i][j] <= 200

    题解

    方法一

    • 分析
      • 动态规划
    class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        for (int i = 1; i < n; i++)
        {
            grid[0][i] += grid[0][i - 1];
        }
        for (int i = 1; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (!j) grid[i][j] += grid[i - 1][j];
                else grid[i][j] = min(grid[i - 1][j], grid[i][j - 1]) + grid[i][j];
            }
        }
        return grid[m - 1][n - 1];
    }
};

    leetcode hot100 152 乘积最大子数组

    题目

    给你一个整数数组 nums ，请你找出数组中乘积最大的非空连续 子数组（该子数组中至少包含一个数字），并返回该子数组所对应的乘积。

    测试用例的答案是一个 32-位 整数。

    示例

    示例一

    输入: nums = [2,3,-2,4]
    输出: 6
    解释: 子数组 [2,3] 有最大乘积 6。

    示例二

    输入: nums = [-2,0,-1]
    输出: 0
    解释: 结果不能为 2, 因为 [-2,-1] 不是子数组。

    提示

    • 1 <= nums.length <= 2 * 104
    • -10 <= nums[i] <= 10
    • nums 的任何子数组的乘积都 保证 是一个 32-位 整数

    题解

    方法一

    • 分析
      • 动态规划
      • 最优子结构
        • maxproduct[i] = max(maxproduct[i - 1] * nums[i], minproduct[i - 1] * nums[i], nums[i])
        • minproduct[i] = min(maxproduct[i - 1] * nums[i], minproduct[i - 1] * nums[i], nums[i])
    • 提交结果
      • 通过
    class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int n = nums.size();
        vector<int> maxnum(n, 0);
        vector<int> minnum(n, 0);
        maxnum[0] = nums[0];
        minnum[0] = nums[0];
        for (int i = 1; i < n; i++)
        {
            maxnum[i] = max(maxnum[i - 1] * nums[i], nums[i]);
            maxnum[i] = max( minnum[i - 1] * nums[i], maxnum[i]);
            minnum[i] = min(minnum[i - 1] * nums[i], nums[i]);
            minnum[i] = min(maxnum[i - 1] * nums[i], minnum[i]);
        }
        int ans = -INT32_MAX;
        for (int num : maxnum)
        {
            ans = max(ans, num);
        }
        return ans;
    }
};

    方法二

    • 分析
      • 思路与方法一相同
      • 在方法一的基础上，其实只需要记录minproduct[i - 1], maxproduct[i - 1], minproduct[i], maxproduct[i]这四个数字，并不断更新即可
    • 提交结果
      • 通过
    class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int beforemax = nums[0];
        int beforemin = nums[0];
        int maxnum, minnum;
        int ans = beforemax;
        for (int i = 1; i < nums.size(); i++)
        {
            maxnum = max(beforemax * nums[i], nums[i]);
            maxnum = max(beforemin * nums[i], maxnum);
            minnum = min(beforemax * nums[i], nums[i]);
            minnum = min(beforemin * nums[i], minnum);
            if (maxnum > ans)
                ans = maxnum;
            beforemax = maxnum;
            beforemin = minnum;
        }
        return ans;
    }
};

    leetcode hot100 300 最长递增子序列

    题目

    给你一个整数数组 nums ，找到其中最长严格递增子序列的长度。
    子序列 是由数组派生而来的序列，删除（或不删除）数组中的元素而不改变其余元素的顺序。例如，[3,6,2,7] 是数组 [0,3,1,6,2,2,7] 的子序列。

    示例

    示例 1：

    输入：nums = [10,9,2,5,3,7,101,18]
    输出：4
    解释：最长递增子序列是 [2,3,7,101]，因此长度为 4 。

    示例 2：

    输入：nums = [0,1,0,3,2,3]
    输出：4

    示例 3：

    输入：nums = [7,7,7,7,7,7,7]
    输出：1

    提示

    • 1 <= nums.length <= 2500
    • -104 <= nums[i] <= 104

    题解

    方法一

    • 分析
      • 从前往后遍历一边数组
      • 使用unordered_map记录当前已经遍历到的数字对应的最长递增子串长度
      • 在遍历到一个数字时，查阅在此之前遍历的、比当前数字小的数字对应的递增子串长度，取最大长度，加1，即为当前数字对应的最大子传唱度
    • 提交结果
      • 通过
    • 复杂度
      • 时间复杂度O(n2)
      • 空间复杂度O(n)
    class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n = nums.size();
        if (n <= 1) return n;
        int ans = 1;
        unordered_map<int, int> nowlen;
        nowlen[nums[0]] = 1;
        for (int i = 1; i < n; i++)
        {
            for (int j = i - 1; j >= 0; j--)
            {
                if (nums[j] < nums[i]) 
                {
                    nowlen[nums[i]] = max(nowlen[nums[j]] + 1, nowlen[nums[i]]);
                }
            }
            if (nowlen[nums[i]] == 0) nowlen[nums[i]] = 1;
            ans = max(ans, nowlen[nums[i]]);
        }
        
        return ans;
    }
};

    输出: [[1],[1,1],[1,2,1],[1,3,3,1],[1,4,6,4,1]]

    class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> ans(numRows);
        ans[0].push_back(1);
        for (int i = 1; i < numRows; i++)
        {
            ans[i].push_back(1);
            for (int j = 0; j < i - 1; j++)
            {
                ans[i].push_back(ans[i - 1][j] + ans[i - 1][j + 1]);
            }
            ans[i].push_back(1);
        }
        return ans;
    }
};

    题解

    方法一

    • 分析
      • 动态规划

    • 提交结果
      • 通过
    • 复杂度

    class Solution {
public:
    int rob(vector<int>& nums) {
        int n = nums.size();
        if (n == 0) return 0;
        else if (n == 1) return nums[0];
        vector<int> maxmoney(n);
        maxmoney[0] = nums[0];
        maxmoney[1] = max(nums[0], nums[1]);
        for (int i = 2; i < n; i++)
        {
            maxmoney[i] = max(nums[i] + maxmoney[i - 2], maxmoney[i - 1]);
        }
        return maxmoney.back();
    }
};

    • 提交结果

    class Solution {
public:
    int rob(vector<int>& nums) {
        int n = nums.size();
        if (n == 0) return 0;
        else if (n == 1) return nums[0];
        int x = nums[0], y = max(nums[0], nums[1]);
        int z = 0;
        for (int i = 2; i < n; i++)
        {
            z = max(nums[i] + x, y);
            x = y;
            y = z;
            z = 0;
        }
        return y;
    }
};

    leetcode hot100 279 完全平方数

    题目

    给你一个整数 n ，返回 和为 n 的完全平方数的最少数量 。
    完全平方数 是一个整数，其值等于另一个整数的平方；换句话说，其值等于一个整数自乘的积。例如，1、4、9 和 16 都是完全平方数，而 3 和 11 不是。

    示例

    示例一

    输入：n = 12
    输出：3
    解释：12 = 4 + 4 + 4

    示例二

    输入：n = 13
    输出：2
    解释：13 = 4 + 9

    提示

    • 1 <= n <= 104

    题解

    方法一

    • 分析
      • 当i是完全平方数的时候，num[i] = 1
      • 当i不是完全平方数的时候，num[i] = min(num[j] + num[i - j])
    • 提交结果
      • 通过
    • 复杂度
      • 时间复杂度O(n2)
      • 空间复杂度O(n)
    class Solution {
public:
    int numSquares(int n) {
        if (n == 1) return 1;
        vector<int> num(n + 1, INT32_MAX);
        for (int i = 1; i * i <= n; i++)
        {
            num[i * i] = 1;
        }
        for (int i = 2; i <= n; i++)
        {
            if (num[i] == INT32_MAX)
            {
                for (int j = 1; j <= i / 2; j++)
                {
                    num[i] = min(num[j] + num[i - j], num[i]);
                }
            }
        }
        return num[n];
    }
};

    方法二

    • 分析
      • 一个数学定理可以帮助解决本题：「四平方和定理」。
      • 四平方和定理证明了任意一个正整数都可以被表示为至多四个正整数的平方和。这给出了本题的答案的上界。
      • 同时四平方和定理包含了一个更强的结论：当且仅当 n ≠ 4k×(8m+7) 时，n 可以被表示为至多三个正整数的平方和。因此，当 n=4k × (8m + 7) 时，n 只能被表示为四个正整数的平方和。此时我们可以直接返回 4。当 n ≠ 4k × (8m + 7) 时，我们需要判断到底多少个完全平方数能够表示 n，我们知道答案只会是 1,2,3 中的一个：
        • 答案为 1 时，则必有 n 为完全平方数，这很好判断；
        • 答案为 2 时，则有 n = a2 + b2，我们只需要枚举所有的 a(1 ≤ a ≤ n)，判断 n − a2是否为完全平方数即可；
        • 答案为 3 时，我们很难在一个优秀的时间复杂度内解决它，但我们只需要检查答案为 1 或 2 的两种情况，即可利用排除法确定答案。
    • 提交结果
      • 通过
    class Solution {
public:
    // 判断是否为完全平方数
    bool isPerfectSquare(int x) {
        int y = sqrt(x);
        return y * y == x;
    }

    // 判断是否能表示为 4^k*(8m+7)
    bool checkAnswer4(int x) {
        while (x % 4 == 0) {
            x /= 4;
        }
        return x % 8 == 7;
    }

    int numSquares(int n) {
        if (isPerfectSquare(n)) {
            return 1;
        }
        if (checkAnswer4(n)) {
            return 4;
        }
        for (int i = 1; i * i <= n; i++) {
            int j = n - i * i;
            if (isPerfectSquare(j)) {
                return 2;
            }
        }
        return 3;
    }
};
