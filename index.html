    int findlen(const string& s, vector<int>& nowlen, int pos, int prepos)
    {
        if (s[prepos] == '(')
        {
            return pos - prepos + 1;
        }
        else
        {
            if (nowlen[prepos] == 0)
            {
                return 0;
            }
            else
            {
                int newprepos = prepos - nowlen[prepos];
                if (newprepos < 0) return 0;

        }
        return 0;

    void combine(vector<int>& nowlen, int pos)
    {
        int prepos = pos - nowlen[pos];
        if (prepos >= 0 && nowlen[prepos] > 0)
        {
            nowlen[pos] += nowlen[prepos];
            combine(nowlen, pos);
        }
    }

    int longestValidParentheses(string s) {
        int n = s.length();
        if (!n) return 0;
        vector<int> nowlen(n);

        int ans = 0;
        for (int i = 1; i < n; i++)
        {
            if (s[i] == '(') nowlen[i] = 0;
            else
            {
                nowlen[i] = findlen(s, nowlen, i, i - 1);
                combine(nowlen, i);
            }
            ans = max(ans, nowlen[i]);
        }

        return ans;
    }

        if (n <= 1) return n;
        int ans = 1;
        unordered_map<int, int> nowlen;
        nowlen[nums[0]] = 1;
        for (int i = 1; i < n; i++)
        {
            for (int j = i - 1; j >= 0; j--)
            {
                if (nums[j] < nums[i]) 
                {
                    nowlen[nums[i]] = max(nowlen[nums[j]] + 1, nowlen[nums[i]]);
                }
            }
            if (nowlen[nums[i]] == 0) nowlen[nums[i]] = 1;
            ans = max(ans, nowlen[nums[i]]);
        }
        
        return ans;
    }
