    题目

    给定一个仅包含数字 2-9 的字符串，返回所有它能表示的字母组合。答案可以按 任意顺序 返回。
    给出数字到字母的映射如下（与电话按键相同）。注意 1 不对应任何字母。

    示例

    示例一

    输入：digits = “23”
    输出：[“ad”,“ae”,“af”,“bd”,“be”,“bf”,“cd”,“ce”,“cf”]

    示例二

    输入：digits = ""
    输出：[]

    示例三

    输入：digits = “2”
    输出：[“a”,“b”,“c”]

    提示

    • 0 <= digits.length <= 4
    • digits[i] 是范围 [‘2’, ‘9’] 的一个数字。

    题解

    方法一

    • 分析
      • 递归
    • 提交结果

    class Solution {
public:
    void dfs(string digits, int pos)
    {
        if (pos == n)
        {
            ans.push_back(agroup);
        }
        else
        {
            char c0 = digits[pos];
            for (char c : numtodigits[c0])
            {
                agroup[pos] = c;
                dfs(digits, pos + 1);
            }
        }
    }

    vector<string> letterCombinations(string digits) {
        n = digits.length();
        if (!n)
        {
            return ans;
        }
        numtodigits['2'] = {'a', 'b', 'c'};
        numtodigits['3'] = {'d', 'e', 'f'};
        numtodigits['4'] = {'g', 'h', 'i'};
        numtodigits['5'] = {'j', 'k', 'l'};
        numtodigits['6'] = {'m', 'n', 'o'};
        numtodigits['7'] = {'p', 'q', 'r', 's'};
        numtodigits['8'] = {'t', 'u', 'v'};
        numtodigits['9'] = {'w', 'x', 'y', 'z'};
        agroup.resize(n);
        dfs(digits, 0);
        return ans;
    }
private:
    vector<string> ans;
    unordered_map<char, vector<char>> numtodigits;
    string agroup;
    int n;
};

    题目

    给定两个整数数组 preorder 和 inorder ，其中 preorder 是二叉树的先序遍历， inorder 是同一棵树的中序遍历，请构造二叉树并返回其根节点。

    输入: preorder = [3,9,20,15,7], inorder = [9,3,15,20,7]
    输出: [3,9,20,null,null,15,7]

    示例二

    输入: preorder = [-1], inorder = [-1]
    输出: [-1]

    • 1 <= preorder.length <= 3000
    • inorder.length == preorder.length
    • -3000 <= preorder[i], inorder[i] <= 3000
    • preorder 和 inorder 均 无重复 元素
    • inorder 均出现在 preorder
    • preorder 保证 为二叉树的前序遍历序列
    • inorder 保证 为二叉树的中序遍历序列

      • 利用先序遍历确认当前子树的根节点

class Solution {
public:
    int findendl(vector<int> &preorder, int rootpos, int beginpos)
    {
        int i = beginpos + 1;
        while (i < preorder.size() && inpos[preorder[i]] < rootpos)
        {
            i++;
        }
        return i - 1;
    }

    TreeNode* buidlsubTree(vector<int> &preorder, vector<int>& inorder, int l1, int r1, int l2, int r2)
    {

        {
            return nullptr;
        }
        int rootval = preorder[l1];
        TreeNode* root = new TreeNode(rootval);
        if (r1 == l1)
        {
            return root;
        }
        if (inpos[rootval] == l2)
        {
            root->right = buidlsubTree(preorder, inorder, l1 + 1, r1, l2 + 1, r2);
            return root;
        }

        // cout << "preendl = " << preendl << endl;
        root->left = buidlsubTree(preorder, inorder, l1 + 1, preendl, l2, inpos[rootval] - 1);
        root->right = buidlsubTree(preorder, inorder, preendl + 1, r1,  inpos[rootval] + 1, r2);
        return root;
    }

    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        TreeNode* root = nullptr;
        if (preorder.size() == 0)
        {
            return root;
        }
        for (int i = 0; i < preorder.size(); i++)
        {
            prepos[preorder[i]] = i;
        }
        for (int i = 0; i < inorder.size(); i++)
        {
            inpos[inorder[i]] = i;
        }
        root = buidlsubTree(preorder, inorder, 0, preorder.size() - 1, 0, inorder.size() - 1);
        return root;
    }
private:
    unordered_map<int, int> prepos;
    unordered_map<int, int> inpos;

    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        TreeNode* root = nullptr;
        if (preorder.size() == 0)
        {
            return root;
        }
        for (int i = 0; i < inorder.size(); i++)
        {
            inpos[inorder[i]] = i;
        }
        root = new TreeNode(preorder[0]);
        stack<TreeNode*> st;
        st.push(root);
        for (int i = 1; i < preorder.size(); i++)
        {
            int preval = st.top()->val;
            int nowval = preorder[i];
            if (inpos[nowval] < inpos[preval])
            {
                TreeNode* node = new TreeNode(nowval);
                st.top()->left = node;
                st.push(node);
            }
            else
            {
                TreeNode* prenode = nullptr;
                while (!st.empty() && inpos[nowval] > inpos[preval])
                {
                    prenode = st.top();
                    st.pop();      
                    if (st.empty())
                    {
                        break;
                    }
                    preval = st.top()->val;
                }
                TreeNode* node = new TreeNode(nowval);
                st.push(node);
                prenode->right = node;
            }
        }
        return root;
    }
private:
    unordered_map<int, int> inpos;
};

    leetcode hot100 146 LRU缓存

    题目

    请你设计并实现一个满足 LRU (最近最少使用) 缓存 约束的数据结构。
    实现 LRUCache 类：

    • LRUCache(int capacity) 以 正整数 作为容量 capacity 初始化 LRU 缓存
    • int get(int key) 如果关键字 key 存在于缓存中，则返回关键字的值，否则返回 -1 。
    • void put(int key, int value) 如果关键字 key 已经存在，则变更其数据值 value ；如果不存在，则向缓存中插入该组 key-value 。如果插入操作导致关键字数量超过 capacity ，则应该 逐出 最久未使用的关键字。
      函数 get 和 put 必须以 O(1) 的平均时间复杂度运行。

    示例

    输入
    [“LRUCache”, “put”, “put”, “get”, “put”, “get”, “put”, “get”, “get”, “get”]
    [[2], [1, 1], [2, 2], [1], [3, 3], [2], [4, 4], [1], [3], [4]]
    输出 [null, null, null, 1, null, -1, null, -1, 3, 4]
    解释 LRUCache lRUCache = new LRUCache(2);
    lRUCache.put(1, 1); // 缓存是 {1=1}
    lRUCache.put(2, 2); // 缓存是 {1=1, 2=2}
    lRUCache.get(1); // 返回 1
    lRUCache.put(3, 3); // 该操作会使得关键字 2 作废，缓存是 {1=1, 3=3}
    lRUCache.get(2); // 返回 -1 (未找到)
    lRUCache.put(4, 4); // 该操作会使得关键字 1 作废，缓存是 {4=4, 3=3}
    lRUCache.get(1); // 返回 -1 (未找到)
    lRUCache.get(3); // 返回 3
    lRUCache.get(4); // 返回 4

        if (head == nullptr || head->next == nullptr)

        ListNode *fast = head->next->next;
        ListNode *slow = head;
        while (fast != slow)
        {
            if (fast == nullptr || fast->next == nullptr)
            {
                return false;
            }
            fast = fast->next->next;
            slow = slow->next;
        }
        return true;
    }
};

    /**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        unordered_set<ListNode*> nodes;
        while (head != nullptr && head->next != nullptr)
        {
            if (nodes.count(head) > 0)
            {
                return head;
            }
            nodes.insert(head);
            head = head->next;
        }
        return nullptr;
    }
};

    方法二

    • 分析
      • 快慢指针
      • 设置一个快指针fast和一个慢指针slow，fast每次移动两下，slow每次移动一下
      • 如果存在环，当fast与slow相遇的时候，fast走过的距离是slow走过的距离的两倍
      • 此时slow走过的距离，减去它进入环之后走过的距离，即为从head到入环点的距离
      • ->让slow继续走，同时设置另一个指针ptr，与slow同步移动，当ptr走到入环点的时候，slow刚好也会走到入环点(即slow与ptr相遇的点即未入环点)
    • 提交结果
      • 通过
    class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        ListNode *slow = head, *fast = head;
        while (fast != nullptr) {
            slow = slow->next;
            if (fast->next == nullptr) {
                return nullptr;
            }
            fast = fast->next->next;
            if (fast == slow) {
                ListNode *ptr = head;
                while (ptr != slow) {
                    ptr = ptr->next;
                    slow = slow->next;
                }
                return ptr;
            }
        }
        return nullptr;
    }
};

    leetcode hot100 41 缺失的第一个正数

    题目

    给你一个未排序的整数数组 nums ，请你找出其中没有出现的最小的正整数。
    请你实现时间复杂度为 O(n) 并且只使用常数级别额外空间的解决方案。

    示例

    示例 1：

    输入：nums = [1,2,0]
    输出：3
    解释：范围 [1,2] 中的数字都在数组中。

    示例 2：

    输入：nums = [3,4,-1,1]
    输出：2
    解释：1 在数组中，但 2 没有。

    示例 3：

    输入：nums = [7,8,9,11,12]
    输出：1
    解释：最小的正数 1 没有出现。

    提示

    • 1 <= nums.length <= 105
    • -231 <= nums[i] <= 231 - 1

    题解

    方法一

    • 分析
      • 哈希表记录next，对于出现的每个数字num，将next[num - 1] = num
      • 从0开始寻找next，第一个找不到的next就是第一个没有出现的正数
    • 提交结果
      • 通过
    class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        unordered_map<int, int> next;
        for (int num: nums)
        {
            if (num > 0)
            {
                next[num - 1] = num;
            } 
        }
        int i = 0;
        while (next[i])
        {
            i++;
        }
        return i + 1;
    }
};

    方法二

    • 分析
      • 在原数组上构建哈希表，nums[i]->i + 1这个数字
      • 构建完毕之后，从头开始遍历nums数组，找到的第一个nums[i] != i + 1的数字即为第一个未出现的正整数
      • 如果nums数组内所有的映射都正确，则n + 1即为第一个未出现的正整数
    • 提交结果
      • 通过
    class Solution {
public:
    int firstMissingPositive(vector<int>& nums) {
        int n = nums.size();
        for (int i = 0; i < n; i++)
        {
            while (nums[i] > 0 && nums[i] <= n && nums[i] != i + 1 && nums[i] != nums[nums[i] - 1])
            {
                int idx = nums[i] - 1;
                nums[i] = nums[idx];
                nums[idx] = idx + 1;
            }
        }
        for (int i = 0; i < n; i++)
        {
            if (nums[i] != i + 1)
            {
                return i + 1;
            }
        }
        return n + 1;
    }
};

  • leetcode hot100 73 矩阵置0

    leetcode hot100 73 矩阵置0

    题目

    给定一个 m x n 的矩阵，如果一个元素为 0 ，则将其所在行和列的所有元素都设为 0 。请使用 原地 算法。

    示例

    示例 1：

    输入：matrix = [[1,1,1],[1,0,1],[1,1,1]]
    输出：[[1,0,1],[0,0,0],[1,0,1]]

    示例 2：

    输入：matrix = [[0,1,2,0],[3,4,5,2],[1,3,1,5]]
    输出：[[0,0,0,0],[0,4,5,0],[0,3,1,0]]

    提示

    • m == matrix.length
    • n == matrix[0].length
    • 1 <= m, n <= 200
    • -231 <= matrix[i][j] <= 231 - 1

    题解

    方法一

    • 分析
      • 递归
      • 不显式使用任何额外空间
      • 但递归会占用空间……
    • 提交结果
      • 通过
    • 编码过程中的问题
      • 两层循环只break掉一个
      • 第二层循环不一定是从col开始的
    class Solution {
public:
    void rowto0(vector<vector<int>> &matrix, int row)
    {
        for (int j = 0; j < n; j++)
        {
            matrix[row][j] = 0;
        }
    }

    void colto0(vector<vector<int>> &matrix, int col)
    {
        for (int i = 0; i < m; i++)
        {
            matrix[i][col] = 0;
        }
    }

    void begintofind0(vector<vector<int>>& matrix, int row, int col)
    {
        int flag = 0;
        for (int i = row; i < m; i++)
        {
            int j;
            if (i == row)
                j = col;
            else
                j = 0;
            for (; j < n; j++)
            {
                if (!matrix[i][j])
                {
                    if (j < n - 1)
                    {
                        begintofind0(matrix, i, j + 1);
                    }
                    else if (i < m - 1)
                    {
                        begintofind0(matrix, i + 1, 0);
                    }
                    rowto0(matrix, i);
                    colto0(matrix, j);
                    flag = 1;
                    break;
                }
            }
            if (flag)
                break;
        }
    }
    void setZeroes(vector<vector<int>>& matrix) {
        m = matrix.size();
        n = matrix[0].size();
        begintofind0(matrix, 0, 0);
    }

private:
    int m;
    int n;
};

    方法二

    • 分析
      • 用m + n的额外空间存储需要置0的行和列
      • 遍历一遍matrix，找到所有需要置0的行和列
    • 提交结果
      • 通过
    class Solution {
public:
    void rowto0(vector<vector<int>> &matrix, int row)
    {
        for (int j = 0; j < n; j++)
        {
            if (matrix[row][j])
                matrix[row][j] = 0;
        }
    }

    void colto0(vector<vector<int>> &matrix, int col)
    {
        for (int i = 0; i < m; i++)
        {
            if (matrix[i][col])
                matrix[i][col] = 0;
        }
    }

    void setZeroes(vector<vector<int>>& matrix) {
        m = matrix.size();
        n = matrix[0].size();
        vector<int> row(m, 0);
        vector<int> col(n, 0);
        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (!matrix[i][j])
                {
                    row[i] = 1;
                    col[j] = 1;
                }
            }
        }
        for (int i = 0; i < m; i++)
        {
            if (row[i])
            {
                rowto0(matrix, i);
            }
        }
        for (int j = 0; j < n; j++)
        {
            if (col[j])
            {
                colto0(matrix, j);
            }
        }
    }

private:
    int m;
    int n;
};

    • 分析
      • 遍历一边matrix，同时将所有需要置0的行置0(除第一行)，并在第一行标记需要置0的列
      • 用一个额外的bool标记第一行是否需要置0
      • 将所有需要置0的列置0
      • 仅需要额外空间0

    class Solution {
public:
    void rowto0(vector<vector<int>> &matrix, int row)
    {
        for (int j = 0; j < n; j++)
        {
            if (matrix[row][j])
                matrix[row][j] = 0;
        }
    }

    void colto0(vector<vector<int>> &matrix, int col)
    {
        for (int i = 0; i < m; i++)
        {
            if (matrix[i][col])
                matrix[i][col] = 0;
        }
    }

    void setZeroes(vector<vector<int>>& matrix) {
        m = matrix.size();
        n = matrix[0].size();
        bool flag = false;
        for (int j = 0; j < n; j++)
        {
            if (!matrix[0][j])
            {
                flag = true;
            }
        }
        for (int i = m - 1; i > 0; i--)
        {
            bool needto0 = false;
            for (int j = 0; j < n; j++)
            {
                if (!matrix[i][j])
                {
                    matrix[0][j] = 0;
                    needto0 = true;
                }
            }
            if (needto0)
            {
                rowto0(matrix, i);
            }
        }
        for (int j = 0; j < n; j++)
        {
            if (!matrix[0][j])
            {
                colto0(matrix, j);
            }
        }
        if (flag)
        {
            rowto0(matrix, 0);
        }
    }

private:
    int m;
    int n;
};
