leetcode 1847 最近的房间

题目

一个酒店里有 n 个房间，这些房间用二维整数数组 rooms 表示，其中 rooms[i] = [roomIdi, sizei] 表示有一个房间号为 roomIdi 的房间且它的面积为 sizei。每一个房间号 roomIdi 保证是 独一无二 的。
同时给你 k 个查询，用二维数组 queries 表示，其中 queries[j] = [preferredj, minSizej] 。第 j 个查询的答案是满足如下条件的房间 id ：
房间的面积 至少 为 minSizej ，且abs(id - preferredj) 的值 最小 ，其中 abs(x) 是 x 的绝对值。 如果差的绝对值有 相等 的，选择 最小 的 id 。如果 没有满足条件的房间 ，答案为 -1 。
请你返回长度为 k 的数组 answer ，其中 answer[j] 为第 j 个查询的结果。

示例

示例一

输入：rooms = [[2,2],[1,2],[3,2]], queries = [[3,1],[3,3],[5,2]]
输出：[3,-1,3]
解释：查询的答案如下：
查询 [3,1] ：房间 3 的面积为 2 ，大于等于 1 ，且号码是最接近 3 的，为 abs(3 - 3) = 0 ，所以答案为 3 。
查询 [3,3] ：没有房间的面积至少为 3 ，所以答案为 -1 。
查询 [5,2] ：房间 3 的面积为 2 ，大于等于 2 ，且号码是最接近 5 的，为 abs(3 - 5) = 2 ，所以答案为 3 。

示例二

输入：rooms = [[1,4],[2,3],[3,5],[4,1],[5,2]], queries = [[2,3],[2,4],[2,5]]
输出：[2,1,3]
解释：查询的答案如下：
查询 [2,3] ：房间 2 的面积为 3 ，大于等于 3 ，且号码是最接近的，为 abs(2 - 2) = 0 ，所以答案为 2 。
查询 [2,4] ：房间 1 和 3 的面积都至少为 4 ，答案为 1 因为它房间编号更小。
查询 [2,5] ：房间 3 是唯一面积大于等于 5 的，所以答案为 3 。

提示

  • n == rooms.length
  • 1 <= n <= 105
  • k == queries.length
  • 1 <= k <= 104
  • 1 <= roomIdi, preferredj <= 107
  • 1 <= sizei, minSizej <= 107

题解

方法一

class Solution {
public:
    vector<int> closestRoom(vector<vector<int>>& rooms, vector<vector<int>>& queries) {
        auto cmp = [](vector<int> a, vector<int> b) -> bool {
            return a[1] < b[1];
        };
        sort(rooms.begin(), rooms.end(), cmp);
        vector<int> ans;
        for (int i = 0; i < queries.size(); i++)
        {
            vector<int> possbileans;
            int l = 0, r = rooms.size() - 1;
            
            while (l <= r) // 最后求得的l是第一个大与等于min
            {
                int mid = (l + r) >> 1;
                if (rooms[mid][1] < queries[i][1])
                {
                    l = mid + 1;
                }
                else 
                {
                    r = mid - 1;
                }
            }
            for (int j = l; j < rooms.size(); j++)
            {
                possbileans.push_back(rooms[j][0]);
            }

            if (!possbileans.size())
            {
                ans.push_back(-1);
            }
            else
            {
                sort(possbileans.begin(), possbileans.end());
                l = 0, r = possbileans.size() - 1;
                while (l <= r)
                {
                    int mid = (l + r) >> 1; 
                    if (possbileans[mid] <= queries[i][0])
                    {
                        l = mid + 1;
                    }
                    else
                    {
                        r = mid - 1;
                    }
                }
                if (!l)
                {
                    ans.push_back(possbileans[l]);
                }
                else if (r == possbileans.size() - 1)
                {
                    ans.push_back(possbileans.back());
                }
                else
                {
                    if (abs(possbileans[l] - queries[i][0]) < abs(possbileans[r] - queries[i][0]))
                    {
                        ans.push_back(possbileans[l]);
                    }
                    else
                    {
                        ans.push_back(possbileans[r]);
                    }
                }
                
            }
        }
        return ans;
    }
};

方法二

struct Event {
    int type; // 时间类型，0表示房间，1表示询问
    int size; 
    int id;
    int origin; // 事件在原数组中的编号

    Event(int _type, int _size, int _id, int _origin): type(_type), size(_size), id(_id), origin(_origin) {}

    bool operator< (const Event& that) const
    {
        return size > that.size || (size == that.size && type < that.type);
    }
};

class Solution {
public:
    vector<int> closestRoom(vector<vector<int>>& rooms, vector<vector<int>>& queries) {
        int m = rooms.size();
        int n = queries.size();

        vector<Event> events;
        for (int i = 0; i < m; ++i)
        {
            events.emplace_back(0, rooms[i][1], rooms[i][0], i);
        }
        for (int i = 0; i < n; ++i) 
        {
            events.emplace_back(1, queries[i][1], queries[i][0], i);
        }

        sort(events.begin(), events.end()); // 按照size降序排列，如果size相同，则将房间事件放在query事件之前
        vector<int> ans(n, -1);
        set<int> valid;
        for (const auto& event: events)
        {
            if (event.type == 0)
            {
                valid.insert(event.id); // 将符合下一次query的房间时间的id加入有序集合
            }
            else 
            {
                int dist = INT_MAX;
                auto it = valid.lower_bound(event.id); // 查找需爱笑的大于等于preferred的元素
                if (it != valid.end() && *it - event.id < dist)  // 有大于等于prefered的元素
                {
                    dist = *it - event.id;
                    ans[event.origin] = *it;
                }
                if (it != valid.begin()) // 并非所有元素都大于等于preferred
                {
                    // 查找最大的严格小于prefered的元素
                    it = prev(it);
                    if (event.id - *it <= dist)
                    {
                        dist = event.id - *it;
                        ans[event.origin] = *it;
                    }
                }
            }
        }
        return ans;
        
    }
};

      • 离线处理
      • 考虑到不同查询构造出来的size符合条件的id的集合有重复，则将query按照size从大到小排序，依次处理
      • 考虑到room与query具有相对应的属性，采用统一的数据结构存储、统一排序，可以减少在rooms中查询的时间消耗
