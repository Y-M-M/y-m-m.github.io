leetcode hot100 994 腐烂的橘子

题目

在给定的 m x n 网格 grid 中，每个单元格可以有以下三个值之一：
值 0 代表空单元格；
值 1 代表新鲜橘子；
值 2 代表腐烂的橘子。
每分钟，腐烂的橘子 周围 4 个方向上相邻 的新鲜橘子都会腐烂。
返回 直到单元格中没有新鲜橘子为止所必须经过的最小分钟数。如果不可能，返回 -1 。

示例

示例一


输入：grid = [[2,1,1],[1,1,0],[0,1,1]]
输出：4

示例二

输入：grid = [[2,1,1],[0,1,1],[1,0,1]]
输出：-1
解释：左下角的橘子（第 2 行， 第 0 列）永远不会腐烂，因为腐烂只会发生在 4 个方向上。

示例三

输入：grid = [[0,2]]
输出：0
解释：因为 0 分钟时已经没有新鲜橘子了，所以答案就是 0 。

提示

  • m == grid.length
  • n == grid[i].length
  • 1 <= m, n <= 10
  • grid[i][j] 仅为 0、1 或 2

题解

方法一

  • 分析
    • vector<vector> times记录腐烂的天数
    • 将所有起始已经腐烂的橘子在times中置为0
    • 用一个queue记录当前已经腐烂且将会扩散的橘子
    • 不断扩散腐烂，并且将新腐烂的橘子加入队列中
    • 注意：如果有橘子a能让另一个橘子b比当前times天数更少时就腐烂，也要将b加入对立，更新其他被b传染的橘子的腐烂天数

class Solution {
public:
    void spreadroot(vector<vector<int>>& grid, vector<vector<int>>& times, int i, int j)
    {
        if (i > 0 && grid[i - 1][j] == 1)
        {
            if (times[i - 1][j] < 0)
            {
                times[i - 1][j] = times[i][j] + 1;
                rotted.push(make_pair(i - 1, j));
            }
            else
            {
                if (times[i][j] + 1 < times[i - 1][j])
                {
                    times[i - 1][j] = times[i][j] + 1;
                    rotted.push(make_pair(i - 1, j));
                }
            }
        }
        if (i < m - 1 && grid[i + 1][j] == 1)
        {
            if (times[i + 1][j] < 0)
            {
                times[i + 1][j] = times[i][j] + 1;
                rotted.push(make_pair(i + 1, j));
            }
            else
            {
                if (times[i][j] + 1 < times[i + 1][j])
                {
                    times[i + 1][j] = times[i][j] + 1;
                    rotted.push(make_pair(i + 1, j));
                }
            }
        }
        if (j > 0 && grid[i][j - 1] == 1)
        {
            if (times[i][j - 1] < 0)
            {
                times[i][j - 1] = times[i][j] + 1;
                rotted.push(make_pair(i, j - 1));
            }
            else
            {
                if (times[i][j] + 1 < times[i][j - 1])
                {
                    times[i][j - 1] = times[i][j] + 1;
                    rotted.push(make_pair(i, j - 1));
                }
            }
        }
        if (j < n - 1 && grid[i][j + 1] == 1)
        {
            if (times[i][j + 1] < 0)
            {
                times[i][j + 1] = times[i][j] + 1;
                rotted.push(make_pair(i, j + 1));
            }
            else
            {
                if (times[i][j] + 1 < times[i][j + 1])
                {
                    times[i][j + 1] = times[i][j] + 1;
                    rotted.push(make_pair(i, j + 1));
                }
            }
        }
    }

    int orangesRotting(vector<vector<int>>& grid) {
        m = grid.size();
        if (!m)
        {
            return 0;
        }
        n = grid[0].size();
        if (!n)
        {
            return 0;
        }
        vector<vector<int>> times(m, vector<int>(n, -1));
        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (grid[i][j] == 2)
                {
                    rotted.push(make_pair(i, j));
                    times[i][j] = 0;
                }
            }
        }
        while (!rotted.empty())
        {
            auto rotfruit = rotted.front();
            rotted.pop();
            spreadroot(grid, times, rotfruit.first, rotfruit.second);
        }
        int maxtime = 0;
        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (grid[i][j] == 1)
                {
                    if (times[i][j] < 0)
                    {
                        return -1;
                    }
                    else
                    {
                        maxtime = max(maxtime, times[i][j]);
                    }
                }
            }
        }
        return maxtime;
    }
private:
    int m;
    int n;
    queue<pair<int, int>> rotted;
};
