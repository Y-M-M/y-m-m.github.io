    leetcode hot100 300 最长递增子序列

    题目

    给你一个整数数组 nums ，找到其中最长严格递增子序列的长度。
    子序列 是由数组派生而来的序列，删除（或不删除）数组中的元素而不改变其余元素的顺序。例如，[3,6,2,7] 是数组 [0,3,1,6,2,2,7] 的子序列。

    示例

    示例 1：

    输入：nums = [10,9,2,5,3,7,101,18]
    输出：4
    解释：最长递增子序列是 [2,3,7,101]，因此长度为 4 。

    示例 2：

    输入：nums = [0,1,0,3,2,3]
    输出：4

    示例 3：

    输入：nums = [7,7,7,7,7,7,7]
    输出：1

    提示

    • 1 <= nums.length <= 2500
    • -104 <= nums[i] <= 104

    题解

    方法一

    • 分析
      • 从前往后遍历一边数组
      • 使用unordered_map记录当前已经遍历到的数字对应的最长递增子串长度
      • 在遍历到一个数字时，查阅在此之前遍历的、比当前数字小的数字对应的递增子串长度，取最大长度，加1，即为当前数字对应的最大子传唱度
    • 提交结果
      • 通过
    • 复杂度
      • 时间复杂度O(n2)
      • 空间复杂度O(n)
    class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n = nums.size();
        if (n <= 1) return n;
        int ans = 1;
        unordered_map<int, int> nowlen;
        nowlen[nums[0]] = 1;
        for (int i = 1; i < n; i++)
        {
            for (int j = i - 1; j >= 0; j--)
            {
                if (nums[j] < nums[i]) 
                {
                    nowlen[nums[i]] = max(nowlen[nums[j]] + 1, nowlen[nums[i]]);
                }
            }
            if (nowlen[nums[i]] == 0) nowlen[nums[i]] = 1;
            ans = max(ans, nowlen[nums[i]]);
        }
        
        return ans;
    }
};

    题目

    给定两个大小分别为 m 和 n 的正序（从小到大）数组 nums1 和 nums2。请你找出并返回这两个正序数组的 中位数 。
    算法的时间复杂度应该为 O(log (m+n)) 。

    示例

    示例一

    输入：nums1 = [1,3], nums2 = [2]
    输出：2.00000
    解释：合并数组 = [1,2,3] ，中位数 2

    示例二

    输入：nums1 = [1,2], nums2 = [3,4]
    输出：2.50000
    解释：合并数组 = [1,2,3,4] ，中位数 (2 + 3) / 2 = 2.5

    提示

    • nums1.length == m
    • nums2.length == n
    • 0 <= m <= 1000
    • 0 <= n <= 1000
    • 1 <= m + n <= 2000
    • -106 <= nums1[i], nums2[i] <= 106

    题解

    方法一

    • 分析
      • 合并排序
      • 以额外的空间排好序之后可以立刻获得答案
    • 提交结果

    class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        vector<int> nums;
        int i = 0, j = 0;
        int m = nums1.size(), n = nums2.size();
        while (i < m && j < n)
        {
            if (nums1[i] <= nums2[j])
            {
                nums.push_back(nums1[i]);
                i++;
            }
            else
            {
                nums.push_back(nums2[j]);
                j++;
            }
        }
        while (i < m)
        {
            nums.push_back(nums1[i]);
            i++;
        }
        while (j < n)
        {
            nums.push_back(nums2[j]);
            j++;
        }
        int mid = (m + n) / 2;
        if ((m + n) % 2)
        {
            return nums[mid];
        }
        return ((double)nums[mid - 1] + nums[mid]) / 2;
    }
};

    class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        if (nums1.size() > nums2.size())
        {
            return findMedianSortedArrays(nums2, nums1);
        }
        // 保证m < n
        int m = nums1.size();
        int n = nums2.size();
        int left = 0, right = m;
        int median1 = 0, median2 = 0;

        while (left <= right)
        {
            int i = (left + right) / 2;
            int j = (m + n + 1) / 2 - i;

            int nums_im1 = (i == 0 ? INT_MIN : nums1[i - 1]); // 第一个数组左半部分的最大值
            int nums_i = (i == m ? INT_MAX : nums1[i]); // 第一个数组右半部分的最小值
            int nums_jm1 = (j == 0 ? INT_MIN : nums2[j - 1]); // 第二个数组左半部分的最大值
            int nums_j = (j == n ? INT_MAX : nums2[j]); // 第二个数组右半部分的最小值

            if (nums_im1 <= nums_j)
            {
                median1 = max(nums_im1, nums_jm1);
                median2 = min(nums_i, nums_j);
                left = i + 1;
            }
            else
            {
                right = i - 1;
            }
        }

        return (m + n) % 2 == 0 ? (median1 + median2) / 2.0 : median1;
    }
};

    如果数组中不存在目标值 target，返回 [-1, -1]。
    你必须设计并实现时间复杂度为 O(log n) 的算法解决此问题。

    示例

    示例一

    输入：nums = [5,7,7,8,8,10], target = 8
    输出：[3,4]

    示例二

    输入：nums = [5,7,7,8,8,10], target = 6
    输出：[-1,-1]

    示例三

    输入：nums = [], target = 0
    输出：[-1,-1]

    提示

    • 0 <= nums.length <= 105
    • -109 <= nums[i] <= 109
    • nums 是一个非递减数组
    • -109 <= target <= 109

    题解

    • 分析
      • 二分查找找到元素出现的最后一个位置
      • 从该位置开始往前查找
    • 提交结果
      • 通过
    class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        vector<int> ans = {-1, -1};
        if (!nums.size()) return ans; 
        int l = 0, r = nums.size() - 1;
        while (l <= r)
        {
            int mid = (l + r) / 2;
            if (nums[mid] <= target)
            {
                l = mid + 1;
            }
            else
            {
                r = mid - 1;
            }
        }
        if (l == 0)
        {
            if (nums[0] == target)
            {
                ans[0] = 0;
                ans[1] = 0;
            }
            return ans;
        }
        if (nums[l - 1] == target)
        {
            ans[1] = l - 1;
            l--;
            while (l >= 0 && nums[l] == target) l--;
            ans[0] = l + 1;
        }
        return ans;
    }
};

    class Solution {

    class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        int m = matrix.size();
        if (!m) return false;
        int n = matrix[0].size();
        int l = 0, r = m * n - 1;

            }
            else
            {
                r = mid - 1;
            }
        }

    class Solution {
public:
    bool binaryfind(vector<vector<int>>& matrix, const int& target, const int& bias)
    {
        int l = 0, r = mindim - 1;
        while (l <= r)
        {
            int mid = (l + r) / 2;
            int nownum;
            if (m > n)
            {
                nownum = matrix[mid + bias][mid];
            }
            else if (m < n)
            {
                nownum = matrix[mid][mid + bias];
            }
            else
            {
                nownum = matrix[mid][mid];
            }
            if (nownum > target)
            {
                r = mid - 1;
            }
            else
            {
                l = mid + 1;
            }
        }
        if (r < 0)
        {
            return false;
        }
        int lastr = mindim - 1;
        int lastc = mindim - 1;
        int firstr = r;
        int firstc = r;
        int secondr = l;
        int secondc = l;
        if (m > n)
        {
            lastr += bias;
            firstr += bias;
            secondr += bias;
        }
        else if (m < n)
        {
            lastc += bias;
            firstc += bias;
            secondc += bias;
        }

        if (l >= mindim && matrix[lastr][lastc] != target)
        {
            return false;
        }
        if (matrix[firstr][firstc] == target)
        {
            return true;
        }
        for (int i = firstr + 1; i < m; i++)
        {
            if (matrix[i][firstc] == target)
            {
                return true;
            }
            if (matrix[i][firstc] > matrix[secondr][secondc])
            {
                break;
            }
        }
        for (int j = firstc + 1; j < n; j++)
        {
            if (matrix[firstr][j] == target)
            {
                return true;
            }
            if (matrix[firstr][j] > matrix[secondr][secondc])
            {
                break;
            }
        }
        for (int i = firstr - 1; i >= 0; i--)
        {
            if (matrix[i][secondc] == target)
            {
                return true;
            }
            if (matrix[i][secondc] < matrix[firstr][firstc])
            {
                break;
            }
        }
        for (int j = firstc - 1; j >= 0; j--)
        {
            if (matrix[secondr][j] == target)
            {
                return true;
            }
            if (matrix[secondr][j] < matrix[firstr][firstc])
            {
                break;
            }
        }
        if (matrix[firstr + 1][firstc + 1] == target)
        {
            return true;
        }
        return false;
    }

        m = matrix.size();

            auto it = lower_bound(row.begin(), row.end(), target);
            if (it != row.end() && *it == target)
            {
                return true;
            }
        }
        return false;
    }
};

      • 一个元素(x, y)可以确定
        • 如果matrix(x, y) > target，则第其以下、以右的元素均 > target
        • 如果matrix(x, y) < target, 则其以左、以上的元素均 < target
      • 同时考虑这两个因素，可知左下角和右上角的元素可以很容易的通过简单比较而让矩阵缩小

    class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        int m = matrix.size();
        int n = matrix[0].size();
        int x = 0, y = n - 1;
        while (x <= m - 1 && y >= 0)
        {
            if (matrix[x][y] > target)
            {
                y--;
            }
            else if (matrix[x][y] < target)
            {
                x++;
            }
            else
            {
                return true;
            }
        }
        return false;
    }
};
