    class Solution {

    bool judgeSquareSum(int c) {
        unordered_set<int> little_squares;
        long long base = 0;
        long long square = base * base;
        while (square <= c / 2) {
            little_squares.insert(square);
            if (square + square == c) return true;
            base++;
            square = base * base;
        }
        while (square <= c) {

            base++;
            square = base * base;
        }
        return false;

    class Solution {

    int minimumDifference(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> differ(n);
        for (int i = 0; i < n; i++) {
            differ[i] = abs(nums[i] - k);
        }
        vector<int> or_results(n);
        or_results[0] = nums[0];
        for (int i = 1; i < n; i++) {
            or_results[i] = or_results[i - 1] | nums[i]; 
            int now_differ = abs(or_results[i] - k);
            if (now_differ < differ[i]) {
                differ[i] = now_differ;
            }
            else {
                or_results[i] = nums[i];
            }
        }
        int min_differ = differ[0];
        for (int i = 1; i < n; i++) {
            min_differ = min(min_differ, differ[i]);
        }
        return min_differ;
    }
};

    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        vector<int> rest_gas(gas.size());
        
        int first = 0;
        int min_gas = 10000;
        for (int i = 0; i < rest_gas.size(); i++) {

            else rest_gas[i] = rest_gas[i - 1] + gas[i] - cost[i];

                first = (i + 1) % rest_gas.size();
                min_gas = rest_gas[i];

    class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int n = gas.size();
        int i = 0;
        while (i < n) {
            int sumOfGas = 0, sumOfCost = 0;
            int cnt = 0;
            while (cnt < n) {
                int j = (i + cnt) % n;
                sumOfGas += gas[j];
                sumOfCost += cost[j];
                if (sumOfCost > sumOfGas) {
                    break;
                }
                cnt++;
            }
            if (cnt == n) {
                return i;
            } else {
                i = i + cnt + 1;
            }
        }
        return -1;
    }
};
