    {
        if (root == nullptr)
        {
            return 0;
        }
        int leftmax = dfs(root->left);
        int rightmax = dfs(root->right);
        int maxpath = 0;
        if (leftmax > 0)  maxpath += leftmax;
        if (rightmax > 0) maxpath += rightmax;
        maxpath += root->val;
        if (maxpath > ans)

            ans = maxpath;
        }
        int addnum = max(leftmax, rightmax);
        if (addnum > 0)
            return addnum + root->val;
        return root->val;

        if (root == nullptr)

            return 0;
        }
        ans = -INT32_MAX;
        dfs(root);
        return ans;
    }
private:
    int ans;

    /**
 * Definition for a binary tree node.
 * struct TreeNode {

 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void dfs(TreeNode* root, TreeNode* p, TreeNode* q)
    {
        if (root == nullptr)
        {
            return;
        }
        st.push(root);
        if (root == p || root == q)
        {
            if (st1.empty())
            {
                st1 = st;
            }
            else
            {
                st2 = st;
                return;
            }
        }
        dfs(root->left, p, q);
        dfs(root->right, p, q);
        st.pop();
    }

    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        dfs(root, p, q);
        int cnt = st1.size() - st2.size();
        if (cnt > 0)
        {
            for (int i = 0; i < cnt; i++)
            {
                st1.pop();
            }
        }
        else if (cnt < 0)
        {
            cnt = -cnt;
            for (int i = 0; i < cnt; i++)
            {
                st2.pop();
            }
        }
        while (!st1.empty() && st1.top() != st2.top())
        {
            st1.pop();
            st2.pop();
        }
        // cout << st1.top()->val << ", " << st2.top()->val << endl;
        if (!st1.empty())
        {
            return st1.top();
        }
        return nullptr;
    }
private:
    stack<TreeNode*> st;
    stack<TreeNode*> st1;
    stack<TreeNode*> st2;
};

    /**

    /**

 * struct TreeNode {

 *     TreeNode *left;

class Solution {
public:
    void addtoSub(TreeNode* root, int addval) {
        if (root == nullptr)
        {
            return ;
        }
        root->val += addval;
        // cout << "root->val = " << root->val << endl;
        if (root->left != nullptr)
        {
            addtoSub(root->left, root->val);
        }
        if (root->right != nullptr)
        {
            addtoSub(root->right, root->val);
        }
    }
    void sub(TreeNode* root, const int&targetSum, int subnum)
    {
        if (root == nullptr)
        {
            return;
        }
        root->val -= subnum;
        // cout << root->val << endl;
        if (root->val == targetSum)
        {
            ans++;
        }
        if (root->left != nullptr)
        {
            sub(root->left, targetSum, subnum);
        }
        if (root->right != nullptr)
        {
            sub(root->right, targetSum, subnum);
        }
    }
    void findans(TreeNode* root, const int& targetSum, int subnum)
    {
        //cout << "root->val = " << root->val << endl;
        if (root == nullptr)
        {
            return;
        }
        sub(root, targetSum, subnum);
        if (root->left != nullptr)
        {
            findans(root->left, targetSum, root->val);
        }
        if (root->right != nullptr)
        {
            findans(root->right, targetSum, root->val);
        }
    }
    int pathSum(TreeNode* root, int targetSum) {
        ans = 0;
        if (root == nullptr || root->val == 1000000000)
        {
            return ans;
        }
        addtoSub(root, 0);
        findans(root, targetSum, 0);
        return ans;
    }
private:
    int ans;
};

    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int dfs(TreeNode* root, long long curr, int targetSum)
    {
        if (!root)
        {
            return 0;
        }

        int ret = 0;
        curr += root->val;
        if (prefix.count(curr - targetSum))
        {
            ret = prefix[curr - targetSum];
        }
        prefix[curr]++;
        ret += dfs(root->left, curr, targetSum);
        ret += dfs(root->right, curr, targetSum);
        prefix[curr]--;

        return ret;
    }


    int pathSum(TreeNode* root, int targetSum) {
        prefix[0] = 1;
        return dfs(root, 0, targetSum);
    }
private:
    unordered_map<long long, int> prefix;
};

    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void flatten(TreeNode* root) {
        if (root == nullptr)
        {
            return;
        }
        TreeNode* temp = nullptr;
        TreeNode* rend = nullptr;
        while (root->left != nullptr || root->right != nullptr)
        {
            if (root->left != nullptr)
            {
                temp = root->right;
                root->right = root->left;
                root->left = nullptr;
                root = root->right;
                rend = root;
                while (rend->right != nullptr)
                {
                    rend = rend->right;
                }
                rend->right = temp;
            }
            else
            {
                root = root->right;
            }
        }
    }
};
