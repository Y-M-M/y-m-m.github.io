    class Solution {
public:
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        for (int i = 0; i < k; i++)
        {
            int pos = 0;
            int minnum = nums[0];
            for (int j = 1; j < nums.size(); j++)
            {
                if (nums[j] < minnum)
                {
                    pos = j;
                    minnum = nums[j];
                }
            }
            nums[pos] *= multiplier;
        }
        return nums;
    }
};

    class Solution {
public:
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        auto cmp = [&nums](int left, int right) {return nums[left] > nums[right];};
        priority_queue<int, vector<int>, decltype(cmp)> id(cmp);

        for (int i = 0; i < nums.size(); i++)
        {
            id.push(i);
        }
        for (int i = 0; i < k; i++)
        {
            int updatetarget = id.top();
            id.pop();
            nums[updatetarget] *= multiplier;
            id.push(updatetarget);
        }
        return nums;
    }
};

    方法三

    class Solution {
public:
    long long quickMul(long long x, long long y, long long m)
    {
        long long res = 1;
        while (y)
        {
            if (y & 1) {
                res = (res * x) % m;
            }
            y >>= 1;
            x = (x * x) % m;
        }
        return res;
    }
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        if (multiplier == 1)
        {
            return nums;
        }
        long long n = nums.size(), m = 1e9 + 7;
        long long mx = *max_element(nums.begin(), nums.end()); // 取最大值
        vector<pair<long long, int>> v(n);
        for (int i = 0; i < n; i++)
        {
            v[i] = {nums[i], i};
        }
        make_heap(v.begin(), v.end(), greater<>());
        for (; v[0].first < mx && k; k--)
        {
            pop_heap(v.begin(), v.end(), greater<>());
            v.back().first *= multiplier;
            push_heap(v.begin(), v.end(), greater<>());
        }
        sort(v.begin(), v.end());
        for (int i = 0; i < n; i++)
        {
            int t = k / n + (i < k % n);
            nums[v[i].second] = ((v[i].first % m) * quickMul(multiplier, t, m)) % m;
        }
        return nums;
    }
};
    • 提交结果
      • 一：通过
      • 二：通过
    • 分析
      • 通过pair的排序规则来进行堆排序，先比较first，first相同的时候在比较second
      • 对过程进行优化
        • 如果nums中所有数都被更新过了，那么此后更新会从小到大把nums中的所有数都*k
        • 这个过程可直接遍历进行，无需再进行堆排序

    class Solution {
public:
    void exchange(const int& i, const int & j, vector<int>& rows)
    {
        int t = rows[i];
        rows[i] = rows[j];
        rows[j] = t;
    }

    int movesToChessboard(vector<vector<int>>& board) {
        int n = board.size();
        vector<int> rows(n, 0);
        int ans = 0;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                rows[i] ^= (board[i][j] << j);
            }
        }
        if (n % 2)
        {
            int cnt = 0;
            for (int i = 1; i < n; i++)
            {
                if ((rows[i] & rows[0]) == 0)
                {
                    cnt++;
                }
                if (cnt % 2)
                {
                    for (int i = 1; i < n; i++)
                    {
                        if ((rows[i] & rows[0]) == 0)
                        {
                            exchange(0, i, rows);
                            ans++;
                        }
                    }
                }
            }
        }
        for (int i = 0; i < n - 1; i++)
        {
            int j;
            for (j = i + 1; j < n; j++)
            {
                if ((rows[i] & rows[j]) == 0)
                {
                    if (j != i + 1)
                    {
                        exchange(i + 1, j, rows);
                        ans++;
                    }
                    break;
                }
            }
            if (j == n)
                return -1;
        }
        vector<int> cols(n, 0);
        for (int j = 0; j < n; j++)
        {
            for (int i = 0; i < n; i++)
            {
                cols[j] ^= (board[i][j] << i);
            }
        }
        if (n % 2)
        {
            int cnt = 0;
            for (int i = 1; i < n; i++)
            {
                if ((cols[i] & cols[0]) == 0)
                {
                    cnt++;
                }
                if (cnt % 2)
                {
                    for (int i = 1; i < n; i++)
                    {
                        if ((cols[i] & cols[0]) == 0)
                        {
                            exchange(0, i, cols);
                            ans++;
                        }
                    }
                }
            }
        }
        for (int i = 0; i < n - 1; i++)
        {
            int j;
            for (j = i + 1; j < n; j++)
            {
                if ((cols[i] & cols[j]) == 0)
                {
                    if (j != i + 1)
                    {
                        exchange(i + 1, j, cols);
                        ans++;
                    }
                    break;
                }
            }
            if (j == n)
                return -1;
        }
        return ans;
    }
};

    class Solution {
public:
    int getMoves(int mask, int count, int n) {
        int ones = __builtin_popcount(mask);
        if (n & 1) {
            /* 如果 n 为奇数，则每一行中 1 与 0 的数目相差为 1，且满足相邻行交替 */
            if (abs(n - 2 * ones) != 1 || abs(n - 2 * count) != 1 ) {
                return -1;
            }
            if (ones == (n >> 1)) {
                /* 偶数位变为 1 的最小交换次数 */
                return n / 2 - __builtin_popcount(mask & 0xAAAAAAAA);
            } else {
                /* 奇数位变为 1 的最小交换次数 */
                return (n + 1) / 2 - __builtin_popcount(mask & 0x55555555);
            }
        } else { 
            /* 如果 n 为偶数，则每一行中 1 与 0 的数目相等，且满足相邻行交替 */
            if (ones != (n >> 1) || count != (n >> 1)) {
                return -1;
            }
            /* 偶数位变为 1 的最小交换次数 */
            int count0 = n / 2 - __builtin_popcount(mask & 0xAAAAAAAA);
            /* 奇数位变为 1 的最小交换次数 */
            int count1 = n / 2 - __builtin_popcount(mask & 0x55555555);  
            return min(count0, count1);
        }
    }

    int movesToChessboard(vector<vector<int>>& board) {
        int n = board.size();
        int rowMask = 0, colMask = 0;        

        /* 检查棋盘的第一行与第一列 */
        for (int i = 0; i < n; i++) {
            rowMask |= (board[0][i] << i);
            colMask |= (board[i][0] << i);
        }
        int reverseRowMask = ((1 << n) - 1) ^ rowMask;
        int reverseColMask = ((1 << n) - 1) ^ colMask;
        int rowCnt = 0, colCnt = 0;
        for (int i = 0; i < n; i++) {
            int currRowMask = 0;
            int currColMask = 0;
            for (int j = 0; j < n; j++) {
                currRowMask |= (board[i][j] << j);
                currColMask |= (board[j][i] << j);
            }
            /* 检测每一行的状态是否合法 */
            if (currRowMask != rowMask && currRowMask != reverseRowMask) {
                return -1;
            } else if (currRowMask == rowMask) {
                /* 记录与第一行相同的行数 */
                rowCnt++;
            }
            /* 检测每一列的状态是否合法 */
            if (currColMask != colMask && currColMask != reverseColMask) {
                return -1;
            } else if (currColMask == colMask) {
                /* 记录与第一列相同的列数 */
                colCnt++;
            }
        }
        int rowMoves = getMoves(rowMask, rowCnt, n);
        int colMoves = getMoves(colMask, colCnt, n);
        return (rowMoves == -1 || colMoves == -1) ? -1 : (rowMoves + colMoves); 
    }
};

    leetcode 683 平方数之和

    题目

    给定一个非负整数 c ，你要判断是否存在两个整数 a 和 b，使得 a2 + b2 = c 。

    示例一

    方法一

    class Solution {
public:
    bool judgeSquareSum(int c) {
        unordered_set<int> little_squares;
        long long base = 0;
        long long square = base * base;
        while (square <= c / 2) {
            little_squares.insert(square);
            if (square + square == c) return true;
            base++;
            square = base * base;
        }
        while (square <= c) {
            if (little_squares.count(c - square)) return true;
            base++;
            square = base * base;
        }
        return false;
    }
};

    class Solution {
public:
    int mySqrt(int x) {
        int l = 0, r = x;
        int ans = -1;
        while (l <= r)
        {
            int mid = l + (r - l) / 2;
            if ((long long)mid * mid <= x)
            {
                ans = mid;
                l = mid + 1;
            }
            else {
                r = mid - 1;
            }
        }
        return ans;
    }
};

    偏序关系partial orderings

    • 偏序集合partially ordered sets
    • 偏序关系partial order 自反、反对称、传递
    • 线序关系：可比(comparable)，不可比(incomparable)
    • 全序集合linearly ordered set：所有元素对之间均可比 (linear order全序关系或线序关系->chain)
    • 良序关系well-ordered 所有非空集合都有一个最小元
    • 乘积偏序product partial order
    • 词典顺序lexicographic 先看第一个元素，第一个元素相同再看第二个元素
    • 哈斯图Hasse Diagrams
    • 极大元maximal element
    • 极小元minimal element
    • 最大元greatest element(单位元unit element) 不一定有
    • 最小元least element(零元 zero element) 不一定有
    • 拓扑排序topological sorting（找极小元，找完去掉，继续找极小元）
    • 最小上界LUB
    • 最大下界GLB
    • 格lattices：所有子集都有LUB和GLB
    • 子格sublattice：如果a和b都在原格中，则a和b的LUB和GLB也都在原格中
    • 有界：如果L有最大元和最小元，then lattice L is said to be bounded
    • 如果L是有限格，那么L是有界的
    • 分配格Distributive lattice：最大上界和最小下界的运算符和分配律 -> nondistributive
    • 补元(complement)：如果a和b的最小上界是最大元，最小下界是最小元，则a和b互为补元

    半群和群

    半群

    • 封闭 closure
    • 二元运算 binary operation
    • 交换律 commutative
    • 结合率 associative
    • 分配律 distributive property
    • 德·摩根定理 De Morgan’s lawns
    • 单位元 Identity for an operation
    • 逆元 Iverse
    • 广群：非空、封闭的
    • 半群：非空、封闭、可结合
    • 单位元、幺元 identity
    • 独异点、含幺半群 monoid
    • 子半群 subsemigroup
    • 同构映射 isomorphism：one to one，f(a * b) = f(a) *’ f(b)
    • 同态映射 homomorphism：everywhere-defined function, f(a * b) = f(a) *’ f(b)
    • products semmigroup 乘积半群
    • quotients of semigroups 商半群
    • 同余关系 congruence relation aRa’ and bRb’ imply (a*b)R(a’*b')
    • 自然同态 natural homomorphism
    • 满同态函数 aRb if and only if f(a) = f(b)
    • 同态基本定理：满同态与商群（自然同态）同构

    群

    • 有幺元identity的群成为独异点
    • 半群+幺元
    • 阿贝尔群Abelian：群+交换律
    • 有限群finite group：阶(order)是元素的数目
    • 含有一个元素的群：1种，两个：1种，三个：V和Z4两种
    • S3：f2,f3,f1分别是三角形绕O点旋转120度，240度和360度；g1,g2,g3分别是让三角形关于I1,I2和I3对称
    • 子群subgroup
    • 平凡子群trivial subgroups：本身和只含幺元的集合
    • S3不是阿贝尔群，Z6是阿贝尔群，二者不同构
    • Zm*Zn与Zmn同构当且仅当m与n的最大公因数为1
    • 左陪集left coset aH
    • 右陪集right coset Ha
    • 正规子群normal aH = Ha（集合相等，不一定要每个元素都相等）
    • 阿贝尔群的任意一个子群都是正规子群
    • 核kernel：同态映射中映射到幺元的元素
