            return;

        st.push(root);
        if (root == p || root == q)
        {
            if (st1.empty())
            {
                st1 = st;
            }
            else
            {
                st2 = st;
                return;
            }
        }
        dfs(root->left, p, q);
        dfs(root->right, p, q);
        st.pop();
    }

        dfs(root, p, q);

            for (int i = 0; i < cnt; i++)
            {
                st1.pop();
            }
        }

            {
                st2.pop();

            st1.pop();
            st2.pop();
        }
        // cout << st1.top()->val << ", " << st2.top()->val << endl;
        if (!st1.empty())
        {
            return st1.top();
        }
        return nullptr;
    }

    /**

    int dfs(TreeNode* root, long long curr, int targetSum)
    {
        if (!root)
        {
            return 0;
        }

        int ret = 0;
        curr += root->val;
        if (prefix.count(curr - targetSum))
        {
            ret = prefix[curr - targetSum];
        }
        prefix[curr]++;
        ret += dfs(root->left, curr, targetSum);
        ret += dfs(root->right, curr, targetSum);
        prefix[curr]--;

        return ret;

        prefix[0] = 1;
        return dfs(root, 0, targetSum);
    }
private:
    unordered_map<long long, int> prefix;

    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* changetoBST(vector<int>& nums, const int& l, const int& r)
    {
        TreeNode* node = nullptr;
        if (l == r)
        {
            node = new TreeNode(nums[l]);
        }
        else if (r - l == 1)
        {
            node = new TreeNode(nums[l]);
            TreeNode* another_node = new TreeNode(nums[r]);
            node->right = another_node;
        }
        else
        {
            int mid = (l + r) / 2;
            node = new TreeNode(nums[mid]);
            node->left = changetoBST(nums, l, mid - 1);
            node->right = changetoBST(nums, mid + 1, r);
        }
        return node;
    }
    
    TreeNode* sortedArrayToBST(vector<int>& nums) {
        return changetoBST(nums, 0, nums.size() - 1);
    }
};
