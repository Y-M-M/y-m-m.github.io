leetcode 438 找到字符串中所有字母异位词

题目

给定两个字符串 s 和 p，找到 s 中所有 p 的 异位词 的子串，返回这些子串的起始索引。不考虑答案输出的顺序。

示例

示例一

输入: s = “cbaebabacd”, p = “abc”
输出: [0,6]
解释:

  • 起始索引等于 0 的子串是 “cba”, 它是 “abc” 的异位词。
  • 起始索引等于 6 的子串是 “bac”, 它是 “abc” 的异位词。

示例二

输入: s = “abab”, p = “ab”
输出: [0,1,2]
解释:

  • 起始索引等于 0 的子串是 “ab”, 它是 “ab” 的异位词。
  • 起始索引等于 1 的子串是 “ba”, 它是 “ab” 的异位词。
  • 起始索引等于 2 的子串是 “ab”, 它是 “ab” 的异位词。

提示

  • 1 <= s.length, p.length <= 3 * 104
  • s 和 p 仅包含小写字母

题解

方法一

  • 分析
    • 用unordered_map<char, int> neednums 记录p中各字母的个数
    • 用unordered_map<char, int> nownums 记录子串中各字母的个数
    • 用重载的运算符==判断两个unordered_map是否相等
    • 注意：second为0的元素要删掉
class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        unordered_map<char, int> neednums;
        unordered_map<char, int> nownums;
        int modelen = p.length();
        vector<int> ans;
        for (char c: p)
        {
            neednums[c]++;
        }
        int border = s.length() - p.length();
        
        for (int i = 0; i < p.length(); i++)
        {
            nownums[s[i]]++;
        }
        if (nownums == neednums)
        {
            ans.push_back(0);
        }
        for (int i = 1; i <= border; i++)
        {
            nownums[s[i - 1]]--;
            if (nownums[s[i - 1]] == 0)
            {
                nownums.erase(s[i - 1]);
            }
            nownums[s[i + modelen - 1]]++;
            if (nownums[s[i + modelen - 1]] == 0)
            {
                nownums.erase(s[i + modelen - 1]);
            }
            if (nownums == neednums)
            {
                ans.push_back(i);
            }
        }
        return ans;
    }
};

方法二

  • 分析
    • 用vector differnum(26);记录当前子串与p的字母数量的差异
    • 用int differ记录有差异的字母的个数(注意 不是记录总的字母数量之差)
    • 每次移动滑动窗口的时候更新differnum，并更新differ
    • differ为0的时候，该子串即为符合要求的子串
    • 重点是各字符的数量差异而不是数量本身
class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        vector<int> differnum(26, 0);
        int Slen = s.length();
        int Plen = p.length();
        vector<int> ans;
        if (Slen < Plen)
        {
            return ans;
        }
        for (int i = 0; i < Plen; i++)
        {
            differnum[s[i] - 'a']++;
            differnum[p[i] - 'a']--;
        }
        int differ = 0;
        for (int i = 0; i < 26; i++)
        {
            if (differnum[i])
                differ++;
        }
        if (!differ)
        {
            ans.push_back(0);
        }
        int len = Slen - Plen;
        for (int i = 1; i <= len; i++)
        {
            differnum[s[i - 1] - 'a']--;
            if (!differnum[s[i - 1] - 'a'])
            {
                differ--;
            }
            else if (differnum[s[i - 1] - 'a'] == -1)
            {
                differ++;
            }
            differnum[s[i + Plen - 1] - 'a']++;
            if (!differnum[s[i + Plen - 1] - 'a'])
            {
                differ--;
            }
            else if (differnum[s[i + Plen - 1] - 'a'] == 1)
            {
                differ++;
            }
            if (!differ)
                ans.push_back(i);
        }
        return ans;
    }
};
