    class Solution {
public:
    bool binaryfind(vector<vector<int>>& matrix, const int& target, const int& bias)
    {
        int l = 0, r = mindim - 1;
        while (l <= r)
        {
            int mid = (l + r) / 2;
            int nownum;
            if (m > n)
            {
                nownum = matrix[mid + bias][mid];
            }
            else if (m < n)
            {
                nownum = matrix[mid][mid + bias];
            }
            else
            {
                nownum = matrix[mid][mid];
            }
            if (nownum > target)
            {
                r = mid - 1;
            }
            else
            {
                l = mid + 1;
            }
        }
        if (r < 0)
        {
            return false;
        }
        int lastr = mindim - 1;
        int lastc = mindim - 1;
        int firstr = r;
        int firstc = r;
        int secondr = l;
        int secondc = l;
        if (m > n)
        {
            lastr += bias;
            firstr += bias;
            secondr += bias;
        }
        else if (m < n)
        {
            lastc += bias;
            firstc += bias;
            secondc += bias;
        }

        if (l >= mindim && matrix[lastr][lastc] != target)
        {
            return false;
        }
        if (matrix[firstr][firstc] == target)
        {
            return true;
        }
        for (int i = firstr + 1; i < m; i++)
        {
            if (matrix[i][firstc] == target)
            {
                return true;
            }
            if (matrix[i][firstc] > matrix[secondr][secondc])
            {
                break;
            }
        }
        for (int j = firstc + 1; j < n; j++)
        {
            if (matrix[firstr][j] == target)
            {
                return true;
            }
            if (matrix[firstr][j] > matrix[secondr][secondc])
            {
                break;
            }
        }
        for (int i = firstr - 1; i >= 0; i--)
        {
            if (matrix[i][secondc] == target)
            {
                return true;
            }
            if (matrix[i][secondc] < matrix[firstr][firstc])
            {
                break;
            }
        }
        for (int j = firstc - 1; j >= 0; j--)
        {
            if (matrix[secondr][j] == target)
            {
                return true;
            }
            if (matrix[secondr][j] < matrix[firstr][firstc])
            {
                break;
            }
        }
        if (matrix[firstr + 1][firstc + 1] == target)
        {
            return true;
        }
        return false;
    }

    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        m = matrix.size();
        n = matrix[0].size();
        if (m > n)
        {
            shape = 1;
        }
        else if (m < n)
        {
            shape = -1;
        }
        else
        {
            shape = 0;
        }
        mindim = min(m, n);
        int cnt = max(m, n) - mindim + 1;
        for (int i = 0; i < cnt; i++)
        {
            if (binaryfind(matrix, target, i))
            {
                return true;
            }
        }
        return false;
    }
private:
    int m;
    int n;
    int shape;
    int mindim;
};

    int uniquePathsWithObstacles(vector<vector<int>>& obstacleGrid) {
        int m = obstacleGrid.size();
        int n = obstacleGrid[0].size();
        vector<vector<int>> roads(m, vector<int> (n, 0));
        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (obstacleGrid[i][j] > 0)
                {
                    roads[i][j] = 0;
                }
                else
                {
                    if (!i && !j)
                    {
                        roads[i][j] = 1;
                    }
                    else if (!i)

                    }
                    else if (!j)

                        roads[i][j] = roads[i - 1][j];
                    }

                        roads[i][j] = roads[i - 1][j] + roads[i][j - 1];
                    }
                }
            }
        }
        return roads[m - 1][n - 1];
    }
