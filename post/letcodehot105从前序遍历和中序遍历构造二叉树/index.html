leetcodehot105 从前序与中序遍历构造二叉树

题目

给定两个整数数组 preorder 和 inorder ，其中 preorder 是二叉树的先序遍历， inorder 是同一棵树的中序遍历，请构造二叉树并返回其根节点。

示例

示例一


输入: preorder = [3,9,20,15,7], inorder = [9,3,15,20,7]
输出: [3,9,20,null,null,15,7]

示例二

输入: preorder = [-1], inorder = [-1]
输出: [-1]

提示

  • 1 <= preorder.length <= 3000
  • inorder.length == preorder.length
  • -3000 <= preorder[i], inorder[i] <= 3000
  • preorder 和 inorder 均 无重复 元素
  • inorder 均出现在 preorder
  • preorder 保证 为二叉树的前序遍历序列
  • inorder 保证 为二叉树的中序遍历序列

题解

方法一

  • 分析
    • 递归的构造
    • 利用先序遍历确认当前子树的根节点
    • 利用中序遍历确定左右子树在先序遍历序列中的范围
  • 提交结果
    • 通过
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int findendl(vector<int> &preorder, int rootpos, int beginpos)
    {
        int i = beginpos + 1;
        while (i < preorder.size() && inpos[preorder[i]] < rootpos)
        {
            i++;
        }
        return i - 1;
    }

    TreeNode* buidlsubTree(vector<int> &preorder, vector<int>& inorder, int l1, int r1, int l2, int r2)
    {
        if (l1 > r1 || l2 > r2)
        {
            return nullptr;
        }
        int rootval = preorder[l1];
        TreeNode* root = new TreeNode(rootval);
        if (r1 == l1)
        {
            return root;
        }
        if (inpos[rootval] == l2)
        {
            root->right = buidlsubTree(preorder, inorder, l1 + 1, r1, l2 + 1, r2);
            return root;
        }
        int preendl = findendl(preorder, inpos[rootval], l1);
        // cout << "preendl = " << preendl << endl;
        root->left = buidlsubTree(preorder, inorder, l1 + 1, preendl, l2, inpos[rootval] - 1);
        root->right = buidlsubTree(preorder, inorder, preendl + 1, r1,  inpos[rootval] + 1, r2);
        return root;
    }

    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        TreeNode* root = nullptr;
        if (preorder.size() == 0)
        {
            return root;
        }
        for (int i = 0; i < preorder.size(); i++)
        {
            prepos[preorder[i]] = i;
        }
        for (int i = 0; i < inorder.size(); i++)
        {
            inpos[inorder[i]] = i;
        }
        root = buidlsubTree(preorder, inorder, 0, preorder.size() - 1, 0, inorder.size() - 1);
        return root;
    }
private:
    unordered_map<int, int> prepos;
    unordered_map<int, int> inpos;
};

  • 分析
    • 在方法一的基础上改进
    • 在preorder中确定左子树最后一个节点的位置，可以无需挨个判断，只需要在inorder中确定左子树的元素个数，即可直接确定

方法三

  • 分析
    • 迭代
    • 利用栈记录祖先节点
  • 提交结果
    • 通过
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        TreeNode* root = nullptr;
        if (preorder.size() == 0)
        {
            return root;
        }
        for (int i = 0; i < inorder.size(); i++)
        {
            inpos[inorder[i]] = i;
        }
        root = new TreeNode(preorder[0]);
        stack<TreeNode*> st;
        st.push(root);
        for (int i = 1; i < preorder.size(); i++)
        {
            int preval = st.top()->val;
            int nowval = preorder[i];
            if (inpos[nowval] < inpos[preval])
            {
                TreeNode* node = new TreeNode(nowval);
                st.top()->left = node;
                st.push(node);
            }
            else
            {
                TreeNode* prenode = nullptr;
                while (!st.empty() && inpos[nowval] > inpos[preval])
                {
                    prenode = st.top();
                    st.pop();      
                    if (st.empty())
                    {
                        break;
                    }
                    preval = st.top()->val;
                }
                TreeNode* node = new TreeNode(nowval);
                st.push(node);
                prenode->right = node;
            }
        }
        return root;
    }
private:
    unordered_map<int, int> inpos;
};
