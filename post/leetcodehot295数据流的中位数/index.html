leetcode hot100 295 数据流的中位数

题目

中位数是有序整数列表中的中间值。如果列表的大小是偶数，则没有中间值，中位数是两个中间值的平均值。
例如 arr = [2,3,4] 的中位数是 3 。
例如 arr = [2,3] 的中位数是 (2 + 3) / 2 = 2.5 。
实现 MedianFinder 类:
MedianFinder() 初始化 MedianFinder 对象。
void addNum(int num) 将数据流中的整数 num 添加到数据结构中。
double findMedian() 返回到目前为止所有元素的中位数。与实际答案相差 10-5 以内的答案将被接受。

示例

示例一

输入
[“MedianFinder”, “addNum”, “addNum”, “findMedian”, “addNum”, “findMedian”]
[[], [1], [2], [], [3], []]
输出
[null, null, null, 1.5, null, 2.0]
解释
MedianFinder medianFinder = new MedianFinder();
medianFinder.addNum(1); // arr = [1]
medianFinder.addNum(2); // arr = [1, 2]
medianFinder.findMedian(); // 返回 1.5 ((1 + 2) / 2)
medianFinder.addNum(3); // arr[1, 2, 3]
medianFinder.findMedian(); // return 2.0

提示

  • -105 <= num <= 105
  • 在调用 findMedian 之前，数据结构中至少有一个元素
  • 最多 5 * 104 次调用 addNum 和 findMedian

题解

方法一

  • 分析
    • 维护左右堆
    • 保持左右堆数目相等，或者相差为1即可
  • 提交结果
    • 通过
class MedianFinder {
private:
    vector<int> leftHeap;  // 最大堆
    vector<int> rightHeap; // 最小堆

    void heapifyUp(vector<int>& heap, bool maxHeap) {
        int i = heap.size() - 1;
        while (i > 0) {
            int parent = (i - 1) / 2;
            if ((maxHeap && heap[i] > heap[parent]) ||
                (!maxHeap && heap[i] < heap[parent])) {
                swap(heap[i], heap[parent]);
                i = parent;
            } else {
                break;
            }
        }
    }

    void heapifyDown(vector<int>& heap, int i, bool maxHeap) {
        int n = heap.size();
        while (2 * i + 1 < n) {
            int l = 2 * i + 1, r = 2 * i + 2, best = i;
            if (l < n && ((maxHeap && heap[l] > heap[best]) || (!maxHeap && heap[l] < heap[best])))
                best = l;
            if (r < n && ((maxHeap && heap[r] > heap[best]) || (!maxHeap && heap[r] < heap[best])))
                best = r;
            if (best == i) break;
            swap(heap[i], heap[best]);
            i = best;
        }
    }

    void pushHeap(vector<int>& heap, int val, bool maxHeap) {
        heap.push_back(val);
        heapifyUp(heap, maxHeap);
    }

    int popHeap(vector<int>& heap, bool maxHeap) {
        int top = heap[0];
        heap[0] = heap.back();
        heap.pop_back();
        heapifyDown(heap, 0, maxHeap);
        return top;
    }

public:
    void addNum(int num) {
        if (leftHeap.empty() || num <= leftHeap[0]) {
            pushHeap(leftHeap, num, true); // maxHeap
        } else {
            pushHeap(rightHeap, num, false); // minHeap
        }

        if (leftHeap.size() > rightHeap.size() + 1) {
            int moved = popHeap(leftHeap, true);
            pushHeap(rightHeap, moved, false);
        } else if (rightHeap.size() > leftHeap.size()) {
            int moved = popHeap(rightHeap, false);
            pushHeap(leftHeap, moved, true);
        }
    }

    double findMedian() {
        if (leftHeap.size() == rightHeap.size()) {
            return (leftHeap[0] + rightHeap[0]) / 2.0;
        } else {
            return leftHeap[0];
        }
    }
};
