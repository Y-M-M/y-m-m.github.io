    class Solution {

    void findans(vector<vector<string>>&ans, vector<int>&quene, unordered_set<int>&dig1, unordered_set<int>& dig2, unordered_set<int>&col, const int& row, const int& pos, const int& n)
    {
        quene[row] = pos;
        col.insert(pos);
        if (row == n - 1)
        {
            ans.push_back(buildsolution(quene, n));
        }
        else
        {
            for (int i = 0; i < n; i++)
            {
                if (i == pos) continue;
                if (col.find(i) != col.end()) continue;
                int dignal1 = row + 1 - i;
                int dignal2 = row + 1 + i;
                if (dig1.find(dignal1) != dig1.end() || dig2.find(dignal2) != dig2.end())
                {
                    continue;
                }
                dig1.insert(dignal1);
                dig2.insert(dignal2);
                findans(ans, quene, dig1, dig2, col, row+1, i, n);
                dig1.erase(dignal1);
                dig2.erase(dignal2);
            }
        }
        
        col.erase(pos);
        quene[row] = -1;
    }

    vector<string> buildsolution(const vector<int>& quene, const int& n)
    {
        vector<string> asolution;
        for (int i = 0; i < n; i++)
        {
            string arow = string(n, '.');
            arow[quene[i]] = 'Q';
            asolution.push_back(arow);
        }
        return asolution;
    }

    vector<vector<string>> solveNQueens(int n) {
        vector<vector<string>> ans;
        unordered_set<int> dig1;
        unordered_set<int> dig2;
        unordered_set<int> col;
        vector<int> quene(n, -1);
        for (int i = 0; i < n; i++)
        {
            dig1.insert(0 - i);
            dig2.insert(i);
            findans(ans, quene, dig1, dig2, col, 0, i, n);
            dig1.erase(0 - i);
            dig2.erase(i);
        }
        return ans;
    }

        else

            int available = ((1 << n) - 1) & (~(col | dig1 | dig2));

    class Solution {
public:
    bool valid(const string& str)
    {
        int balance = 0;
        for (char c : str)
        {
            if (c == '(')
            {
                ++balance;
            }
            else
            {
                --balance;
            }
            if (balance < 0)
            {
                return false;
            }
        }
        return balance == 0;
    }

    void generate_all(string& current, int n, vector<string>& result)
    {
        if (n == current.size())
        {
            if (valid(current))
            {
                result.push_back(current);
            }
            return;
        }
        current += '(';
        generate_all(current, n, result);
        current.pop_back();
        current += ')';
        generate_all(current, n, result);
        current.pop_back();
    }

    vector<string> generateParenthesis(int n) {
        vector<string> result;
        string current;
        generate_all(current, n * 2, result);
        return result;
    }
};

    class Solution {
public:
    void backtrack(vector<string>& ans, string& cur, int open, int close, int n)
    {
        if (cur.size() == n * 2)
        {
            ans.push_back(cur);
            return;
        }

        if (open < n)
        {
            cur.push_back('(');
            backtrack(ans, cur, open + 1, close, n);
            cur.pop_back();
        }

        if (close < open)
        {
            cur.push_back(')');
            backtrack(ans, cur, open, close + 1, n);
            cur.pop_back();
        }
    }

    vector<string> generateParenthesis(int n) {
        vector<string> result;
        string current;
        backtrack(result, current, 0, 0, n);
        return result;
    }
};

    class Solution {
public:
    shared_ptr<vector<string>> generate(int n)
    {
        if (cache[n] != nullptr)
        {
            return cache[n];
        }
        if (!n)
        {
            cache[0] = shared_ptr<vector<string>> (new vector<string> {""});
        }
        else
        {
            auto result = shared_ptr<vector<string>> (new vector<string>);
            for (int i = 0; i != n; i++)
            {
                auto lefts = generate(i);
                auto rights = generate(n - i - 1);
                for (const string& left : *lefts)
                {
                    for (const string& right : *rights)
                    {
                        result->push_back("(" + left + ")" + right);
                    }
                }
            }
            cache[n] = result;
        }
        return cache[n];
    }

    vector<string> generateParenthesis(int n) {
        return *generate(n);
    }
private:
    shared_ptr<vector<string>> cache[100] = {nullptr};
};
