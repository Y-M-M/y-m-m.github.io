    class Solution {
public:
    int takeCharacters(string s, int k) {
        int len = s.length();
        vector<int> cnt(3);
        for (int i = 0; i < len; i++) {
            cnt[s[i] - 'a']++;
        }
        int min_len = len;
        if (cnt[0] < k || cnt[1] < k || cnt[2] < k)
            return -1;
        if (k == 0) return 0;
        int l = 0;
        for (int r = 0; r < len; r++) {
            // cout << "r = " << r << endl;
            cnt[s[r] - 'a']--;
            // cout << cnt[s[r] - 'a'] << endl;
            while ((cnt[0] < k || cnt[1] < k || cnt[2] < k) && l <= r) {
                // cout << "l = " << l << endl;
                cnt[s[l] - 'a']++;
                l++;
            }
            if (cnt[0] >= k && cnt[1] >= k && cnt[2] >= k) {
                // cout << "l = " << l << ", r = " << r << endl;
                min_len = min(min_len, len-r+l-1);
            }
        } 
        return min_len;
    }
};
