leetcode hot100 240 搜索二维矩阵二

题目

编写一个高效的算法来搜索 m x n 矩阵 matrix 中的一个目标值 target 。该矩阵具有以下特性：

  • 每行的元素从左到右升序排列。
  • 每列的元素从上到下升序排列。

示例

示例一

输入：matrix = [[1,4,7,11,15],[2,5,8,12,19],[3,6,9,16,22],[10,13,14,17,24],[18,21,23,26,30]], target = 5
输出：true

示例二

输入：matrix = [[1,4,7,11,15],[2,5,8,12,19],[3,6,9,16,22],[10,13,14,17,24],[18,21,23,26,30]], target = 20 输出：false

提示

  • m == matrix.length
  • n == matrix[i].length
  • 1 <= n, m <= 300
  • -109 <= matrix[i][j] <= 109
  • 每行的所有元素从左到右升序排列
  • 每列的所有元素从上到下升序排列
  • -109 <= target <= 109

题解

方法一

  • 分析
    • 沿矩形对称轴二分查找，确定与target相邻的两个数，再进一步确定target是否存在
  • 提交结果
    • 125/130个通过的测试样例
    • 解答错误
  • 存在的问题
    • 如果存在i1, j1; i2, j2
    • i1 > i2
    • j1 < j2
    • 不能确定两者的大小关系
class Solution {
public:
    bool binaryfind(vector<vector<int>>& matrix, const int& target, const int& bias)
    {
        int l = 0, r = mindim - 1;
        while (l <= r)
        {
            int mid = (l + r) / 2;
            int nownum;
            if (m > n)
            {
                nownum = matrix[mid + bias][mid];
            }
            else if (m < n)
            {
                nownum = matrix[mid][mid + bias];
            }
            else
            {
                nownum = matrix[mid][mid];
            }
            if (nownum > target)
            {
                r = mid - 1;
            }
            else
            {
                l = mid + 1;
            }
        }
        if (r < 0)
        {
            return false;
        }
        int lastr = mindim - 1;
        int lastc = mindim - 1;
        int firstr = r;
        int firstc = r;
        int secondr = l;
        int secondc = l;
        if (m > n)
        {
            lastr += bias;
            firstr += bias;
            secondr += bias;
        }
        else if (m < n)
        {
            lastc += bias;
            firstc += bias;
            secondc += bias;
        }

        if (l >= mindim && matrix[lastr][lastc] != target)
        {
            return false;
        }
        if (matrix[firstr][firstc] == target)
        {
            return true;
        }
        for (int i = firstr + 1; i < m; i++)
        {
            if (matrix[i][firstc] == target)
            {
                return true;
            }
            if (matrix[i][firstc] > matrix[secondr][secondc])
            {
                break;
            }
        }
        for (int j = firstc + 1; j < n; j++)
        {
            if (matrix[firstr][j] == target)
            {
                return true;
            }
            if (matrix[firstr][j] > matrix[secondr][secondc])
            {
                break;
            }
        }
        for (int i = firstr - 1; i >= 0; i--)
        {
            if (matrix[i][secondc] == target)
            {
                return true;
            }
            if (matrix[i][secondc] < matrix[firstr][firstc])
            {
                break;
            }
        }
        for (int j = firstc - 1; j >= 0; j--)
        {
            if (matrix[secondr][j] == target)
            {
                return true;
            }
            if (matrix[secondr][j] < matrix[firstr][firstc])
            {
                break;
            }
        }
        if (matrix[firstr + 1][firstc + 1] == target)
        {
            return true;
        }
        return false;
    }

    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        m = matrix.size();
        n = matrix[0].size();
        if (m > n)
        {
            shape = 1;
        }
        else if (m < n)
        {
            shape = -1;
        }
        else
        {
            shape = 0;
        }
        mindim = min(m, n);
        int cnt = max(m, n) - mindim + 1;
        for (int i = 0; i < cnt; i++)
        {
            if (binaryfind(matrix, target, i))
            {
                return true;
            }
        }
        return false;
    }
private:
    int m;
    int n;
    int shape;
    int mindim;
};

方法二

  • 分析
    • 通过对当前矩阵的第一行、第一列进行二分查找，缩小矩阵的范围，舍弃第一行、第一列、比target大的行、比target大的列
  • 提交结果
    • 通过
class Solution {
public:
    bool subfind(vector<vector<int>>& matrix, int target, int beginr, int beginc, int endr, int endc)
    {
        // cout << beginr << ", " << beginc << ", " << endr << ", " << endc << endl;
        if (beginr > endr || beginc > endc)
        {
            return false;
        }
        int rl = beginr, rr = endr;
        while (rl <= rr)
        {
            int mid = (rl + rr) / 2;
            if (matrix[mid][beginc] > target)
            {
                rr = mid - 1;
            } 
            else if (matrix[mid][beginc] < target)
            {
                rl = mid + 1;
            }
            else
            {
                return true;
            }
        }
        if (rr < beginr)
        {
            return false;
        }
        int newendr = rr;
        int ll = beginc, lr = endc;
        while (ll <= lr)
        {
            int mid = (ll + lr) / 2;
            if (matrix[beginr][mid] > target)
            {
                lr = mid - 1;
            }
            else if (matrix[beginr][mid] < target)
            {
                ll = mid + 1;
            }
            else
            {
                return true;
            }
        }
        if (lr < beginc)
        {
            return false;
        }
        // cout << "lr = " << lr << endl;
        int newendc = lr;
        return subfind(matrix, target, beginr + 1, beginc + 1, newendr, newendc);
    }
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        return subfind(matrix, target, 0, 0, matrix.size() - 1, matrix[0].size() - 1);
    }
};

class Solution {

        int m = matrix.size();
        int n = matrix[0].size();
        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (matrix[i][j] == target)
                {
                    return true;
                }
            }
        }
        return false;
    }
};

class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        for (const auto& row: matrix) {
            auto it = lower_bound(row.begin(), row.end(), target);
            if (it != row.end() && *it == target)
            {
                return true;
            }
        }
        return false;
    }
};

方法五

  • 分析
    • 一个元素(x, y)可以确定
      • 如果matrix(x, y) > target，则第其以下、以右的元素均 > target
      • 如果matrix(x, y) < target, 则其以左、以上的元素均 < target
    • 同时考虑这两个因素，可知左下角和右上角的元素可以很容易的通过简单比较而让矩阵缩小
  • 提交结果
    • 通过
class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        int m = matrix.size();
        int n = matrix[0].size();
        int x = 0, y = n - 1;
        while (x <= m - 1 && y >= 0)
        {
            if (matrix[x][y] > target)
            {
                y--;
            }
            else if (matrix[x][y] < target)
            {
                x++;
            }
            else
            {
                return true;
            }
        }
        return false;
    }
};
