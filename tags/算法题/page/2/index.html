    leetcode 3240 最少翻转次数使二进制矩阵回文二

    题目

    给你一个 m x n 的二进制矩阵 grid 。
    如果矩阵中一行或者一列从前往后与从后往前读是一样的，那么我们称这一行或者这一列是 回文 的。
    你可以将 grid 中任意格子的值 翻转 ，也就是将格子里的值从 0 变成 1 ，或者从 1 变成 0 。
    请你返回 最少 翻转次数，使得矩阵中 所有 行和列都是 回文的 ，且矩阵中 1 的数目可以被 4 整除 。

    示例

    示例一

    输入：grid = [[1,0,0],[0,1,0],[0,0,1]]
    输出：3

    示例二

    输入：grid = [[0,1],[0,1],[0,0]]
    输出：2

    示例三

    输入：grid = [[1],[1]]
    输出：2

    提示

    • m == grid.length
    • n == grid[i].length
    • 1 <= m * n <= $2 * 10^5$
    • 0 <= grid[i][j] <= 1

    题解

    方法一

    class Solution {
public:
    int minFlips(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        int ans = 0;
        int num1 = 0; // 不会天然形成4的倍数的1的个数(默认所有修改均将1修改为0)
        if (m % 2)
        {
            int l = 0, r = n - 1;
            while (l < r)
            {
                if (grid[m / 2][l] != grid[m / 2][r])
                {
                    ans++;
                }
                else if (grid[m / 2][l] == 1)
                    num1 += 2;
                l++;
                r--;
            }
            if (l == r) num1 += grid[m / 2][l];
        }
        if (n % 2)
        {
            int top = 0, bottom = m - 1;
            while (top < bottom)
            {
                if (grid[top][n / 2] != grid[bottom][n / 2])
                {
                    ans++;
                }
                else if (grid[top][n / 2] == 1) 
                    num1 += 2;
                top++;
                bottom--;
            }
        }
        if (num1 % 4 != 0)
        {
            int flag = 0;
            if (m % 2 && n % 2)
                flag = 1;
            if (num1 % 4 == 1)
            {
                int add = 0;
                if (ans == 0)
                {
                    if (flag)
                    {
                        if (grid[m / 2][n / 2]) add++;
                        else add += 3;
                    }
                }
                else if (ans == 1)
                {
                    add++;
                }
                else if (ans >= 2)
                {
                    add++;
                }
                ans += add;
            }
            else if (num1 % 4 == 2)
            {
                int add = 0;
                if (ans == 0)
                {
                    add += 2;
                }
                ans += add;
            }
            else if (num1 % 4 == 3)
            {
                int add = 0;
                if (ans == 0)
                {
                    if (flag && !grid[m / 2][n / 2]) add++;
                    else add += 3;
                }
                else 
                {
                    if (flag) add++;
                    else add += 3;
                }
                ans += add;
            }
        }
        for (int i = 0; i < m / 2; i++)
        {
            for (int j = 0; j < n / 2; j++)
            {
                int difference = 0;
                if (grid[i][j] ^ grid[i][n - j - 1]) difference++;
                if (grid[i][j] ^ grid[m - i - 1][j]) difference++;
                if (grid[i][j] ^ grid[m - i - 1][n - j - 1]) difference++;
                if (difference == 1) ans += 1;
                else if (difference == 2) ans += 2;
                else if (difference == 3) ans += 1;
            }
        }
        return ans;
    }
};

        • 对于行和列均为回文的矩阵来说，可分为两部分

    leetcode 3249 统计好节点的数目

    题目

    现有一棵 无向 树，树中包含 n 个节点，按从 0 到 n - 1 标记。树的根节点是节点 0 。给你一个长度为 n - 1 的二维整数数组 edges，其中 edges[i] = [ai, bi] 表示树中节点 ai 与节点 bi 之间存在一条边。 如果一个节点的所有子节点为根的 子树 包含的节点数相同，则认为该节点是一个 好节点。 返回给定树中 好节点 的数量。 子树 指的是一个节点以及它所有后代节点构成的一棵树。

    示例

    示例一

    输入：edges = [[0,1],[0,2],[1,3],[1,4],[2,5],[2,6]] 输出：7 说明

    示例二

    输入：edges = [[0,1],[1,2],[2,3],[3,4],[0,5],[1,6],[2,7],[3,8]] 输出：6 说明：

    示例三

    输入：edges = [[0,1],[1,2],[1,3],[1,4],[0,5],[5,6],[6,7],[7,8],[0,9],[9,10],[9,12],[10,11]] 输出：12 解释：

    提示

    • 2 <= n <= $10^5$
    • edges.length == n - 1
    • edges[i].length == 2
    • 0 <= $a_i$, $b_i$ < n
    • 输入确保 edges 总表示一棵有效的树。

    题解

    方法一

    class Solution {
public:
    int dfs (const int& root, const vector<vector<int>>& tree, vector<int>& sum) // 深度优先搜索，统计每棵子树的节点数
    {
        int cnt = 1;
        for (int i = 0; i < tree[root].size(); i++)
        {
            cnt += dfs(tree[root][i], tree, sum);
        }
        sum[root] = cnt;
        return cnt;
    }

    void buildit(vector<vector<int>>& tree, const int& root) // 将连接关系缩减为父子关系
    {
        for (int i = 0; i < tree[root].size(); i++)
        {
            int son = tree[root][i];
            for (int j = 0; j < tree[son].size(); j++)
            {
                if (tree[son][j] == root)
                    tree[son].erase(tree[son].begin() + j);
            }
            buildit(tree, son);
        }
    }

    int countGoodNodes(vector<vector<int>>& edges) {
        int n = edges.size();
        vector<vector<int>> tree(n + 1);
        for (int i = 0; i < n ; i++) // 统计连接关系
        {
            int fa = edges[i][0], son = edges[i][1];
            tree[fa].push_back(son);
            tree[son].push_back(fa);
        }
        buildit(tree, 0); // 将连接关系缩减为父子关系
        
        int ans = 0;
        vector<int> sum(n + 1);
        dfs(0, tree, sum);
        for (int i = 0; i <= n; i++) // 遍历所有节点，判断是不是好节点并统计数目
        {
            int isgood = 1;
            for (int j = 1; j < tree[i].size(); j++)
            {
                if (sum[tree[i][j]] != sum[tree[i][0]])
                {
                    isgood = 0;
                    break;
                }
            }
            ans += isgood;
        }
        return ans;
    }
};

      • 时间复杂度O($n^2$)
      • 空间复杂度O(n)
      • 思路
        • 深度优先搜索统计子树的节点数
        • 因为所给的连接关系不会隐含父子关系，需要从根节点开始自己判断，所以先建立连接关系再缩减为父子关系

    方法二

    class Solution {
public:
    int dfs (const int& root, const vector<vector<int>>& tree, vector<int>& sum, const int& fa, int& ans) // 深度优先搜索，统计每棵子树的节点数
    {
        int cnt = 1;
        int num = 0;
        int isgood = 1;
        for (int i = 0; i < tree[root].size(); i++)
        {
            if (tree[root][i] != fa)
            {
                int add_num = dfs(tree[root][i], tree, sum, root, ans);
                if (!num) num = add_num;
                else {
                    if (add_num != num)
                        isgood = 0;
                }
                cnt += add_num;
            } 
        }
        if (isgood) ans++;
        sum[root] = cnt;
        return cnt;
    }

    int countGoodNodes(vector<vector<int>>& edges) {
        int n = edges.size();
        vector<vector<int>> tree(n + 1);
        for (int i = 0; i < n ; i++) // 统计连接关系
        {
            int fa = edges[i][0], son = edges[i][1];
            tree[fa].push_back(son);
            tree[son].push_back(fa);
        }
        
        int ans = 0;
        vector<int> sum(n + 1);
        dfs(0, tree, sum, -1, ans);
        return ans;
    }
};
    • 提交结果
      • 通过
    • 分析
      • 在方法一的基础上改进
        • dfs的同时统计好节点的数目
        • 不手动删除冗余的连接关系，只要在dfs的时候加个判断，不要访问父节点就可以了
      • 时间复杂度O(n)
      • 空间复杂度O(n)

    题目

    给你一个仅由整数组成的有序数组，其中每个元素都会出现两次，唯有一个数只会出现一次。 请你找出并返回只出现一次的那个数。 你设计的解决方案必须满足 O(log n) 时间复杂度和 O(1) 空间复杂度。

    示例

    示例一

    输入: nums = [1,1,2,3,3,4,4,8,8] 输出: 2

    示例二

    输入: nums = [3,3,7,7,10,11,11] 输出: 10

    提示

    • 1 <= nums.length <= 105
    • 0 <= nums[i] <= 105

    题解

    方法一

    class Solution {
public:
    int singleNonDuplicate(vector<int>& nums) {
        int l = 0, r = nums.size() - 1;
        while (l < r)
        {
            int mid = (l + r) / 2;
            if (nums[mid - 1] == nums[mid])
            {
                if (mid % 2 == 0)
                    r = mid - 2;
                else 
                    l = mid + 1;
                continue;
            } 
            else if (nums[mid + 1] == nums[mid])
            {
                if (mid % 2 == 0)
                    l = mid + 2;
                else 
                    r = mid - 1;
                continue;
            }
            else 
                return nums[mid];
        }
        return nums[l];
    }
};
    • 提交结果
      • 通过
    • 分析
      • 整个数组有序排列，考虑二分查找
      • 每次只需查找奇数区间即可

    方法二

    class Solution {
public:
    int singleNonDuplicate(vector<int>& nums) {
        int res = nums[0];
        for (int i = 1; i < nums.size(); i++)
            res ^= nums[i];
        return res;
    }
};

      • 将nums中所有元素进行异或操作
      • 最后得到的结果即为仅出现一次的数
      • 其实时间复杂度是O(n)，但位运算的思路值得学习

    leetcode 683 平方数之和

    题目

    给定一个非负整数 c ，你要判断是否存在两个整数 a 和 b，使得 a2 + b2 = c 。

    示例

    示例一

    输入：c = 5 输出：true 解释：1 * 1 + 2 * 2 = 5

    示例二

    输入：c = 3 输出：false

    提示

    0 <= c <= 231 - 1

    题解

    方法一

    class Solution {
public:
    bool judgeSquareSum(int c) {
        unordered_set<int> little_squares;
        long long base = 0;
        long long square = base * base;
        while (square <= c / 2) {
            little_squares.insert(square);
            if (square + square == c) return true;
            base++;
            square = base * base;
        }
        while (square <= c) {
            if (little_squares.count(c - square)) return true;
            base++;
            square = base * base;
        }
        return false;
    }
};
