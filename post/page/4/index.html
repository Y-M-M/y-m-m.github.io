    leetcode 3249 统计好节点的数目

    题目

    现有一棵 无向 树，树中包含 n 个节点，按从 0 到 n - 1 标记。树的根节点是节点 0 。给你一个长度为 n - 1 的二维整数数组 edges，其中 edges[i] = [ai, bi] 表示树中节点 ai 与节点 bi 之间存在一条边。 如果一个节点的所有子节点为根的 子树 包含的节点数相同，则认为该节点是一个 好节点。 返回给定树中 好节点 的数量。 子树 指的是一个节点以及它所有后代节点构成的一棵树。

    示例

    示例一

    输入：edges = [[0,1],[0,2],[1,3],[1,4],[2,5],[2,6]] 输出：7 说明

    示例二

    输入：edges = [[0,1],[1,2],[2,3],[3,4],[0,5],[1,6],[2,7],[3,8]] 输出：6 说明：

    示例三

    输入：edges = [[0,1],[1,2],[1,3],[1,4],[0,5],[5,6],[6,7],[7,8],[0,9],[9,10],[9,12],[10,11]] 输出：12 解释：

    提示

    • 2 <= n <= $10^5$
    • edges.length == n - 1
    • edges[i].length == 2
    • 0 <= $a_i$, $b_i$ < n
    • 输入确保 edges 总表示一棵有效的树。

    题解

    方法一

    class Solution {
public:
    int dfs (const int& root, const vector<vector<int>>& tree, vector<int>& sum) // 深度优先搜索，统计每棵子树的节点数
    {
        int cnt = 1;
        for (int i = 0; i < tree[root].size(); i++)
        {
            cnt += dfs(tree[root][i], tree, sum);
        }
        sum[root] = cnt;
        return cnt;
    }

    void buildit(vector<vector<int>>& tree, const int& root) // 将连接关系缩减为父子关系
    {
        for (int i = 0; i < tree[root].size(); i++)
        {
            int son = tree[root][i];
            for (int j = 0; j < tree[son].size(); j++)
            {
                if (tree[son][j] == root)
                    tree[son].erase(tree[son].begin() + j);
            }
            buildit(tree, son);
        }
    }

    int countGoodNodes(vector<vector<int>>& edges) {
        int n = edges.size();
        vector<vector<int>> tree(n + 1);
        for (int i = 0; i < n ; i++) // 统计连接关系
        {
            int fa = edges[i][0], son = edges[i][1];
            tree[fa].push_back(son);
            tree[son].push_back(fa);
        }
        buildit(tree, 0); // 将连接关系缩减为父子关系
        
        int ans = 0;
        vector<int> sum(n + 1);
        dfs(0, tree, sum);
        for (int i = 0; i <= n; i++) // 遍历所有节点，判断是不是好节点并统计数目
        {
            int isgood = 1;
            for (int j = 1; j < tree[i].size(); j++)
            {
                if (sum[tree[i][j]] != sum[tree[i][0]])
                {
                    isgood = 0;
                    break;
                }
            }
            ans += isgood;
        }
        return ans;
    }
};
    • 提交结果
      • 通过
    • 分析
      • 时间复杂度O($n^2$)
      • 空间复杂度O(n)
      • 思路
        • 深度优先搜索统计子树的节点数
        • 因为所给的连接关系不会隐含父子关系，需要从根节点开始自己判断，所以先建立连接关系再缩减为父子关系

    方法二

    class Solution {
public:
    int dfs (const int& root, const vector<vector<int>>& tree, vector<int>& sum, const int& fa, int& ans) // 深度优先搜索，统计每棵子树的节点数
    {
        int cnt = 1;
        int num = 0;
        int isgood = 1;
        for (int i = 0; i < tree[root].size(); i++)
        {
            if (tree[root][i] != fa)
            {
                int add_num = dfs(tree[root][i], tree, sum, root, ans);
                if (!num) num = add_num;
                else {
                    if (add_num != num)
                        isgood = 0;
                }
                cnt += add_num;
            } 
        }
        if (isgood) ans++;
        sum[root] = cnt;
        return cnt;
    }

    int countGoodNodes(vector<vector<int>>& edges) {
        int n = edges.size();
        vector<vector<int>> tree(n + 1);
        for (int i = 0; i < n ; i++) // 统计连接关系
        {
            int fa = edges[i][0], son = edges[i][1];
            tree[fa].push_back(son);
            tree[son].push_back(fa);
        }
        
        int ans = 0;
        vector<int> sum(n + 1);
        dfs(0, tree, sum, -1, ans);
        return ans;
    }
};
    • 提交结果
      • 通过
    • 分析
      • 在方法一的基础上改进
        • dfs的同时统计好节点的数目
        • 不手动删除冗余的连接关系，只要在dfs的时候加个判断，不要访问父节点就可以了
      • 时间复杂度O(n)
      • 空间复杂度O(n)

  • leetcode 540 有序数组中的单一元素

    题目

    给你一个仅由整数组成的有序数组，其中每个元素都会出现两次，唯有一个数只会出现一次。 请你找出并返回只出现一次的那个数。 你设计的解决方案必须满足 O(log n) 时间复杂度和 O(1) 空间复杂度。

    示例

    示例一

    输入: nums = [1,1,2,3,3,4,4,8,8] 输出: 2

    示例二

    输入: nums = [3,3,7,7,10,11,11] 输出: 10

    提示

    • 1 <= nums.length <= 105
    • 0 <= nums[i] <= 105

    题解

    方法一

    class Solution {
public:
    int singleNonDuplicate(vector<int>& nums) {
        int l = 0, r = nums.size() - 1;
        while (l < r)
        {
            int mid = (l + r) / 2;
            if (nums[mid - 1] == nums[mid])
            {
                if (mid % 2 == 0)
                    r = mid - 2;
                else 
                    l = mid + 1;
                continue;
            } 
            else if (nums[mid + 1] == nums[mid])
            {
                if (mid % 2 == 0)
                    l = mid + 2;
                else 
                    r = mid - 1;
                continue;
            }
            else 
                return nums[mid];
        }
        return nums[l];
    }
};
    • 提交结果
      • 通过
    • 分析
      • 整个数组有序排列，考虑二分查找
      • 每次只需查找奇数区间即可

    方法二

    class Solution {
public:
    int singleNonDuplicate(vector<int>& nums) {
        int res = nums[0];
        for (int i = 1; i < nums.size(); i++)
            res ^= nums[i];
        return res;
    }
};
    • 提交结果
      • 通过
    • 分析
      • 将nums中所有元素进行异或操作
      • 最后得到的结果即为仅出现一次的数
      • 其实时间复杂度是O(n)，但位运算的思路值得学习

    介绍

    • 线段树
      • 用来维护区间信息的数据结构
      • 可以在O(logN)的时间复杂度内实现单点修改，区间修改，区间查询等操作

    线段树是什么样的？

    • 将每个长度不为1的区间划分成左右两个区间递归求解
    • 图中红色字体标明的区间，表示该节点管辖的位置区间

    $d_i$的左儿子为$d_{2 * i}$，右儿子为$d_{2 * i + 1}$ $d_i$表示的区间是[s, t]，$d_{2 * i}$表示的区间是[s, $\frac{s + t}{2}$]，$d_{2 * i + 1}$表示的区间是$[\frac{s + t}{2} + 1, t]$

    建树

    • 递归
    void build(int s, int t, int p) {
  // 对 [s,t] 区间建立线段树,当前根的编号为 p
  if (s == t) {
    d[p] = a[s];
    return;
  }
  int m = s + ((t - s) >> 1);
  // 移位运算符的优先级小于加减法，所以加上括号
  // 如果写成 (s + t) >> 1 可能会超出 int 范围
  build(s, m, p * 2), build(m + 1, t, p * 2 + 1);
  // 递归对左右区间建树
  d[p] = d[p * 2] + d[(p * 2) + 1];
}

    区间查询

    • 如果要查询的区间是[l, r]，则可将其拆成至多O(logn)个极大的区间，合并即可
    int getsum(int l, int r, int s, int t, int p) {
  // [l, r] 为查询区间, [s, t] 为当前节点包含的区间, p 为当前节点的编号
  if (l <= s && t <= r)
    return d[p];  // 当前区间为询问区间的子集时直接返回当前区间的和
  int m = s + ((t - s) >> 1), sum = 0;
  if (l <= m) sum += getsum(l, r, s, m, p * 2);
  // 如果左儿子代表的区间 [s, m] 与询问区间有交集, 则递归查询左儿子
  if (r > m) sum += getsum(l, r, m + 1, t, p * 2 + 1);
  // 如果右儿子代表的区间 [m + 1, t] 与询问区间有交集, 则递归查询右儿子
  return sum;
}

    区间修改

    • 懒惰标记
      • 修改区间[l, r]时，只修改节点并标记(懒惰标记)，而不修改其子节点
      • 等到下一次访问带有标记的节点时再进行

    leetcode 683 平方数之和

    题目

    给定一个非负整数 c ，你要判断是否存在两个整数 a 和 b，使得 a2 + b2 = c 。

    示例

    示例一

    输入：c = 5 输出：true 解释：1 * 1 + 2 * 2 = 5

    示例二

    输入：c = 3 输出：false

    提示

    0 <= c <= 231 - 1

    题解

    方法一

    class Solution {
public:
    bool judgeSquareSum(int c) {
        unordered_set<int> little_squares;
        long long base = 0;
        long long square = base * base;
        while (square <= c / 2) {
            little_squares.insert(square);
            if (square + square == c) return true;
            base++;
            square = base * base;
        }
        while (square <= c) {
            if (little_squares.count(c - square)) return true;
            base++;
            square = base * base;
        }
        return false;
    }
};

    题目

    • 来自未来的体育科学家给你两个整数数组 energyDrinkA 和 energyDrinkB，数组长度都等于 n。这两个数组分别代表 A、B 两种不同能量饮料每小时所能提供的强化能量。
    • 你需要每小时饮用一种能量饮料来 最大化 你的总强化能量。然而，如果从一种能量饮料切换到另一种，你需要等待一小时来梳理身体的能量体系（在那个小时里你将不会获得任何强化能量）。
    • 返回在接下来的 n 小时内你能获得的 最大 总强化能量。
    • 注意 你可以选择从饮用任意一种能量饮料开始。

    示例

    示例一

    输入：energyDrinkA = [1,3,1], energyDrinkB = [3,1,1] 输出：5 解释： 要想获得 5 点强化能量，需要选择只饮用能量饮料 A（或者只饮用 B）。

    示例二

    输入：energyDrinkA = [4,1,1], energyDrinkB = [1,1,3] 输出：7 解释： 第一个小时饮用能量饮料 A。 切换到能量饮料 B ，在第二个小时无法获得强化能量。 第三个小时饮用能量饮料 B ，并获得强化能量。

    提示

    • n == energyDrinkA.length == energyDrinkB.length
    • 3 <= n <= 105
    • 1 <= energyDrinkA[i], energyDrinkB[i] <= 105

    题解

    方法一

    class Solution {
public:
    long long maxEnergyBoost(vector<int>& energyDrinkA, vector<int>& energyDrinkB) {
        if (energyDrinkA.size() <= 2)
        {
            return max(energyDrinkA[0] + energyDrinkA[1], energyDrinkB[0] + energyDrinkB[1]);
        }
        vector<long long> max_energyA(energyDrinkA.size());
        vector<long long> max_energyB(energyDrinkB.size());
        max_energyA[0] = energyDrinkA[0];
        max_energyB[0] = energyDrinkB[0];
        max_energyA[1] = energyDrinkA[0] + energyDrinkA[1];
        max_energyB[1] = energyDrinkB[0] + energyDrinkB[1];
        
        for (int i = 2; i < max_energyA.size(); i++) {
            max_energyA[i] = max(max_energyB[i - 2], max_energyA[i - 1]) + energyDrinkA[i];
            max_energyB[i] = max(max_energyA[i - 2], max_energyB[i - 1]) + energyDrinkB[i];
        }
        return max(max_energyA.back(), max_energyB.back());
    }
};
