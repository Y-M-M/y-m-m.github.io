    class Solution {
public:
    bool valid(const string& str)
    {
        int balance = 0;
        for (char c : str)
        {
            if (c == '(')
            {
                ++balance;
            }
            else
            {
                --balance;
            }
            if (balance < 0)
            {
                return false;
            }
        }
        return balance == 0;
    }

    void generate_all(string& current, int n, vector<string>& result)
    {
        if (n == current.size())
        {
            if (valid(current))
            {
                result.push_back(current);
            }
            return;
        }
        current += '(';
        generate_all(current, n, result);
        current.pop_back();
        current += ')';
        generate_all(current, n, result);
        current.pop_back();
    }

    vector<string> generateParenthesis(int n) {
        vector<string> result;
        string current;
        generate_all(current, n * 2, result);
        return result;
    }
};

    class Solution {

    void backtrack(vector<string>& ans, string& cur, int open, int close, int n)
    {
        if (cur.size() == n * 2)
        {
            ans.push_back(cur);
            return;
        }

        if (open < n)
        {
            cur.push_back('(');
            backtrack(ans, cur, open + 1, close, n);
            cur.pop_back();
        }

        if (close < open)
        {
            cur.push_back(')');
            backtrack(ans, cur, open, close + 1, n);
            cur.pop_back();
        }
    }

    vector<string> generateParenthesis(int n) {
        vector<string> result;
        string current;
        backtrack(result, current, 0, 0, n);
        return result;
    }
};

                for (const string& left : *lefts)

                    for (const string& right : *rights)

                    }
                }
            }
            cache[n] = result;
        }
        return cache[n];
    }

        return *generate(n);
    }

    class Solution {
public:
    bool search(vector<vector<char>>& board, const string& word, int pos, vector<pair<int, int>>& nowposs)
    {
        // cout << pos << ", " << nowposs.size() << endl;
        if (pos >= word.size())
        {
            if (nowposs.size())
                return true;
            else
                return false;
        }
        if (!pos)
        {
            for (int i = 0; i < m; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    if (board[i][j] == word[pos])
                    {
                        nowposs.push_back(make_pair(i, j));
                        visited[i][j] = true;
                    }
                }
            }
        }
        else
        {
            if (!nowposs.size()) return false;
            vector<pair<int, int>> nextposss;
            for (pair<int, int> apos : nowposs)
            {
                int i = apos.first;
                int j = apos.second;
                if (i > 0 && board[i - 1][j] == word[pos])
                {
                    if (!visited[i - 1][j])
                    {
                        nextposss.push_back(make_pair(i - 1, j));
                        visited[i - 1][j] = true;
                    }
                }
                if (i < m - 1 && board[i + 1][j] == word[pos])
                {
                    if (!visited[i + 1][j])
                    {
                        nextposss.push_back(make_pair(i + 1, j));
                        visited[i + 1][j] = true;
                    }
                }
                if (j > 0 && board[i][j - 1] == word[pos])
                {
                    if (!visited[i][j - 1])
                    {
                        nextposss.push_back(make_pair(i, j - 1));
                        visited[i][j - 1] = true;
                    }
                    
                }
                if (j < n - 1 && board[i][j + 1] == word[pos])
                {
                    if (!visited[i][j + 1])
                    {
                        nextposss.push_back(make_pair(i, j + 1));
                        visited[i][j + 1] = true;
                    }
                    
                }
            }
            nowposs = nextposss;
        }
        return search(board, word, pos + 1, nowposs);
    }

    bool exist(vector<vector<char>>& board, string word) {
        if (!word.size())
        {
            return true;

        if (!m) return false;

        return search (board, word, 0, nowposs);
