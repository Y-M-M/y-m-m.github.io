leetcode 239 滑动窗口最大值

题目

给你一个整数数组 nums，有一个大小为 k 的滑动窗口从数组的最左侧移动到数组的最右侧。你只可以看到在滑动窗口内的 k 个数字。滑动窗口每次只向右移动一位。
返回 滑动窗口中的最大值 。

示例

示例一

输入：nums = [1,3,-1,-3,5,3,6,7], k = 3
输出：[3,3,5,5,6,7]
解释：

示例二

输入：nums = [1], k = 1
输出：[1]

提示

  • 1 <= nums.length <= 105
  • -104 <= nums[i] <= 104
  • 1 <= k <= nums.length

题解

方法一

  • 分析
    • 维护一个priority_queue，初始存储第一个窗口内有用的值，第一个窗口内的最大值即为pq.top()
    • 维护一个unordered_map，记录所有数值在原数组中的最大位置
    • 每次移动窗口后，判断当前最大值是否被移出窗口
  • 提交结果
    • 通过
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        priority_queue<int, vector<int>> pq;
        unordered_map<int, int> pos;
        vector<int> ans;
        if (nums.size() < k)
        {
            return ans;
        }

        for (int i = 0; i < k; i++)
        {
            pq.emplace(nums[i]);
            pos[nums[i]] = i;
        }

        ans.push_back(pq.top());

        if (nums.size() == k)
        {
            return ans;
        }

        for (int i = k ; i < nums.size(); i++)
        {
            pos[nums[i]] = i;
            pq.emplace(nums[i]);

            int nowtop = pq.top();
            while (pos.find(nowtop) != pos.end())
            {
                if (pos[nowtop] <= i - k)
                {
                    pq.pop();
                    nowtop = pq.top();
                } 
                else
                    break;
            }
            ans.push_back(pq.top());
        }
        return ans;
    }
};

方法二

  • 分析
    • 在方法一的基础上改进
    • priorty_queue中存储数字->位置的pair
  • 提交结果
    • 通过
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        vector<int> ans;
        if (nums.size() < k)
        {
            return ans;
        }
        priority_queue<pair<int, int>> pq; // 值->位置
        for (int i = 0; i < k; i++)
        {
            pq.emplace(nums[i], i);
        }
        ans.push_back(pq.top().first);
        for (int i = k; i < nums.size(); i++)
        {
            pq.emplace(nums[i], i);
            while (pq.top().second <= i - k)
            {
                pq.pop();
            }
            ans.push_back(pq.top().first);
        }
        return ans;
    }
};

方法三

单调队列：满足单调性的双端队列

  • 分析
    • 如果i < j, nums[i] < nums[j]，则nums[i]在nums[j]存在的时候对答案毫无贡献
    • 用一个双端队列存储窗口中还没有被移除的下标
    • 每次窗口移动时，对新的即将加入双端队列的元素，移出其前面值比它小的元素，再加入队列中
    • ！队列的顺序存储nums[i]的相对大小，队列的内容存储i
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        int n = nums.size();
        deque<int> pq;
        for (int i = 0; i < k; i++)
        {
            // 移除前面对应的值比当前数大的元素
            while (!pq.empty() && nums[i] >= nums[pq.back()]) {
                pq.pop_back();
            }
            pq.push_back(i);
        }
        vector<int> ans;
        ans.push_back(nums[pq.front()]);
        for (int i = k; i < n; i++)
        {
            while (!pq.empty() && nums[i] >= nums[pq.back()])
            {
                pq.pop_back();
            }
            pq.push_back(i);
            while (pq.front() <= i - k)
            {
                pq.pop_front();
            }
            ans.push_back(nums[pq.front()]);
        }
        return ans;
    }
};

方法四

  • 分析
    • 要求所有窗口中元素的最大值，可事先对数据做预处理，然后提取最大值构成ans数组即可
    • 以 i = n * k 为分界
    • 求preMax和sufMax
      • preMax[i]存储从最近的 n * k 到i的最大元素
      • sufMax[i]存储从i到最近的 n * k 的最大元素
    • [i, j]区间的最大值 = max (preMax[j], sufMax[i])
  • 提交结果
    • 通过
class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        int n = nums.size();
        vector<int> ans;
        if (n < k)
        {
            return ans;
        }
        vector<int> preMax(n, 0);
        vector<int> sufMax(n, 0);
        for (int i = 0; i < n; i++)
        {
            if (i % k == 0)
            {
                preMax[i] = nums[i];
            }
            else
            {
                preMax[i] = max(preMax[i - 1], nums[i]);
            }
        }
        sufMax[n - 1] = nums.back();
        for (int i = n - 2; i >= 0; i--)
        {
            if (i % k == (k - 1))
            {
                sufMax[i] = nums[i];
            }
            else
            {
                sufMax[i] = max(sufMax[i + 1], nums[i]);
            }
        }
        for (int i = 0; i <= n - k; i++)
        {
            ans.push_back(max(preMax[i + k - 1], sufMax[i]));
        }
        return ans;
    }
};
