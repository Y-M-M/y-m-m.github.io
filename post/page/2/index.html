 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int dfs(TreeNode* root)
    {
        if (root == nullptr)
        {
            return 0;
        }
        int leftmax = dfs(root->left);
        int rightmax = dfs(root->right);
        int maxpath = 0;
        if (leftmax > 0)  maxpath += leftmax;
        if (rightmax > 0) maxpath += rightmax;
        maxpath += root->val;
        if (maxpath > ans)
        {
            ans = maxpath;
        }
        int addnum = max(leftmax, rightmax);
        if (addnum > 0)
            return addnum + root->val;
        return root->val;
        
    }

    int maxPathSum(TreeNode* root) {
        if (root == nullptr)
        {
            return 0;
        }
        ans = -INT32_MAX;
        dfs(root);
        return ans;
    }
private:

            {
                st2.pop();

            st1.pop();

        }

class Solution {
public:
    void flatten(TreeNode* root) {
        if (root == nullptr)
        {
            return;
        }
        TreeNode* temp = nullptr;
        TreeNode* rend = nullptr;
        while (root->left != nullptr || root->right != nullptr)
        {
            if (root->left != nullptr)
            {
                temp = root->right;
                root->right = root->left;
                root->left = nullptr;
                root = root->right;
                rend = root;
                while (rend->right != nullptr)
                {
                    rend = rend->right;
                }
                rend->right = temp;
            }
            else
            {
                root = root->right;
            }
        }
    }
};

 * struct TreeNode {

 *     TreeNode *right;

 * };

class Solution {
public:
    vector<int> rightSideView(TreeNode* root) {
        queue<TreeNode*> layer;
        vector<int> ans;
        if (root == nullptr)
        {
            return ans;
        }
        layer.push(root);
        while (!layer.empty())
        {
            int len = layer.size();
            for (int i = 0; i < len; i++)
            {
                TreeNode* nownode = layer.front();
                layer.pop();
                if (i == len - 1)
                {
                    ans.push_back(nownode->val);
                }
                if (nownode->left != nullptr)
                {
                    layer.push(nownode->left);
                }
                if (nownode->right != nullptr)
                {
                    layer.push(nownode->right);
                }
            }
        }
        return ans;
    }
};
