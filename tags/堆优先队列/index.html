  • leetcode 1705 吃苹果的最大数目

    题目

    有一棵特殊的苹果树，一连 n 天，每天都可以长出若干个苹果。在第 i 天，树上会长出 apples[i] 个苹果，这些苹果将会在 days[i] 天后（也就是说，第 i + days[i] 天时）腐烂，变得无法食用。也可能有那么几天，树上不会长出新的苹果，此时用 apples[i] == 0 且 days[i] == 0 表示。
    你打算每天 最多 吃一个苹果来保证营养均衡。注意，你可以在这 n 天之后继续吃苹果。
    给你两个长度为 n 的整数数组 days 和 apples ，返回你可以吃掉的苹果的最大数目。

    示例

    示例一

    输入：apples = [1,2,3,5,2], days = [3,2,1,4,2]
    输出：7
    解释：你可以吃掉 7 个苹果：

    • 第一天，你吃掉第一天长出来的苹果。
    • 第二天，你吃掉一个第二天长出来的苹果。
    • 第三天，你吃掉一个第二天长出来的苹果。过了这一天，第三天长出来的苹果就已经腐烂了。
    • 第四天到第七天，你吃的都是第四天长出来的苹果。

    示例二

    输入：apples = [3,0,0,0,0,2], days = [3,0,0,0,0,2]
    输出：5
    解释：你可以吃掉 5 个苹果：

    • 第一天到第三天，你吃的都是第一天长出来的苹果。
    • 第四天和第五天不吃苹果。
    • 第六天和第七天，你吃的都是第六天长出来的苹果。

    题解

    方法一

    class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        vector<int> specialdays;
        unordered_map<int, int> rotapples;
        for (int i = 0; i < apples.size(); i++)
        {
            if (apples[i])
            {
                if (rotapples.find(i + days[i]) != rotapples.end())
                {
                    rotapples[i + days[i]] += apples[i];
                }
                else 
                {
                    specialdays.push_back(i + days[i]);
                    rotapples[i + days[i]] = apples[i];
                }   
            }
        }
        sort(specialdays.begin(), specialdays.end());
        int ans = min(specialdays[0], rotapples[specialdays[0]]);
        for (int i = 0; i < specialdays.size() - 1; i++)
        {
            int nowday = specialdays[i];
            int nextday = specialdays[i + 1];
            ans += min(rotapples[nextday], nextday - nowday);
        }
        return ans;
    }
};

public:
    bool eatapple(int& day, priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>>& pq)
    {
        while (!pq.empty() && pq.top().first <= day)
        {
            // cout << "过期的苹果" << pq.top().first << ", " << pq.top().second << endl;
            pq.pop();

        {
            pair<int, int> newapples = pq.top();
            // cout << "吃掉一个苹果" << pq.top().first << ", " << pq.top().second << endl;
            newapples.second--;
            pq.pop();
            if (newapples.second)
            {
                // cout << "添加一个新苹果堆" << newapples.first << ", " << newapples.second << endl;
                pq.push(newapples);
            }
        }
        else
        {
            return false;
        }
        return true;
    }

    int eatenApples(vector<int>& apples, vector<int>& days) {
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
        int ans = 0;
        for (int i = 0; i < apples.size(); i++)
        {
            if (apples[i])
            {
                pq.push(make_pair(i + days[i], apples[i]));
            }
            ans += eatapple(i, pq);
        }
        for (int i = apples.size(); ; i++)
        {
            if (eatapple(i, pq))
            {
                ans++;
            }
            else 
            {
                break;
            }
        }
        return ans;
    }
};

  • leetcode 855 考场就坐

    题目

    在考场里，一排有 N 个座位，分别编号为 0, 1, 2, …, N - 1 。
    当学生进入考场后，他必须坐在能够使他与离他最近的人之间的距离达到最大化的座位上。如果有多个这样的座位，他会坐在编号最小的座位上。(另外，如果考场里没有人，那么学生就坐在 0 号座位上。)
    返回 ExamRoom(int N) 类，它有两个公开的函数：其中，函数 ExamRoom.seat() 会返回一个 int （整型数据），代表学生坐的位置；函数 ExamRoom.leave(int p) 代表坐在座位 p 上的学生现在离开了考场。每次调用 ExamRoom.leave(p) 时都保证有学生坐在座位 p 上。

    示例

    输入：[“ExamRoom”,“seat”,“seat”,“seat”,“seat”,“leave”,“seat”], [[10],[],[],[],[],[4],[]]
    输出：[null,0,9,4,2,null,5]
    解释：
    ExamRoom(10) -> null
    seat() -> 0，没有人在考场里，那么学生坐在 0 号座位上。
    seat() -> 9，学生最后坐在 9 号座位上。
    seat() -> 4，学生最后坐在 4 号座位上。
    seat() -> 2，学生最后坐在 2 号座位上。
    leave(4) -> null
    seat() -> 5，学生最后坐在 5 号座位上。

    提示

    • 1 <= N <= $10^9$
    • 在所有的测试样例中 ExamRoom.seat() 和 ExamRoom.leave() 最多被调用 $10^4$ 次。
    • 保证在调用 ExamRoom.leave(p) 时有学生正坐在座位 p 上。

    题解

    方法一

    struct Comp {
    bool operator() (const pair<int, int> &p1, const pair<int, int> &p2) {
        int d1 = p1.second - p1.first, d2 = p2.second - p2.first;
        return d1 / 2 < d2 / 2 || (d1 / 2 == d2 / 2 && p1.first > p2.first);
    }
};

class ExamRoom {
public:
    ExamRoom(int n) : n(n) {
    }
    
    int seat() {
        if (seats.empty()) {
            seats.insert(0);
            return 0;
        }
        int left = *seats.begin(), right = n - 1 - *seats.rbegin();
        while (seats.size() >= 2) {
            auto p = pq.top();
            if (seats.count(p.first) > 0 && seats.count(p.second) > 0 && *next(seats.find(p.first)) == p.second) // 有效，不属于延迟删除的区间
            {
                int d = p.second - p.first;
                if (d / 2 < right || d / 2 <= left) { // 左部和右部的区间更优
                    break;
                }
                pq.pop();
                pq.push({p.first, p.first + d / 2});
                pq.push({p.first + d / 2, p.second});
                seats.insert(p.first + d / 2);
                return p.first + d / 2;
            }
            pq.pop(); // 删除延迟删除的区间
        }
        if (right > left) {
            pq.push({*seats.rbegin(), n - 1});
            seats.insert(n - 1);
            return n - 1;
        } else {
            pq.push({0, *seats.begin()});
            seats.insert(0);
            return 0;
        }
    }
    
    void leave(int p) {
        if (p != *seats.begin() && p != *seats.rbegin()) {
            auto it = seats.find(p);
            pq.push({*prev(it), *next(it)});
        }
        seats.erase(p);
    }

private:
    int n;
    set<int> seats;
    priority_queue<pair<int, int>, vector<pair<int, int>>, Comp> pq;
};

/**
 * Your ExamRoom object will be instantiated and called as such:
 * ExamRoom* obj = new ExamRoom(n);
 * int param_1 = obj->seat();
 * obj->leave(p);
 */

    示例

    输出：2
    解释：选择 {3,7} 使得结果数组为 [5,5,5,2,2]、长度为 5（原数组长度的一半）。
    大小为 2 的可行集合有 {3,5},{3,2},{5,2}。
    选择 {2,7} 是不可行的，它的结果数组为 [3,3,3,3,5,5,5]，新数组长度大于原数组的二分之一。

    示例二

    输入：arr = [7,7,7,7,7,7]
    输出：1
    解释：我们只能选择集合 {7}，结果数组为空

    提示

    • arr.length 为偶数
    • 1 <= arr[i] <= 105

    题解

    方法一

    class Solution {
public:
    int minSetSize(vector<int>& arr) {
        if (arr.size() == 1) return 1;
        vector<int> cnt(100001, 0);
        int r = -1, l = 100001;
        for (int i = 0; i < arr.size(); i++)
        {
            cnt[arr[i]]++;
            if (arr[i] < l) l = arr[i];
            if (arr[i] > r) r = arr[i];
        }
        vector<int> sortedcnt;
        for (int i = l; i <= r; i++)
        {
            sortedcnt.push_back(cnt[i]);
        }
        make_heap(sortedcnt.begin(), sortedcnt.end());
        int ans = 0;
        int num = 0;
        while (num < arr.size() / 2)
        {
            pop_heap(sortedcnt.begin(), sortedcnt.end());
            num += sortedcnt.back();
            ans++;
            sortedcnt.pop_back();
        }
        return ans;
    }
};

    class Solution {
public:
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        auto cmp = [&nums](int left, int right) {return nums[left] > nums[right];};
        priority_queue<int, vector<int>, decltype(cmp)> id(cmp);

        for (int i = 0; i < nums.size(); i++)
        {
            id.push(i);
        }
        for (int i = 0; i < k; i++)
        {
            int updatetarget = id.top();
            id.pop();
            nums[updatetarget] *= multiplier;
            id.push(updatetarget);
        }
        return nums;
    }
};

    class Solution {
public:
    long long quickMul(long long x, long long y, long long m)
    {
        long long res = 1;
        while (y)
        {
            if (y & 1) {
                res = (res * x) % m;
            }
            y >>= 1;
            x = (x * x) % m;
        }
        return res;
    }
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        if (multiplier == 1)
        {
            return nums;
        }
        long long n = nums.size(), m = 1e9 + 7;
        long long mx = *max_element(nums.begin(), nums.end()); // 取最大值
        vector<pair<long long, int>> v(n);
        for (int i = 0; i < n; i++)
        {
            v[i] = {nums[i], i};
        }
        make_heap(v.begin(), v.end(), greater<>());
        for (; v[0].first < mx && k; k--)
        {
            pop_heap(v.begin(), v.end(), greater<>());
            v.back().first *= multiplier;
            push_heap(v.begin(), v.end(), greater<>());
        }
        sort(v.begin(), v.end());
        for (int i = 0; i < n; i++)
        {
            int t = k / n + (i < k % n);
            nums[v[i].second] = ((v[i].first % m) * quickMul(multiplier, t, m)) % m;
        }
        return nums;
    }
};

      • 通过pair的排序规则来进行堆排序，先比较first，first相同的时候在比较second
      • 对过程进行优化
        • 如果nums中所有数都被更新过了，那么此后更新会从小到大把nums中的所有数都*k
        • 这个过程可直接遍历进行，无需再进行堆排序

    leetcode 743 网络延迟时间

    题目

    有 n 个网络节点，标记为 1 到 n。
    给你一个列表 times，表示信号经过 有向 边的传递时间。 times[i] = $(u_i, v_i, w_i)$，其中 $u_i$ 是源节点，$v_i$ 是目标节点， $w_i$ 是一个信号从源节点传递到目标节点的时间。
    现在，从某个节点 K 发出一个信号。需要多久才能使所有节点都收到信号？如果不能使所有节点收到信号，返回 -1 。

    示例

    示例一

    示例二

    输入：times = [[1,2,1]], n = 2, k = 1
    输出：1

    示例三

    输入：times = [[1,2,1]], n = 2, k = 2
    输出：-1

    提示

    • 1 <= k <= n <= 100
    • 1 <= times.length <= 6000
    • times[i].length == 3
    • 1 <= $u_i, v_i$ <= n
    • $u_i$ != $v_i$
    • 0 <= $w_i$ <= 100
    • 所有 (ui, vi) 对都 互不相同（即，不含重复边）

    题解

    方法一

    class Solution {
public:
    int networkDelayTime(vector<vector<int>>& times, int n, int k) {
        vector<vector<int>> next(n + 1); // 记录每个节点的出边存放的位置
        for (int i = 0; i < times.size(); i++) {
            next[times[i][0]].push_back(i);
        }
        
        vector<bool> is_visited(n + 1, false);
        int root = k;
        is_visited[k] = true;
        vector<int> ans(n + 1);
        ans[root] = 0; 
        
        auto cmp = [&](const int &u, const int &v) {
            return ans[times[u][0]] + times[u][2] > ans[times[v][0]] + times[v][2];
        };
        priority_queue<int, vector<int>, decltype(cmp)> pq(cmp);
        int newnode = 0;
        int res = 0;
        
        do {
            for (int i = 0; i < next[root].size(); i++) {
                pq.emplace(next[root][i]);
            }
            
            if (pq.empty()) break; // 检查队列是否为空
            
            newnode = pq.top(); // 加入的边在times中的位置
            pq.pop();
            
            // 找到下一个未访问的节点
            while (!pq.empty() && is_visited[times[newnode][1]]) {
                newnode = pq.top();
                pq.pop();
            }
            
            // 如果pq为空，说明没有未访问的节点，提前退出
            if (is_visited[times[newnode][1]]) break;
            ans[times[newnode][1]] = ans[times[newnode][0]] + times[newnode][2];
            res = ans[times[newnode][1]];
            root = times[newnode][1];
            is_visited[root] = true; // 标记当前节点为已访问
        } while (true);
        
        // 检查是否所有节点都被访问过
        for (int i = 1; i <= n; i++) {
            if (!is_visited[i])
                return -1;
        }
        return res;
    }
};

    • 分析
      • 类比Djkstra算法求最短路径
      • 不在更新时进行松弛操作，而是每次都将剩下节点中离起始节点最近的节点加入visited中
      • 最后加入vistited的一定是离起始节点最远的节点，更新的距离即为离最远节点的最小距离，即为最大传播时间
      • 优先级队列维护最小堆

    class Solution {
public:
    int networkDelayTime(vector<vector<int>> &times, int n, int k) {
        const int inf = INT_MAX / 2;
        vector<vector<int>> g(n, vector<int>(n, inf));
        for (auto &t : times) {
            int x = t[0] - 1, y = t[1] - 1; // x与y分别代表边的两端的节点
            g[x][y] = t[2]; // 二维数组g存储边的权重
        }

        vector<int> dist(n, inf); // 存储距离
        dist[k - 1] = 0; // 跟节点到自己的距离为0
        vector<int> used(n); // 检查节点有么有被使用过
        for (int i = 0; i < n; ++i) {
            int x = -1;
            for (int y = 0; y < n; ++y) { // 找出distance最小的，没有使用过的x
                if (!used[y] && (x == -1 || dist[y] < dist[x])) {
                    x = y;
                }
            }
            used[x] = true; // 将x标记为已经使用过
            for (int y = 0; y < n; ++y) {
                dist[y] = min(dist[y], dist[x] + g[x][y]); // 看新加入的x节点能否使其他节点的距离变小
            }
        }

        int ans = *max_element(dist.begin(), dist.end());
        return ans == inf ? -1 : ans;
    }
};

    题目

    你有 k 个 非递减排列 的整数列表。找到一个 最小 区间，使得 k 个列表中的每个列表至少有一个数包含在其中。
    我们定义如果 b-a < d-c 或者在 b-a == d-c 时 a < c，则区间 [a,b] 比 [c,d] 小。

    示例

    示例一

    输入：nums = [[4,10,15,24,26], [0,9,12,20], [5,18,22,30]]
    输出：[20,24]
    解释：
    列表 1：[4, 10, 15, 24, 26]，24 在区间 [20,24] 中。
    列表 2：[0, 9, 12, 20]，20 在区间 [20,24] 中。
    列表 3：[5, 18, 22, 30]，22 在区间 [20,24] 中。

    示例二

    输入：nums = [[1,2,3],[1,2,3],[1,2,3]]
    输出：[1,1]

    提示

    • nums.length == k
    • 1 <= k <= 3500
    • 1 <= nums[i].length <= 50
    • $-10^5$ <= nums[i][j] <= $10^5$
    • nums[i] 按非递减顺序排列

    题解

    方法一

    class Solution {
public:
    vector<int> smallestRange(vector<vector<int>>& nums) {
        vector<int> region(nums.size());
        for (int i = 0; i < nums.size(); i++)
        {
            region[i] = 0;
        }
        int times = 0;
        int ans; // 区域的长度
        int l, r; // 区域的左右边界
        while (1)
        {
            int mx, mn; // 区域里的最大、最小值
            int mxpos, mnpos; // 位于哪个数组中
            for (int i = 0; i < nums.size(); i++)
            {    
                if (!i) {
                    mx = nums[0][region[0]], mn = nums[0][region[0]];
                    mxpos = 0, mnpos = 0;
                }
                else {
                    int newnum = nums[i][region[i]];
                    if (newnum > mx)
                    {
                        mx = newnum;
                        mxpos = i;
                    }
                    if (newnum < mn)
                    {
                        mn = newnum;
                        mnpos = i;
                    }
                }
            }
            if (!times)
            {
                ans = mx - mn + 1;
                l = mn, r = mx;
            }
            else 
            {
                if (mx - mn + 1 < ans)
                {
                    ans = mx - mn + 1;
                    l = mn, r = mx;
                }
            }
            times++;
            if (++region[mnpos] >= nums[mnpos].size())
            {
                break;
            }
        }
        vector<int> res;
        res.push_back(l);
        res.push_back(r);
        return res;
    }
};

    • 分析
      • 每个数组取一个数，得到nums.size()个数，对这些数进行比较，取得最大值mx和最小值mn, 即为当前的区间边界

    方法二

    class Solution {
public:
    vector<int> smallestRange(vector<vector<int>>& nums) {
        int rangeLeft = 0, rangeRight = INT_MAX;
        int size = nums.size();
        vector<int> next(size); // 每个数组的指针，初始都指向0
        
        auto cmp = [&](const int& u, const int& v) {
            return nums[u][next[u]] > nums[v][next[v]];
        };
        priority_queue<int, vector<int>, decltype(cmp)> pq(cmp);  // 优先级队列实现小顶堆
        int minValue = 0, maxValue = INT_MIN;
        for (int i = 0; i < size; ++i) {
            pq.emplace(i);
            maxValue = max(maxValue, nums[i][0]); // 元素0的最大值
        }

        while (true) {
            int row = pq.top(); // 弹出对应的值最小的指针
            pq.pop();
            minValue = nums[row][next[row]]; // 最小值
            if (maxValue - minValue < rangeRight - rangeLeft) { // 当前区间最小，更新区间
                rangeLeft = minValue;
                rangeRight = maxValue;
            }
            if (next[row] == nums[row].size() - 1) { // 已经遍历到某一个数组的结束位置
                break;
            }
            ++next[row]; // 最小值所在的数组下标后移
            maxValue = max(maxValue, nums[row][next[row]]); // 看新加入的数是否比最大值大
            pq.emplace(row); // 在堆中加入row
        }

        return {rangeLeft, rangeRight};
    }
};

    leetcode 871 最低加油次数

    题目

    • 汽车从起点出发驶向目的地，该目的地位于出发位置东面 target 英里处。

    • 沿途有加油站，用数组 stations 表示。其中 stations[i] = [positioni, fueli] 表示第 i 个加油站位于出发位置东面 positioni 英里处，并且有 fueli 升汽油。

    • 假设汽车油箱的容量是无限的，其中最初有 startFuel 升燃料。它每行驶 1 英里就会用掉 1 升汽油。当汽车到达加油站时，它可能停下来加油，将所有汽油从加油站转移到汽车中。

    • 为了到达目的地，汽车所必要的最低加油次数是多少？如果无法到达目的地，则返回 -1 。

    注意：如果汽车到达加油站时剩余燃料为 0，它仍然可以在那里加油。如果汽车到达目的地时剩余燃料为 0，仍然认为它已经到达目的地。

    示例

    示例一

    • 输入：target = 1, startFuel = 1, stations = []
    • 输出：0
    • 解释：可以在不加油的情况下到达目的地。

    示例二

    • 输入：target = 100, startFuel = 1, stations = [[10,100]]
    • 输出：-1
    • 解释：无法抵达目的地，甚至无法到达第一个加油站。

    示例三

    • 输入：target = 100, startFuel = 10, stations = [[10,60],[20,30],[30,30],[60,40]]
    • 输出：2
    • 解释：
      1. 出发时有 10 升燃料。
      2. 开车来到距起点 10 英里处的加油站，消耗 10 升燃料。将汽油从 0 升加到 60 升。
      3. 然后，从 10 英里处的加油站开到 60 英里处的加油站（消耗 50 升燃料），
      4. 并将汽油从 10 升加到 50 升。然后开车抵达目的地。
      5. 沿途在两个加油站停靠，所以返回 2 。

    提示

    • 1 <= target, startFuel <= 109
    • 0 <= stations.length <= 500
    • 1 <= positioni < positioni+1 < target
    • 1 <= fueli < 109
