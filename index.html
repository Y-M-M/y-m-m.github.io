    void rotate(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int cnt = 0;
        while (n > 1)
        {
            for (int i = 0; i < n - 1; i++)
            {
                swap(matrix[cnt][cnt + i], matrix[cnt + i][cnt + n - 1]);
                swap(matrix[cnt][cnt + i], matrix[cnt + n - 1][cnt + n - i - 1]);

            n -= 2;
        }
    }

        vector<int> pre(n, 1);
        vector<int> suf(n, 1);
