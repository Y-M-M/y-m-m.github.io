    leetcode hot100 98 二叉搜索树

    题目

    给你一个二叉树的根节点 root ，判断其是否是一个有效的二叉搜索树。
    有效 二叉搜索树定义如下：
    节点的左子树只包含 小于 当前节点的数。
    节点的右子树只包含 大于 当前节点的数。
    所有左子树和右子树自身必须也是二叉搜索树。

    示例

    示例一


    输入：root = [2,1,3]
    输出：true

    示例二


    输入：root = [5,1,4,null,null,3,6]
    输出：false
    解释：根节点的值是 5 ，但是右子节点的值是 4 。

    提示

    • 树中节点数目范围在[1, 104] 内
    • -231 <= Node.val <= 231 - 1

    题解

    方法一

    • 分析
      • 递归
      • 根据祖先节点的值限定上下界限
      • ！问题，无法区分最初限定的上下界限(INT32_MAX和-INT32_MAX-1)与实际作为节点的上下界，前者可以相等，后者不允许相等
      • 解决方法
        • 将扩大最初限定的上下界的范围，转化为long long
        • 增加两个参数，记录上下界是否是节点
    • 提交结果
      • 通过

            }

        {
            if (root->right->val <= root->val || !fitbound(root->right, upper, root->val))
            {
                return false;
            }
        }
        return true;

    leetcode hot100 102 二叉树的层序遍历

    题目

    给你二叉树的根节点 root ，返回其节点值的 层序遍历 。 （即逐层地，从左到右访问所有节点）。

    示例

    示例一


    输入：root = [3,9,20,null,null,15,7]
    输出：[[3],[9,20],[15,7]]

    示例二

    输入：root = [1]
    输出：[[1]]

    示例三

    输入：root = []
    输出：[]

    提示

    • 树中节点数目在范围 [0, 2000] 内
    • -1000 <= Node.val <= 1000

    题解

    方法一

    • 分析
      • 用qeue存储每一层的节点，层序遍历
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        queue<TreeNode*> q;
        vector<vector<int>> ans;
        if (root == nullptr)
        {
            return ans;
        }
        q.push(root);
        while (!q.empty())
        {
            int len = q.size();
            vector<int> nowlayer;
            for (int i = 0; i < len; i++)
            {
                TreeNode* curr = q.front();
                q.pop();
                nowlayer.push_back(curr->val);
                if (curr->left != nullptr)
                {
                    q.push(curr->left);
                }
                if (curr->right != nullptr)
                {
                    q.push(curr->right);
                }
            }
            ans.push_back(nowlayer);
        }
        return ans;
    }
};

    方法二

      • 在方法一的基础上改进
      • 通过ans.push_back(vector ());避免创建额外的空间

    题目

    给你一个整数数组 nums ，其中元素已经按 升序 排列，请你将其转换为一棵 平衡 二叉搜索树。

    示例

    示例一


    输入：nums = [-10,-3,0,5,9]
    输出：[0,-3,9,-10,null,5]
    解释：[0,-10,5,null,-3,null,9] 也将被视为正确答案：

    示例二


    输入：nums = [1,3]
    输出：[3,1]
    解释：[1,null,3] 和 [3,1] 都是高度平衡二叉搜索树。

    提示

    • 1 <= nums.length <= 104
    • -104 <= nums[i] <= 104
    • nums 按 严格递增 顺序排列

    题解

    方法一

    • 分析
      • 递归
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* changetoBST(vector<int>& nums, const int& l, const int& r)
    {
        TreeNode* node = nullptr;
        if (l == r)
        {
            node = new TreeNode(nums[l]);
        }
        else if (r - l == 1)
        {
            node = new TreeNode(nums[l]);
            TreeNode* another_node = new TreeNode(nums[r]);
            node->right = another_node;
        }
        else
        {
            int mid = (l + r) / 2;
            node = new TreeNode(nums[mid]);
            node->left = changetoBST(nums, l, mid - 1);
            node->right = changetoBST(nums, mid + 1, r);
        }
        return node;
    }
    
    TreeNode* sortedArrayToBST(vector<int>& nums) {
        return changetoBST(nums, 0, nums.size() - 1);
    }
};

    leetcode hot100 543 二叉树的直径

    题目

    给你一棵二叉树的根节点，返回该树的 直径 。
    二叉树的 直径 是指树中任意两个节点之间最长路径的 长度 。这条路径可能经过也可能不经过根节点 root 。
    两节点之间路径的 长度 由它们之间边数表示。

    示例

    示例一


    输入：root = [1,2,3,4,5]
    输出：3
    解释：3 ，取路径 [4,2,1,3] 或 [5,2,1,3] 的长度。

    示例二

    输入：root = [1,2]
    输出：1

    提示

    • 树中节点数目在范围 [1, 104] 内
    • -100 <= Node.val <= 100

    题解

    方法一

    • 分析
      • 递归
      • 深度优先搜索
      • 应该遍历每棵子树，找到经过根节点的最长路径，即为整棵树的直径
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int maxdepth(TreeNode* root)
    {
        if (root == nullptr)
        {
            return -1;
        }
        int leftdepth = maxdepth(root->left) + 1;
        int rightdepth = maxdepth(root->right) + 1;
        if (leftdepth + rightdepth > ans)
        {
            ans = leftdepth + rightdepth;
        }
        return max(leftdepth, rightdepth);
    }
    int diameterOfBinaryTree(TreeNode* root) {
        ans = 0;
        if (root == nullptr)
        {
            return 0;
        }
        maxdepth(root);
        return ans;
    }
private:
    int ans;
};

    输出：false

    提示

    • -100 <= Node.val <= 100

    题解

    方法一

    • 分析
      • 递归
    • 提交结果

    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool isCommon(TreeNode* left, TreeNode* right)
    {
        if (left == nullptr && right == nullptr)
        {
            return true;
        }
        if (left == nullptr || right == nullptr)
        {
            return false;
        }
        if (left->val == right->val)
        {
            return isCommon(left->left, right->right) && isCommon(left->right, right->left);
        }
        return false;
    }

    bool isSymmetric(TreeNode* root) {
        if (root == nullptr)
        {
            return true;
        }
        return isCommon(root->left, root->right);
    }
};

    104 二叉树的最大深度

    题目

    给定一个二叉树 root ，返回其最大深度。
    二叉树的 最大深度 是指从根节点到最远叶子节点的最长路径上的节点数。

    示例

    示例一


    输入：root = [3,9,20,null,null,15,7]
    输出：3

    示例二

    输入：root = [1,null,2]
    输出：2

    提示

    • 树中节点的数量在 [0, 104] 区间内。
    • -100 <= Node.val <= 100

    题解

    方法一

    • 分析
      • 递归
      • 深度优先搜索
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int maxDepth(TreeNode* root) {
        if (root == nullptr)
        {
            return 0;
        }
        return 1 + max(maxDepth(root->left), maxDepth(root->right));
    }
};

    方法二

    • 分析
      • 广度优先搜索
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int maxDepth(TreeNode* root) {
        if (root == nullptr)
        {
            return 0;
        }
        queue<TreeNode*> Q;
        Q.push(root);
        int ans = 0;
        while (!Q.empty())
        {
            int sz = Q.size();
            while (sz > 0)
            {
                TreeNode* node = Q.front();
                Q.pop();
                if (node->left) Q.push(node->left);
                if (node->right) Q.push(node->right);
                sz -= 1;
            }
            ans += 1;
        }
        return ans;
    }
};

    /**

    }

    • LRUCache(int capacity) 以 正整数 作为容量 capacity 初始化 LRU 缓存
    • int get(int key) 如果关键字 key 存在于缓存中，则返回关键字的值，否则返回 -1 。
    • void put(int key, int value) 如果关键字 key 已经存在，则变更其数据值 value ；如果不存在，则向缓存中插入该组 key-value 。如果插入操作导致关键字数量超过 capacity ，则应该 逐出 最久未使用的关键字。
      函数 get 和 put 必须以 O(1) 的平均时间复杂度运行。

    示例

    输入
    [“LRUCache”, “put”, “put”, “get”, “put”, “get”, “put”, “get”, “get”, “get”]
    [[2], [1, 1], [2, 2], [1], [3, 3], [2], [4, 4], [1], [3], [4]]
    输出 [null, null, null, 1, null, -1, null, -1, 3, 4]
    解释 LRUCache lRUCache = new LRUCache(2);
    lRUCache.put(1, 1); // 缓存是 {1=1}
    lRUCache.put(2, 2); // 缓存是 {1=1, 2=2}
    lRUCache.get(1); // 返回 1
    lRUCache.put(3, 3); // 该操作会使得关键字 2 作废，缓存是 {1=1, 3=3}
    lRUCache.get(2); // 返回 -1 (未找到)
    lRUCache.put(4, 4); // 该操作会使得关键字 1 作废，缓存是 {4=4, 3=3}
    lRUCache.get(1); // 返回 -1 (未找到)
    lRUCache.get(3); // 返回 3
    lRUCache.get(4); // 返回 4

    题解

    方法一

    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}

class Solution {

    方法二

    • 分析
      • 迭代
      • 手动模拟stack
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        stack<TreeNode*>  stk;
        vector<int> ans;
        while (root != nullptr || !stk.empty())
        {
            while (root != nullptr)
            {
                stk.push(root);
                root = root->left;
            }
            root = stk.top();
            stk.pop();
            ans.push_back(root->val);
            root = root->right;
        }
        return ans;
    }
};

    方法三

    • 分析
      • 中序遍历需要栈的原因是等编译完左子树之后无法回到root
      • 可以自己在左子树的最后一个节点处建立到root的连接
      • ！第二次遍历到root节点的时候，如果不做处理，寻找左子树最后一个节点的操作就会一直进行下去，陷入死循环
      • 最后记得把临时建立的连接销毁
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> ans;
        while (root != nullptr)
        {
            if (root->left == nullptr)
            {
                ans.push_back(root->val);
                root = root->right;
            }
            else
            {
                TreeNode* temp = root->left;
                while (temp->right != nullptr && temp->right != root)
                {
                    temp = temp->right;
                }
                if (temp->right == nullptr)
                {
                    temp->right = root;
                    root = root->left;
                }
                else if (temp->right == root)
                {
                    temp->right = nullptr;
                    ans.push_back(root->val);
                    root = root->right;
                }
            }
        }
        return ans;
    }
};

    解释：链表数组如下：
    [
    1->4->5,
    1->3->4,
    2->6
    ]
    将它们合并到一个有序链表中得到。
    1->1->2->3->4->4->5->6

    示例二

    输入：lists = []
    输出：[]

    示例三

    输入：lists = [[]]
    输出：[]

    提示

    • k == lists.length
    • 0 <= k <= 104
    • 0 <= lists[i].length <= 500
    • -104 <= lists[i][j] <= 104
    • lists[i] 按 升序 排列
    • lists[i].length 的总和不超过 104

    题解

    方法一

    • 分析
      • 优先级队列
    • 提交结果
      • 通过
    /**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        ListNode* head = new ListNode(0);
        auto cmp = [](ListNode* left, ListNode* right) { return left->val > right->val; };
        priority_queue<ListNode*, vector<ListNode*>, decltype(cmp)> pq(cmp);
        for (auto list: lists)
        {
            if (list != nullptr)
                pq.push(list);
        }
        ListNode* ptr = head;
        while (pq.size())
        {
            ptr->next = pq.top();
            pq.pop();
            if (ptr->next->next != nullptr)
            {
                pq.push(ptr->next->next);
            }
            ptr = ptr->next;
        }
        return head->next;
    }
};

    方法二

    • 分析
      • 相邻的两个列表合并，一直合并到只剩一个列表
    • 提交结果
      • 通过
    /**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode *a, ListNode *b) {
        if ((!a) || (!b)) return a ? a : b;
        ListNode head, *tail = &head, *aPtr = a, *bPtr = b;
        while (aPtr && bPtr)
        {
            if (aPtr->val < bPtr->val) {
                tail->next = aPtr;
                aPtr = aPtr->next;
            }
            else
            {
                tail->next = bPtr;
                bPtr = bPtr->next;
            }
            tail = tail->next;
        }
        tail->next = (aPtr ? aPtr : bPtr);
        return head.next;
    }

    ListNode* merge(vector <ListNode*> &lists, int l, int r) {
        if (l == r) return lists[l];
        if (l > r) return nullptr;
        int mid = (l + r) >> 1;
        return mergeTwoLists(merge(lists, l, mid), merge(lists, mid + 1, r));
    }

    ListNode* mergeKLists(vector<ListNode*>& lists) {
        return merge(lists, 0, lists.size() - 1);
    }
};
