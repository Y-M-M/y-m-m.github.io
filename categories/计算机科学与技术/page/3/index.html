    class Solution {

    bool valid(const string& str)
    {
        int balance = 0;
        for (char c : str)
        {
            if (c == '(')
            {
                ++balance;
            }
            else
            {
                --balance;
            }
            if (balance < 0)
            {
                return false;
            }
        }
        return balance == 0;
    }

    void generate_all(string& current, int n, vector<string>& result)
    {
        if (n == current.size())
        {
            if (valid(current))
            {
                result.push_back(current);
            }
            return;
        }
        current += '(';
        generate_all(current, n, result);
        current.pop_back();
        current += ')';
        generate_all(current, n, result);
        current.pop_back();
    }

    vector<string> generateParenthesis(int n) {
        vector<string> result;
        string current;
        generate_all(current, n * 2, result);
        return result;
    }
};

    bool search(vector<vector<char>>& board, const string& word, int pos, vector<pair<int, int>>& nowposs)
    {

                    }
                }

            vector<pair<int, int>> nextposss;
            for (pair<int, int> apos : nowposs)
            {
                int i = apos.first;

                if (i > 0 && board[i - 1][j] == word[pos])
                {
                    if (!visited[i - 1][j])
                    {
                        nextposss.push_back(make_pair(i - 1, j));
                        visited[i - 1][j] = true;
                    }
                }
                if (i < m - 1 && board[i + 1][j] == word[pos])
                {
                    if (!visited[i + 1][j])
                    {
                        nextposss.push_back(make_pair(i + 1, j));
                        visited[i + 1][j] = true;
                    }
                }
                if (j > 0 && board[i][j - 1] == word[pos])

                    if (!visited[i][j - 1])

    class Solution {

    void spreadroot(vector<vector<int>>& grid, vector<vector<int>>& times, int i, int j)
    {
        if (i > 0 && grid[i - 1][j] == 1)
        {
            if (times[i - 1][j] < 0)
            {
                times[i - 1][j] = times[i][j] + 1;
                rotted.push(make_pair(i - 1, j));
            }
            else
            {
                if (times[i][j] + 1 < times[i - 1][j])
                {
                    times[i - 1][j] = times[i][j] + 1;
                    rotted.push(make_pair(i - 1, j));
                }
            }
        }
        if (i < m - 1 && grid[i + 1][j] == 1)
        {
            if (times[i + 1][j] < 0)
            {
                times[i + 1][j] = times[i][j] + 1;
                rotted.push(make_pair(i + 1, j));
            }
            else
            {
                if (times[i][j] + 1 < times[i + 1][j])
                {
                    times[i + 1][j] = times[i][j] + 1;
                    rotted.push(make_pair(i + 1, j));
                }
            }
        }
        if (j > 0 && grid[i][j - 1] == 1)
        {
            if (times[i][j - 1] < 0)
            {
                times[i][j - 1] = times[i][j] + 1;
                rotted.push(make_pair(i, j - 1));
            }
            else
            {
                if (times[i][j] + 1 < times[i][j - 1])
                {
                    times[i][j - 1] = times[i][j] + 1;
                    rotted.push(make_pair(i, j - 1));
                }
            }
        }
        if (j < n - 1 && grid[i][j + 1] == 1)
        {
            if (times[i][j + 1] < 0)
            {
                times[i][j + 1] = times[i][j] + 1;
                rotted.push(make_pair(i, j + 1));
            }
            else
            {
                if (times[i][j] + 1 < times[i][j + 1])
                {
                    times[i][j + 1] = times[i][j] + 1;
                    rotted.push(make_pair(i, j + 1));
                }
            }
        }
    }

    int orangesRotting(vector<vector<int>>& grid) {
        m = grid.size();
        if (!m)
        {
            return 0;
        }
        n = grid[0].size();
        if (!n)
        {
            return 0;
        }
        vector<vector<int>> times(m, vector<int>(n, -1));
        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (grid[i][j] == 2)
                {
                    rotted.push(make_pair(i, j));
                    times[i][j] = 0;
                }
            }
        }
        while (!rotted.empty())
        {
            auto rotfruit = rotted.front();
            rotted.pop();
            spreadroot(grid, times, rotfruit.first, rotfruit.second);
        }
        int maxtime = 0;
        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (grid[i][j] == 1)
                {
                    if (times[i][j] < 0)
                    {
                        return -1;
                    }
                    else
                    {
                        maxtime = max(maxtime, times[i][j]);
                    }
                }
            }
        }
        return maxtime;
    }

    int m;
    int n;
