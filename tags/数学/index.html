    leetcode hot100 279 完全平方数

    题目

    给你一个整数 n ，返回 和为 n 的完全平方数的最少数量 。
    完全平方数 是一个整数，其值等于另一个整数的平方；换句话说，其值等于一个整数自乘的积。例如，1、4、9 和 16 都是完全平方数，而 3 和 11 不是。

    示例

    示例一

    输入：n = 12
    输出：3
    解释：12 = 4 + 4 + 4

    示例二

    输入：n = 13
    输出：2
    解释：13 = 4 + 9

    提示

    • 1 <= n <= 104

    题解

    方法一

    • 分析
      • 当i是完全平方数的时候，num[i] = 1
      • 当i不是完全平方数的时候，num[i] = min(num[j] + num[i - j])
    • 提交结果
      • 通过
    • 复杂度
      • 时间复杂度O(n2)
      • 空间复杂度O(n)
    class Solution {
public:
    int numSquares(int n) {
        if (n == 1) return 1;
        vector<int> num(n + 1, INT32_MAX);
        for (int i = 1; i * i <= n; i++)
        {
            num[i * i] = 1;
        }
        for (int i = 2; i <= n; i++)
        {
            if (num[i] == INT32_MAX)
            {
                for (int j = 1; j <= i / 2; j++)
                {
                    num[i] = min(num[j] + num[i - j], num[i]);
                }
            }
        }
        return num[n];
    }
};

    方法二

    • 分析
      • 一个数学定理可以帮助解决本题：「四平方和定理」。
      • 四平方和定理证明了任意一个正整数都可以被表示为至多四个正整数的平方和。这给出了本题的答案的上界。
      • 同时四平方和定理包含了一个更强的结论：当且仅当 n ≠ 4k×(8m+7) 时，n 可以被表示为至多三个正整数的平方和。因此，当 n=4k × (8m + 7) 时，n 只能被表示为四个正整数的平方和。此时我们可以直接返回 4。当 n ≠ 4k × (8m + 7) 时，我们需要判断到底多少个完全平方数能够表示 n，我们知道答案只会是 1,2,3 中的一个：
        • 答案为 1 时，则必有 n 为完全平方数，这很好判断；
        • 答案为 2 时，则有 n = a2 + b2，我们只需要枚举所有的 a(1 ≤ a ≤ n)，判断 n − a2是否为完全平方数即可；
        • 答案为 3 时，我们很难在一个优秀的时间复杂度内解决它，但我们只需要检查答案为 1 或 2 的两种情况，即可利用排除法确定答案。
    • 提交结果
      • 通过
    class Solution {
public:
    // 判断是否为完全平方数
    bool isPerfectSquare(int x) {
        int y = sqrt(x);
        return y * y == x;
    }

    // 判断是否能表示为 4^k*(8m+7)
    bool checkAnswer4(int x) {
        while (x % 4 == 0) {
            x /= 4;
        }
        return x % 8 == 7;
    }

    int numSquares(int n) {
        if (isPerfectSquare(n)) {
            return 1;
        }
        if (checkAnswer4(n)) {
            return 4;
        }
        for (int i = 1; i * i <= n; i++) {
            int j = n - i * i;
            if (isPerfectSquare(j)) {
                return 2;
            }
        }
        return 3;
    }
};

    题目

    假设你正在爬楼梯。需要 n 阶你才能到达楼顶。
    每次你可以爬 1 或 2 个台阶。你有多少种不同的方法可以爬到楼顶呢？

    示例

    示例一

    输入：n = 2
    输出：2
    解释：有两种方法可以爬到楼顶。

    1. 1 阶 + 1 阶
    2. 2 阶

    示例二

    输入：n = 3
    输出：3
    解释：有三种方法可以爬到楼顶。

    1. 1 阶 + 1 阶 + 1 阶
    2. 1 阶 + 2 阶
    3. 2 阶 + 1 阶

    提示

    • 1 <= n <= 45

    题解

    方法一

    • 分析
      • 递归
      • 上到第i个台阶，可通过i - 1台阶，也可通过i - 2台阶，methods[i] = methods[i - 1] + methods[i - 2]
      • 上到台阶1仅有一种方法，上到台阶2有两种方法
    • 提交结果
      • 通过
    • 复杂度
      • 时间复杂度O(n)
      • 空间复杂度O(1)
    class Solution {
public:
    int climbStairs(int n) {
        if (n == 1) return 1;
        int x = 1, y = 2, z = 0;
        for (int i = 0; i < n - 2; i++)
        {
            z = x + y;
            x = y;
            y = z;
        }
        return y;
    }
};

    方法一

    /**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        int carry = 0;
        ListNode* head = l2;
        ListNode* l2node = nullptr;
        while (l1 != nullptr && l2 != nullptr)
        {
            if (l2->next == nullptr)
            {
                l2node = l2;
            }
            int temp = l1->val + l2->val + carry;
            carry = temp / 10;
            l2->val = temp % 10;
            l1 = l1->next;
            l2 = l2->next;
        }
        if (l1 != nullptr)
        {
            l2 = l1;
            l2node->next = l1;
        }
        ListNode* tail;
        tail = l2node;
        while ((l2 != nullptr) && carry)
        {
            if (l2->next == nullptr)
            {
                tail = l2;
            }
            int temp = l2->val + carry;
            l2->val = temp % 10;
            carry = temp / 10;
            l2 = l2->next;
        }
        if (carry)
        {
            ListNode* newnode = new ListNode(carry);
            tail->next = newnode;
        }
        return head;
    }
};

    示例

    示例一

    输出：[[7,4,1],[8,5,2],[9,6,3]]

    示例二

    • 1 <= n <= 20

    提交结果

    • 分析
      • swap
      • 对于每一圈的0～n - 2的每一个数字，交换3次即可全部放到指定位置
    • 提交结果
      • 通过
    class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int cnt = 0;
        while (n > 1)
        {
            for (int i = 0; i < n - 1; i++)
            {
                swap(matrix[cnt][cnt + i], matrix[cnt + i][cnt + n - 1]);
                swap(matrix[cnt][cnt + i], matrix[cnt + n - 1][cnt + n - i - 1]);
                swap(matrix[cnt][cnt + i], matrix[cnt + n - i - 1][cnt]);
            }
            cnt++;
            n -= 2;
        }
    }
};

    题目

    给定一个整数数组 nums，将数组中的元素向右轮转 k 个位置，其中 k 是非负数。

    输出：[3,99,-1,-100]
    解释:

    • 向右轮转 1 步: [99,-1,-100,3]
    • 向右轮转 2 步: [3,99,-1,-100]

    提示

    • 1 <= nums.length <= 105
    • -231 <= nums[i] <= 231 - 1
    • 0 <= k <= 105

    题解

    方法一

    • 分析
      • 直接移动
      • 利用额外的空间vector t(k)存储会被覆盖的值
    • 提交结果

    class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        vector<int> t(k);
        int n = nums.size();
        k = k % n;
        for (int i = 0; i < k; i++)
        {
            t[i] = nums[i];
            nums[i] = nums[i + n - k];
        }
        for (int i = n - 1; i >= 2 * k; i--)
        {
            nums[i] = nums[i - k]; 
        }
        for (int i = 2 * k - 1; i >= k; i--)
        {
            if (i >= n)
            {
                continue;
            }
            nums[i] = t[i - k];
        }
    }
};

    方法二

    • 分析
      • 环状替换
      • 用到了最大公因数函数gcd()，交换函数swap(a, b)
    • 提交结果
      • 通过
    • 数学分析
      • 见参考资料
    class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        int n = nums.size();
        k = k % n;
        int count = gcd(k, n);
        for (int start = 0; start < count; start++)
        {
            int current = start;
            int prev = nums[start];
            do {
                int next = (current + k) % n;
                swap(nums[next], prev);
                current = next;
            } while (start != current);
        }
    }
};

    class Solution {
public:
    void reserve(vector<int>& nums, int start, int end)
    {
        while (start < end)
        {
            swap(nums[start], nums[end]);
            start += 1;
            end -= 1;
        }
    }
    void rotate(vector<int>& nums, int k) {
        k %= nums.size();
        reserve(nums, 0, nums.size() - 1);
        reserve(nums, 0, k - 1);
        reserve(nums, k, nums.size() - 1);
    }
};

        for (int i = 0; i < k; i++)

            int minnum = nums[0];

    class Solution {
public:
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        auto cmp = [&nums](int left, int right) {return nums[left] > nums[right];};
        priority_queue<int, vector<int>, decltype(cmp)> id(cmp);

        for (int i = 0; i < nums.size(); i++)
        {
            id.push(i);
        }
        for (int i = 0; i < k; i++)
        {
            int updatetarget = id.top();
            id.pop();
            nums[updatetarget] *= multiplier;
            id.push(updatetarget);
        }
        return nums;
    }
};
    • 提交结果
      • 一：584/373个通过的测试样例
    • 分析
      • 优先级队列
      • 错误原因
        • 先pop后push，破坏了排序的稳定性，每次取出的最小元素不是按顺序排列的第一个

    class Solution {
public:
    long long quickMul(long long x, long long y, long long m)
    {
        long long res = 1;
        while (y)
        {
            if (y & 1) {
                res = (res * x) % m;
            }
            y >>= 1;
            x = (x * x) % m;
        }
        return res;
    }
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        if (multiplier == 1)
        {
            return nums;
        }
        long long n = nums.size(), m = 1e9 + 7;
        long long mx = *max_element(nums.begin(), nums.end()); // 取最大值
        vector<pair<long long, int>> v(n);
        for (int i = 0; i < n; i++)
        {
            v[i] = {nums[i], i};
        }
        make_heap(v.begin(), v.end(), greater<>());
        for (; v[0].first < mx && k; k--)
        {
            pop_heap(v.begin(), v.end(), greater<>());
            v.back().first *= multiplier;
            push_heap(v.begin(), v.end(), greater<>());
        }
        sort(v.begin(), v.end());
        for (int i = 0; i < n; i++)
        {
            int t = k / n + (i < k % n);
            nums[v[i].second] = ((v[i].first % m) * quickMul(multiplier, t, m)) % m;
        }
        return nums;
    }
};
    • 提交结果
      • 一：通过
      • 二：通过
    • 分析
      • 通过pair的排序规则来进行堆排序，先比较first，first相同的时候在比较second
      • 对过程进行优化
        • 如果nums中所有数都被更新过了，那么此后更新会从小到大把nums中的所有数都*k
        • 这个过程可直接遍历进行，无需再进行堆排序
