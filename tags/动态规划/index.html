    leetcode hot100 55 跳跃游戏

    题目

    给你一个非负整数数组 nums ，你最初位于数组的 第一个下标 。数组中的每个元素代表你在该位置可以跳跃的最大长度。
    判断你是否能够到达最后一个下标，如果可以，返回 true ；否则，返回 false 。

    示例

    示例一

    输入：nums = [2,3,1,1,4]
    输出：true
    解释：可以先跳 1 步，从下标 0 到达下标 1, 然后再从下标 1 跳 3 步到达最后一个下标。

    示例二

    输入：nums = [3,2,1,0,4]
    输出：false
    解释：无论怎样，总会到达下标为 3 的位置。但该下标的最大跳跃长度是 0 ， 所以永远不可能到达最后一个下标。

    提示

    1 <= nums.length <= 104
    0 <= nums[i] <= 105

    题解

    方法一

    • 分析
      • 从0开始，遍历数组，标记能够跳跃到的节点
    class Solution {
public:
    bool canJump(vector<int>& nums) {
        int n = nums.size();
        vector<bool> isaccess(n, false);
        isaccess[0] = true;
        for (int i = 0; i < n; i++)
        {
            if (isaccess[i])
            {
                for (int j = 1; j <= nums[i]; j++)
                {
                    if (i + j < n)
                    {
                        isaccess[i + j] = true;
                    }
                    else
                    {
                        break;
                    }
                }
            }
        }
        return isaccess.back();
    }
};

    方法二

    • 分析
      • 如果能跳到一个点，则这个点之前的所有点都能跳到
      • 只需要求能跳到的最远距离，就能知道它是否能到达最远点
    class Solution {
public:
    bool canJump(vector<int>& nums) {
        int k = 0;
        for (int i = 0; i < nums.size(); i++) {
            if (i > k) return false;
            k = max(k, i + nums[i]);
        }
        return true;
    }
};

    题目

    给定一个数组 prices ，它的第 i 个元素 prices[i] 表示一支给定股票第 i 天的价格。
    你只能选择 某一天 买入这只股票，并选择在 未来的某一个不同的日子 卖出该股票。设计一个算法来计算你所能获取的最大利润。
    返回你可以从这笔交易中获取的最大利润。如果你不能获取任何利润，返回 0 。

    示例

    示例一

    输入：[7,1,5,3,6,4]
    输出：5
    解释：在第 2 天（股票价格 = 1）的时候买入，在第 5 天（股票价格 = 6）的时候卖出，最大利润 = 6-1 = 5 。
    注意利润不能是 7-1 = 6, 因为卖出价格需要大于买入价格；同时，你不能在买入前卖出股票。

    示例二

    输入：prices = [7,6,4,3,1]
    输出：0
    解释：在这种情况下, 没有交易完成, 所以最大利润为 0。

    提示

    • 1 <= prices.length <= 105
    • 0 <= prices[i] <= 104

    题解

    方法一

    • 分析
      • 动态规划
    • 提交结果
      • 通过
    class Solution {
public:
    int maxProfit(vector<int>& prices) {
        if (!prices.size()) return 0;
        int ans = 0;
        int minprice = prices[0];
        for (int price : prices)
        {
            ans = max(price - minprice, ans);
            minprice = min(minprice, price);
        }
        return ans;
    }
};

    给你一个字符串 s，请你将 s 分割成一些 子串，使每个子串都是 回文串 。返回 s 所有可能的分割方案。

    输出：[[“a”,“a”,“b”],[“aa”,“b”]]

    示例二

    输入：s = “a”
    输出：[[“a”]]

    提示

    • 1 <= s.length <= 16
    • s 仅由小写英文字母组成

    题解

    方法一

    • 分析
      • 递归
      • 对于s中每个字母，记录以它开头的回文字符串的子串的结尾位置

    class Solution {
public:
    bool isreverse(const string &s, int i, int j)
    {
        if (i >= j) return true;
        if (isvalid[i][j] != -1)
        {
            return isvalid[i][j];
        }
        
        return s[i] == s[j] && isreverse(s, i + 1, j - 1);
    }

    void paritit(const string &s, vector<string> &solution, int nowpos)
    {
        // cout << nowpos << endl;
        if (nowpos >= s.size())
        {
            ans.push_back(solution);
        }
        else
        {
            // cout << "subend[nowpos].size() = " << subend[nowpos].size() << endl;
            for (int aend : subend[nowpos])
            {
                // cout << aend << endl;
                solution.push_back(s.substr(nowpos, aend - nowpos + 1));
                paritit(s, solution, aend + 1);
                solution.pop_back();
            }
        }
    }


    vector<vector<string>> partition(string s) {
        vector<string> solution;
        int n = s.length();
        for (int i = 0; i < n; i++)
        {
            pos[s[i]].push_back(i);
        }
        isvalid.resize(n);

        for (int i = 0; i < n; i++)
        {
            isvalid[i].resize(n);
            for (int j = 0; j < n; j++)
            {
                isvalid[i][j] = -1;
            }
        }
        for (char c = 'a'; c <= 'z'; c++)
        {
            for (int i = 0; i < pos[c].size(); i++)
            {
                for (int j = i; j < pos[c].size(); j++)
                {
                    // cout << "i = " << i << ", j = " << j << ", pos[c].size() = " << pos[c].size() << endl;
                    if (isreverse(s, pos[c][i], pos[c][j]))
                    {
                        // cout << "j = " << j << endl;
                        subend[pos[c][i]].push_back(pos[c][j]);
                    }
                }
            }
        }

        paritit(s, solution, 0);
        return ans;
    }
private: 
    unordered_map<char, vector<int>> pos;
    unordered_map<int, vector<int>> subend;
    vector<vector<string>> ans;
    vector<vector<int>> isvalid;
};

  • leetcode hot100 22 括号生成

    题目

    数字 n 代表生成括号的对数，请你设计一个函数，用于能够生成所有可能的并且 有效的 括号组合。

    示例

    示例一

    输入：n = 3
    输出：["((()))","(()())","(())()","()(())","()()()"]

    示例二

    输入：n = 1
    输出：["()"]

    提示

    • 1 <= n <= 8

    题解

    方法一

    • 分析
      • 暴力解法
      • 视作2n个字符的全排列，字符有’(’, ‘)‘两种情况
      • 合法字符串应满足的条件
        • ‘)‘不出现在’(‘之前 -> 在遍历字符串的过程中，不能出现balance < 0的情况
        • 左右括号成对出现 -> balance == 0
    • 提交结果
      • 通过
    class Solution {
public:
    bool valid(const string& str)
    {
        int balance = 0;
        for (char c : str)
        {
            if (c == '(')
            {
                ++balance;
            }
            else
            {
                --balance;
            }
            if (balance < 0)
            {
                return false;
            }
        }
        return balance == 0;
    }

    void generate_all(string& current, int n, vector<string>& result)
    {
        if (n == current.size())
        {
            if (valid(current))
            {
                result.push_back(current);
            }
            return;
        }
        current += '(';
        generate_all(current, n, result);
        current.pop_back();
        current += ')';
        generate_all(current, n, result);
        current.pop_back();
    }

    vector<string> generateParenthesis(int n) {
        vector<string> result;
        string current;
        generate_all(current, n * 2, result);
        return result;
    }
};

    方法二

    • 分析
      • 思路与方法一一致
      • 在方法一的基础上做了改进，在构造全排列的过程中判断已有序列是否合法
      • 只存储合法的字符串
    • 提交结果
      • 通过
    class Solution {
public:
    void backtrack(vector<string>& ans, string& cur, int open, int close, int n)
    {
        if (cur.size() == n * 2)
        {
            ans.push_back(cur);
            return;
        }

        if (open < n)
        {
            cur.push_back('(');
            backtrack(ans, cur, open + 1, close, n);
            cur.pop_back();
        }

        if (close < open)
        {
            cur.push_back(')');
            backtrack(ans, cur, open, close + 1, n);
            cur.pop_back();
        }
    }

    vector<string> generateParenthesis(int n) {
        vector<string> result;
        string current;
        backtrack(result, current, 0, 0, n);
        return result;
    }
};

    方法三

    • 分析
      • 递归
      • 将整个字符串分为两部分(a)b，递归求解a和b的所有可能性，再拼成原字符串
    • 提交结果
      • 通过
    class Solution {
public:
    shared_ptr<vector<string>> generate(int n)
    {
        if (cache[n] != nullptr)
        {
            return cache[n];
        }
        if (!n)
        {
            cache[0] = shared_ptr<vector<string>> (new vector<string> {""});
        }
        else
        {
            auto result = shared_ptr<vector<string>> (new vector<string>);
            for (int i = 0; i != n; i++)
            {
                auto lefts = generate(i);
                auto rights = generate(n - i - 1);
                for (const string& left : *lefts)
                {
                    for (const string& right : *rights)
                    {
                        result->push_back("(" + left + ")" + right);
                    }
                }
            }
            cache[n] = result;
        }
        return cache[n];
    }

    vector<string> generateParenthesis(int n) {
        return *generate(n);
    }
private:
    shared_ptr<vector<string>> cache[100] = {nullptr};
};

    leetcode hot100 124 二叉树中的最大路径和

    题目

    二叉树中的 路径 被定义为一条节点序列，序列中每对相邻节点之间都存在一条边。同一个节点在一条路径序列中 至多出现一次 。该路径 至少包含一个 节点，且不一定经过根节点。
    路径和 是路径中各节点值的总和。
    给你一个二叉树的根节点 root ，返回其 最大路径和 。

    示例

    示例一


    输入：root = [1,2,3]
    输出：6
    解释：最优路径是 2 -> 1 -> 3 ，路径和为 2 + 1 + 3 = 6

    示例二

    输入：root = [-10,9,20,null,null,15,7]
    输出：42
    解释：最优路径是 15 -> 20 -> 7 ，路径和为 15 + 20 + 7 = 42

    提示

    • 树中节点数目范围是 [1, 3 * 104]
    • -1000 <= Node.val <= 1000

    题解

    方法一

    • 分析
      • dfs
      • 经过一个节点的最大路径和，为其左右子树最大路径和+root->val
      • 注意最右子树最大路径和如果为负数的话，就无需加该负数
      • 实质是在树上的最大子段和
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int dfs(TreeNode* root)
    {
        if (root == nullptr)
        {
            return 0;
        }
        int leftmax = dfs(root->left);
        int rightmax = dfs(root->right);
        int maxpath = 0;
        if (leftmax > 0)  maxpath += leftmax;
        if (rightmax > 0) maxpath += rightmax;
        maxpath += root->val;
        if (maxpath > ans)
        {
            ans = maxpath;
        }
        int addnum = max(leftmax, rightmax);
        if (addnum > 0)
            return addnum + root->val;
        return root->val;
        
    }

    int maxPathSum(TreeNode* root) {
        if (root == nullptr)
        {
            return 0;
        }
        ans = -INT32_MAX;
        dfs(root);
        return ans;
    }
private:
    int ans;
};

    leetcode hot100 53 最大子数组和

    题目

    给你一个整数数组 nums ，请你找出一个具有最大和的连续子数组（子数组最少包含一个元素），返回其最大和。
    子数组是数组中的一个连续部分。

    示例

    示例 1：

    输入：nums = [-2,1,-3,4,-1,2,1,-5,4]
    输出：6
    解释：连续子数组 [4,-1,2,1] 的和最大，为 6 。

    示例 2：

    输入：nums = [1]
    输出：1

    示例 3：

    输入：nums = [5,4,-1,7,8]
    输出：23

    提示

    • 1 <= nums.length <= 105
    • -104 <= nums[i] <= 104

    题解

    方法一

    • 分析
      • 动态规划
    • 提交结果
      • 通过
    class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int presum = nums[0], nowsum = 0;
        if (nums.size() == 1)
        {
            return nums[0];
        }
        int maxsum = nums[0];
        for (int i = 1; i < nums.size(); i++)
        {
            if (presum > 0)
            {
                nowsum = presum + nums[i];
            }
            else
            {
                nowsum = nums[i];
            }
            presum = nowsum;
            maxsum = nowsum > maxsum ? nowsum : maxsum;
        }
        return maxsum;
    }
};

    方法二

    • 分析
      • 线段树
      • 维护一棵树，根节点为[0, nums.size() - 1], 叶子结点为[i, i]
      • 每个节点存储lSum, rSum, mSum, iSum
      • 从叶子结点开始构建这棵树
    • 提交结果
      • 通过
    class Solution {
public:
    struct Status {
        int lSum, rSum, mSum, iSum;
    };

    Status pushUp (Status l, Status r)
    {
        int iSum = l.iSum + r.iSum;
        int lSum = max(l.lSum, l.iSum + r.lSum);
        int rSum = max(r.rSum, r.iSum + l.rSum);
        int mSum = max(max(l.mSum, r.mSum), l.rSum + r.lSum);
        return (Status) { lSum, rSum, mSum, iSum };
    }

    Status get(vector<int> &a, int l, int r) {
        if (l == r) {
            return (Status) { a[l], a[l], a[l], a[l] };
        }
        int m = (l + r) >> 1;
        Status lSub = get(a, l, m);
        Status rSub = get(a, m + 1, r);
        return pushUp(lSub, rSub);
    }

    int maxSubArray(vector<int>& nums) {
        return get(nums, 0, nums.size() - 1).mSum;
    }
};

    leetcode hot100 42 接雨水

    题目

    给定 n 个非负整数表示每个宽度为 1 的柱子的高度图，计算按此排列的柱子，下雨之后能接多少雨水。

    示例

    示例一

    输入：height = [0,1,0,2,1,0,1,3,2,1,2,1]
    输出：6
    解释：上面是由数组 [0,1,0,2,1,0,1,3,2,1,2,1] 表示的高度图，在这种情况下，可以接 6 个单位的雨水（蓝色部分表示雨水）。

    示例二

    输入：height = [4,2,0,3,2,5]
    输出：9

    提示

    • n == height.length
    • 1 <= n <= 2 * 104
    • 0 <= height[i] <= 105

    题解

    方法一

    • 分析
      • int ans = 0;
      • 从左往右遍历一遍数组，每次增加一个柱子，计算这个柱子增加之后能多接的雨水
      • 每次接完雨水之后，将雨水的高度也视为柱子的高度，防止重复计算
    • 提交结果
      • 通过
    class Solution {
public:
    int trap(vector<int>& height) {
        int n = height.size();
        int ans = 0;
        if (height.size() == 0 || height.size() == 1)
            return 0;
        int maxh = max(height[0], height[1]);
        for (int i = 2; i < n; i++)
        {
            int nowhi = min(maxh, height[i]);
            maxh = max(maxh, height[i]);
            for (int j = i - 1; j >= 0; j--)
            {
                if (height[j] >= nowhi)
                {
                    break;
                }
                else
                {
                    ans += (nowhi - height[j]);
                    height[j] = nowhi;
                }
            }
        }
        return ans;
    }
};

    解释：3 x 3 网格的正中间有一个障碍物。
    从左上角到右下角一共有 2 条不同的路径：

    1. 向右 -> 向右 -> 向下 -> 向下
    2. 向下 -> 向下 -> 向右 -> 向右

    示例二

    输入：obstacleGrid = [[0,1],[0,0]]
    输出：1

    提示

    • m == obstacleGrid.length
    • n == obstacleGrid[i].length
    • 1 <= m, n <= 100
    • obstacleGrid[i][j] 为 0 或 1

    题解

    class Solution {
public:
    int uniquePathsWithObstacles(vector<vector<int>>& obstacleGrid) {
        int m = obstacleGrid.size();
        int n = obstacleGrid[0].size();
        vector<vector<int>> roads(m, vector<int> (n, 0));
        for (int i = 0; i < m; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (obstacleGrid[i][j] > 0)
                {
                    roads[i][j] = 0;
                }
                else
                {
                    if (!i && !j)
                    {
                        roads[i][j] = 1;
                    }
                    else if (!i)
                    {
                        roads[i][j] = roads[i][j - 1];
                    }
                    else if (!j)
                    {
                        roads[i][j] = roads[i - 1][j];
                    }
                    else
                    {
                        roads[i][j] = roads[i - 1][j] + roads[i][j - 1];
                    }
                }
            }
        }
        return roads[m - 1][n - 1];
    }
};
