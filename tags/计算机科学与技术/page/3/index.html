    vector<int> smallestRange(vector<vector<int>>& nums) {
        vector<int> region(nums.size());

        {
            region[i] = 0;
        }
        int times = 0;
        int ans; // åŒºåŸŸçš„é•¿åº¦
        int l, r; // åŒºåŸŸçš„å·¦å³è¾¹ç•Œ

                break;

    void exchange(const int& i, const int & j, vector<int>& rows)
    {

    int movesToChessboard(vector<vector<int>>& board) {
        int n = board.size();
        vector<int> rows(n, 0);
        int ans = 0;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                rows[i] ^= (board[i][j] << j);

        }

                    cnt++;

                    for (int i = 1; i < n; i++)
                    {
                        if ((rows[i] & rows[0]) == 0)

                            exchange(0, i, rows);
                            ans++;
                        }
                    }

            }

                    if (j != i + 1)

                        exchange(i + 1, j, rows);
                        ans++;

                }
