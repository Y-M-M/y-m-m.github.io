    /**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        int carry = 0;
        ListNode* head = l2;
        ListNode* l2node = nullptr;
        while (l1 != nullptr && l2 != nullptr)
        {
            if (l2->next == nullptr)
            {
                l2node = l2;
            }
            int temp = l1->val + l2->val + carry;
            carry = temp / 10;
            l2->val = temp % 10;
            l1 = l1->next;
            l2 = l2->next;
        }
        if (l1 != nullptr)
        {
            l2 = l1;
            l2node->next = l1;
        }
        ListNode* tail;
        tail = l2node;
        while ((l2 != nullptr) && carry)
        {
            if (l2->next == nullptr)
            {
                tail = l2;
            }
            int temp = l2->val + carry;
            l2->val = temp % 10;
            carry = temp / 10;
            l2 = l2->next;
        }
        if (carry)
        {
            ListNode* newnode = new ListNode(carry);
            tail->next = newnode;
        }
        return head;
    }
};

    class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        int n = matrix.size();
        int cnt = 0;
        while (n > 1)
        {
            for (int i = 0; i < n - 1; i++)
            {
                swap(matrix[cnt][cnt + i], matrix[cnt + i][cnt + n - 1]);
                swap(matrix[cnt][cnt + i], matrix[cnt + n - 1][cnt + n - i - 1]);
                swap(matrix[cnt][cnt + i], matrix[cnt + n - i - 1][cnt]);
            }
            cnt++;
            n -= 2;
        }
    }
};

    class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        vector<int> t(k);
        int n = nums.size();
        k = k % n;
        for (int i = 0; i < k; i++)
        {
            t[i] = nums[i];
            nums[i] = nums[i + n - k];
        }
        for (int i = n - 1; i >= 2 * k; i--)
        {
            nums[i] = nums[i - k]; 
        }
        for (int i = 2 * k - 1; i >= k; i--)
        {
            if (i >= n)
            {
                continue;
            }
            nums[i] = t[i - k];
        }
    }
};

    class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        int n = nums.size();
        k = k % n;
        int count = gcd(k, n);
        for (int start = 0; start < count; start++)
        {
            int current = start;
            int prev = nums[start];
            do {
                int next = (current + k) % n;
                swap(nums[next], prev);
                current = next;
            } while (start != current);
        }
    }
};

public:
    void reserve(vector<int>& nums, int start, int end)
    {

        {
            swap(nums[start], nums[end]);
            start += 1;
            end -= 1;
        }
    }
    void rotate(vector<int>& nums, int k) {
        k %= nums.size();
        reserve(nums, 0, nums.size() - 1);
        reserve(nums, 0, k - 1);
        reserve(nums, k, nums.size() - 1);

    class Solution {
public:
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        auto cmp = [&nums](int left, int right) {return nums[left] > nums[right];};
        priority_queue<int, vector<int>, decltype(cmp)> id(cmp);

        for (int i = 0; i < nums.size(); i++)
        {
            id.push(i);
        }
        for (int i = 0; i < k; i++)
        {
            int updatetarget = id.top();
            id.pop();
            nums[updatetarget] *= multiplier;
            id.push(updatetarget);
        }
        return nums;
    }
};
    • 提交结果
      • 一：584/373个通过的测试样例
    • 分析
      • 优先级队列
      • 错误原因
        • 先pop后push，破坏了排序的稳定性，每次取出的最小元素不是按顺序排列的第一个

    class Solution {
public:
    long long quickMul(long long x, long long y, long long m)
    {
        long long res = 1;
        while (y)
        {
            if (y & 1) {
                res = (res * x) % m;
            }
            y >>= 1;
            x = (x * x) % m;
        }
        return res;
    }
    vector<int> getFinalState(vector<int>& nums, int k, int multiplier) {
        if (multiplier == 1)
        {
            return nums;
        }
        long long n = nums.size(), m = 1e9 + 7;
        long long mx = *max_element(nums.begin(), nums.end()); // 取最大值
        vector<pair<long long, int>> v(n);
        for (int i = 0; i < n; i++)
        {
            v[i] = {nums[i], i};
        }
        make_heap(v.begin(), v.end(), greater<>());
        for (; v[0].first < mx && k; k--)
        {
            pop_heap(v.begin(), v.end(), greater<>());
            v.back().first *= multiplier;
            push_heap(v.begin(), v.end(), greater<>());
        }
        sort(v.begin(), v.end());
        for (int i = 0; i < n; i++)
        {
            int t = k / n + (i < k % n);
            nums[v[i].second] = ((v[i].first % m) * quickMul(multiplier, t, m)) % m;
        }
        return nums;
    }
};
    • 提交结果
      • 一：通过
      • 二：通过
    • 分析
      • 通过pair的排序规则来进行堆排序，先比较first，first相同的时候在比较second
      • 对过程进行优化
        • 如果nums中所有数都被更新过了，那么此后更新会从小到大把nums中的所有数都*k
        • 这个过程可直接遍历进行，无需再进行堆排序

    class Solution {
public:
    void exchange(const int& i, const int & j, vector<int>& rows)
    {
        int t = rows[i];
        rows[i] = rows[j];
        rows[j] = t;
    }

    int movesToChessboard(vector<vector<int>>& board) {
        int n = board.size();
        vector<int> rows(n, 0);
        int ans = 0;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                rows[i] ^= (board[i][j] << j);
            }
        }
        if (n % 2)
        {
            int cnt = 0;
            for (int i = 1; i < n; i++)
            {
                if ((rows[i] & rows[0]) == 0)
                {
                    cnt++;
                }
                if (cnt % 2)
                {
                    for (int i = 1; i < n; i++)
                    {
                        if ((rows[i] & rows[0]) == 0)
                        {
                            exchange(0, i, rows);
                            ans++;
                        }
                    }
                }
            }
        }
        for (int i = 0; i < n - 1; i++)
        {
            int j;
            for (j = i + 1; j < n; j++)
            {
                if ((rows[i] & rows[j]) == 0)
                {
                    if (j != i + 1)
                    {
                        exchange(i + 1, j, rows);
                        ans++;
                    }
                    break;
                }
            }
            if (j == n)
                return -1;
        }
        vector<int> cols(n, 0);
        for (int j = 0; j < n; j++)
        {
            for (int i = 0; i < n; i++)
            {
                cols[j] ^= (board[i][j] << i);
            }
        }
        if (n % 2)
        {
            int cnt = 0;
            for (int i = 1; i < n; i++)
            {
                if ((cols[i] & cols[0]) == 0)
                {
                    cnt++;
                }
                if (cnt % 2)
                {
                    for (int i = 1; i < n; i++)
                    {
                        if ((cols[i] & cols[0]) == 0)
                        {
                            exchange(0, i, cols);
                            ans++;
                        }
                    }
                }
            }
        }
        for (int i = 0; i < n - 1; i++)
        {
            int j;
            for (j = i + 1; j < n; j++)
            {
                if ((cols[i] & cols[j]) == 0)
                {
                    if (j != i + 1)
                    {
                        exchange(i + 1, j, cols);
                        ans++;
                    }
                    break;
                }
            }
            if (j == n)
                return -1;
        }
        return ans;
    }
};

    • 思路
      • 以一个int类型的整数表示矩阵的一行/一列
      • 行与列的交换可以分开操作，不会相互影响

    class Solution {
public:
    int getMoves(int mask, int count, int n) {
        int ones = __builtin_popcount(mask);
        if (n & 1) {
            /* 如果 n 为奇数，则每一行中 1 与 0 的数目相差为 1，且满足相邻行交替 */
            if (abs(n - 2 * ones) != 1 || abs(n - 2 * count) != 1 ) {
                return -1;
            }
            if (ones == (n >> 1)) {
                /* 偶数位变为 1 的最小交换次数 */
                return n / 2 - __builtin_popcount(mask & 0xAAAAAAAA);
            } else {
                /* 奇数位变为 1 的最小交换次数 */
                return (n + 1) / 2 - __builtin_popcount(mask & 0x55555555);
            }
        } else { 
            /* 如果 n 为偶数，则每一行中 1 与 0 的数目相等，且满足相邻行交替 */
            if (ones != (n >> 1) || count != (n >> 1)) {
                return -1;
            }
            /* 偶数位变为 1 的最小交换次数 */
            int count0 = n / 2 - __builtin_popcount(mask & 0xAAAAAAAA);
            /* 奇数位变为 1 的最小交换次数 */
            int count1 = n / 2 - __builtin_popcount(mask & 0x55555555);  
            return min(count0, count1);
        }
    }

    int movesToChessboard(vector<vector<int>>& board) {
        int n = board.size();
        int rowMask = 0, colMask = 0;        

        /* 检查棋盘的第一行与第一列 */
        for (int i = 0; i < n; i++) {
            rowMask |= (board[0][i] << i);
            colMask |= (board[i][0] << i);
        }
        int reverseRowMask = ((1 << n) - 1) ^ rowMask;
        int reverseColMask = ((1 << n) - 1) ^ colMask;
        int rowCnt = 0, colCnt = 0;
        for (int i = 0; i < n; i++) {
            int currRowMask = 0;
            int currColMask = 0;
            for (int j = 0; j < n; j++) {
                currRowMask |= (board[i][j] << j);
                currColMask |= (board[j][i] << j);
            }
            /* 检测每一行的状态是否合法 */
            if (currRowMask != rowMask && currRowMask != reverseRowMask) {
                return -1;
            } else if (currRowMask == rowMask) {
                /* 记录与第一行相同的行数 */
                rowCnt++;
            }
            /* 检测每一列的状态是否合法 */
            if (currColMask != colMask && currColMask != reverseColMask) {
                return -1;
            } else if (currColMask == colMask) {
                /* 记录与第一列相同的列数 */
                colCnt++;
            }
        }
        int rowMoves = getMoves(rowMask, rowCnt, n);
        int colMoves = getMoves(colMask, colCnt, n);
        return (rowMoves == -1 || colMoves == -1) ? -1 : (rowMoves + colMoves); 
    }
};

      • 合法矩阵
        • 其他所有行/列都与第一行/列相同或相反
        • 每一行/列中0的个数与1的个数的差的绝对值为0或1
      • 最小交换次数

    leetcode 683 平方数之和

    题目

    给定一个非负整数 c ，你要判断是否存在两个整数 a 和 b，使得 a2 + b2 = c 。

    示例

    示例一

    输入：c = 5 输出：true 解释：1 * 1 + 2 * 2 = 5

    示例二

    输入：c = 3 输出：false

    提示

    0 <= c <= 231 - 1

    题解

    方法一

    class Solution {
public:
    bool judgeSquareSum(int c) {
        unordered_set<int> little_squares;
        long long base = 0;
        long long square = base * base;
        while (square <= c / 2) {
            little_squares.insert(square);
            if (square + square == c) return true;
            base++;
            square = base * base;
        }
        while (square <= c) {
            if (little_squares.count(c - square)) return true;
            base++;
            square = base * base;
        }
        return false;
    }
};

    • 提交结果

      • 通过

    • 分析

      • 思路
        • 计算不超过c/2的所有平方数，存入unordered_set中，便于查找
        • 计算超过c/2且不超过c的所有平方数x，在unordered_set中查找是否有c - x，如果有，返回true
        • 单独处理c/2刚好是平方数的情况
      • 时间复杂度 O(sqrt(n))
        • unordered_map()节省了查找的时间
      • 空间复杂度 O(sqrt(n))

    class Solution {
public:
    bool judgeSquareSum(int c) {
        long left = 0;
        long right = (int)sqrt(c);
        while (left <= right) {
            long sum = left * left + right * right;
            if (sum == c)
                return true;
            else if (sum > c)
                right--;
            else
                left++;
        }
        return false;
    }
};

    class Solution {
public:
    bool judgeSquareSum(int c) {
        for (int base = 2; base * base <= c; base++) {
            if (c % base != 0) continue;
            int exp = 0;
            while (c % base == 0) // 计算base的幂
            {
                c /= base;
                exp++;
            }
            if (base % 4 == 3 && exp % 2 != 0) {
                return false;
            }
        }
        return c % 4 != 3;
    }
};
    • 提交结果
      • 通过
    • 分析
      • 数学
      • 费马平方和定理
      • 一个非负整数 c 如果能够表示为两个整数的平方和，当且仅当 c 的所有形如 4k+3 的质因子的幂均为偶数。
      • 依次遍历所有质因子
        • 每次都让c /= base，则遍历到的因子一定是质因子(如果因子除了1和它本身之外还有其他因子，则一定会在前面被除掉)
      • 对每个质因子计算幂，判读它是否符合条件
      • 时间复杂度 O(sqrt(n))
      • 空间复杂度 O(1)
