    leetcode hot100 34 在排序数组中查找元素的第一个和最后一个位置

    题目

    给你一个按照非递减顺序排列的整数数组 nums，和一个目标值 target。请你找出给定目标值在数组中的开始位置和结束位置。
    如果数组中不存在目标值 target，返回 [-1, -1]。
    你必须设计并实现时间复杂度为 O(log n) 的算法解决此问题。

    示例

    示例一

    输入：nums = [5,7,7,8,8,10], target = 8
    输出：[3,4]

    示例二

    输入：nums = [5,7,7,8,8,10], target = 6
    输出：[-1,-1]

    示例三

    输入：nums = [], target = 0
    输出：[-1,-1]

    提示

    • 0 <= nums.length <= 105
    • -109 <= nums[i] <= 109
    • nums 是一个非递减数组
    • -109 <= target <= 109

    题解

    • 分析
      • 二分查找找到元素出现的最后一个位置
      • 从该位置开始往前查找
    • 提交结果
      • 通过
    class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        vector<int> ans = {-1, -1};
        if (!nums.size()) return ans; 
        int l = 0, r = nums.size() - 1;
        while (l <= r)
        {
            int mid = (l + r) / 2;
            if (nums[mid] <= target)
            {
                l = mid + 1;
            }
            else
            {
                r = mid - 1;
            }
        }
        if (l == 0)
        {
            if (nums[0] == target)
            {
                ans[0] = 0;
                ans[1] = 0;
            }
            return ans;
        }
        if (nums[l - 1] == target)
        {
            ans[1] = l - 1;
            l--;
            while (l >= 0 && nums[l] == target) l--;
            ans[0] = l + 1;
        }
        return ans;
    }
};

    class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        if (nums.size() == 0)
        {
            return 0;
        }
        int l = 0, r = nums.size() - 1;
        while (l <= r)
        {
            int mid = (l + r) / 2;
            if (nums[mid] <= target)
            {
                l = mid + 1;
            }
            else
            {
                r = mid - 1;
            }
        }
        if (!l)
        {
            return 0;
        }
        if (nums[l - 1] == target)
        {
            return l - 1;
        }
        return l;
    }
};

    leetcode hot100 74 搜索二维矩阵

    题目

    给你一个满足下述两条属性的 m x n 整数矩阵：

    • 每行中的整数从左到右按非严格递增顺序排列。
    • 每行的第一个整数大于前一行的最后一个整数。
      给你一个整数 target ，如果 target 在矩阵中，返回 true ；否则，返回 false 。

    示例

    示例一


    输入：matrix = [[1,3,5,7],[10,11,16,20],[23,30,34,60]], target = 3
    输出：true

    示例二


    输入：matrix = [[1,3,5,7],[10,11,16,20],[23,30,34,60]], target = 13
    输出：false

    提示

    • m == matrix.length
    • n == matrix[i].length
    • 1 <= m, n <= 100
    • -104 <= matrix[i][j], target <= 104

    题解

    方法一

    • 分析
      • 将二维数组的索引i, j -> i * n + j
      • 二分查找
    • 提交结果
      • 通过
    class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        int m = matrix.size();
        if (!m) return false;
        int n = matrix[0].size();
        int l = 0, r = m * n - 1;
        while (l <= r)
        {
            int mid = (l + r) / 2;
            int nownum = matrix[mid / n][mid % n];
            if (nownum == target) return true;
            else if (nownum < target)
            {
                l = mid + 1;
            }
            else
            {
                r = mid - 1;
            }
        }
        return false;
    }
};

    题目

    按照国际象棋的规则，皇后可以攻击与之处在同一行或同一列或同一斜线上的棋子。
    n 皇后问题 研究的是如何将 n 个皇后放置在 n×n 的棋盘上，并且使皇后彼此之间不能相互攻击。
    给你一个整数 n ，返回所有不同的 n 皇后问题 的解决方案。
    每一种解法包含一个不同的 n 皇后问题 的棋子放置方案，该方案中 ‘Q’ 和 ‘.’ 分别代表了皇后和空位。

    示例

    示例一

    输入：n = 4 输出：[[".Q..","…Q",“Q…”,"..Q."],["..Q.",“Q…”,"…Q",".Q.."]] 解释：如上图所示，4 皇后问题存在两个不同的解法。

    示例二

    输入：n = 1 输出：[[“Q”]]

    提示

    • 1 <= n <= 9

    题解

    方法一

    • 分析
      • 判断对角线上有没有皇后：
        • 寻找规律，左上到右下的对角线，行与列之差相等；右上到左下的对角线，行与列之和相等
        • 建立unordered_set dig1, dig2来记录有皇后的对角线
    • 提交结果
      • N皇后一通过
    class Solution {
public:
    void findans(vector<vector<string>>&ans, vector<int>&quene, unordered_set<int>&dig1, unordered_set<int>& dig2, unordered_set<int>&col, const int& row, const int& pos, const int& n)
    {
        quene[row] = pos;
        col.insert(pos);
        if (row == n - 1)
        {
            ans.push_back(buildsolution(quene, n));
        }
        else
        {
            for (int i = 0; i < n; i++)
            {
                if (i == pos) continue;
                if (col.find(i) != col.end()) continue;
                int dignal1 = row + 1 - i;
                int dignal2 = row + 1 + i;
                if (dig1.find(dignal1) != dig1.end() || dig2.find(dignal2) != dig2.end())
                {
                    continue;
                }
                dig1.insert(dignal1);
                dig2.insert(dignal2);
                findans(ans, quene, dig1, dig2, col, row+1, i, n);
                dig1.erase(dignal1);
                dig2.erase(dignal2);
            }
        }
        
        col.erase(pos);
        quene[row] = -1;
    }

    vector<string> buildsolution(const vector<int>& quene, const int& n)
    {
        vector<string> asolution;
        for (int i = 0; i < n; i++)
        {
            string arow = string(n, '.');
            arow[quene[i]] = 'Q';
            asolution.push_back(arow);
        }
        return asolution;
    }

    vector<vector<string>> solveNQueens(int n) {
        vector<vector<string>> ans;
        unordered_set<int> dig1;
        unordered_set<int> dig2;
        unordered_set<int> col;
        vector<int> quene(n, -1);
        for (int i = 0; i < n; i++)
        {
            dig1.insert(0 - i);
            dig2.insert(i);
            findans(ans, quene, dig1, dig2, col, 0, i, n);
            dig1.erase(0 - i);
            dig2.erase(i);
        }
        return ans;
    }
};

    class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {

        return solutions;

    void find_pos(vector<vector<string>>& solutions, vector<int>& quenes, const int& n, const int&col, const int &dig1, const int &dig2, const int& row)
    {
        if (row == n)
        {
            addans(solutions, quenes, n);
        } 
        else 
        {
            int available = ((1 << n) - 1) & (~(col | dig1 | dig2));
            while (available != 0)
            {
                int pos = available & (-available);
                available = available & (available - 1);
                int column = __builtin_ctz(pos);
                quenes[row] = column;
                find_pos(solutions, quenes, n, col | pos, (dig1 | pos) >> 1, (dig2 | pos) << 1, row + 1);
                quenes[row] = -1;
            }
        }
    }

    void addans(vector<vector<string>> &solutions, vector<int> &quenes, const int& n)
    {
        vector<string> asolution;
        for (int i = 0; i < n; i++)
        {
            string row = string(n, '.');
            row[quenes[i]] = 'Q';
            asolution.push_back(row);
        }
        solutions.push_back(asolution);
    }

    leetcode hot100 131 分割回文串

    题目

    给你一个字符串 s，请你将 s 分割成一些 子串，使每个子串都是 回文串 。返回 s 所有可能的分割方案。

    示例

    示例一

    输入：s = “aab”
    输出：[[“a”,“a”,“b”],[“aa”,“b”]]

    示例二

    输入：s = “a”
    输出：[[“a”]]

    提示

    • 1 <= s.length <= 16
    • s 仅由小写英文字母组成

    题解

    方法一

    • 分析
      • 递归
      • 对于s中每个字母，记录以它开头的回文字符串的子串的结尾位置
    • 提交结果
      • 通过
    class Solution {
public:
    bool isreverse(const string &s, int i, int j)
    {
        if (i >= j) return true;
        if (isvalid[i][j] != -1)
        {
            return isvalid[i][j];
        }
        
        return s[i] == s[j] && isreverse(s, i + 1, j - 1);
    }

    void paritit(const string &s, vector<string> &solution, int nowpos)
    {
        // cout << nowpos << endl;
        if (nowpos >= s.size())
        {
            ans.push_back(solution);
        }
        else
        {
            // cout << "subend[nowpos].size() = " << subend[nowpos].size() << endl;
            for (int aend : subend[nowpos])
            {
                // cout << aend << endl;
                solution.push_back(s.substr(nowpos, aend - nowpos + 1));
                paritit(s, solution, aend + 1);
                solution.pop_back();
            }
        }
    }


    vector<vector<string>> partition(string s) {
        vector<string> solution;
        int n = s.length();
        for (int i = 0; i < n; i++)
        {
            pos[s[i]].push_back(i);
        }
        isvalid.resize(n);

        for (int i = 0; i < n; i++)
        {
            isvalid[i].resize(n);
            for (int j = 0; j < n; j++)
            {
                isvalid[i][j] = -1;
            }
        }
        for (char c = 'a'; c <= 'z'; c++)
        {
            for (int i = 0; i < pos[c].size(); i++)
            {
                for (int j = i; j < pos[c].size(); j++)
                {
                    // cout << "i = " << i << ", j = " << j << ", pos[c].size() = " << pos[c].size() << endl;
                    if (isreverse(s, pos[c][i], pos[c][j]))
                    {
                        // cout << "j = " << j << endl;
                        subend[pos[c][i]].push_back(pos[c][j]);
                    }
                }
            }
        }

        paritit(s, solution, 0);
        return ans;
    }
private: 
    unordered_map<char, vector<int>> pos;
    unordered_map<int, vector<int>> subend;
    vector<vector<string>> ans;
    vector<vector<int>> isvalid;
};

    关于shell

    • shell种类
      • B-shell

      • 历史替换
      • 别名替换
      • 流程控制的内部命令(内部命令和外部命令)
    • shell的特点
      • 主要用途：批处理，执行效率比算法语言低
      • shell编程风格和C语言等算法语言的区别
        • shell是面向命令处理的语言，提供的流程控制结构通过对一些内部命令的解释实现
      • 如同C语言设计思路一样，shell本身设计的非常精炼，但是它提供了灵活的机制(策略与机制相分离)
        • shell有许多灵活的功能，通过shell替换实现

    • 理解Unix的shell
      • 学习bash的目的

    bash的启动

    • 三种启动方法
      • 注册shell
      • 键入bash命令
      • 脚本解释器
    • 自动执行的一批命令(用户偏好)
      • 当bash作为注册shell被启动时，自动执行用户住目录下的.bash_profile文件中命令，~/.bash_profile或$HOME/.bash_profile
      • 当bash作为注册shell退出时：自动执行$HOME/.bash_logout
      • 当bash作为交互式shell启动时: 自动执行$HOME/.bashrc
      • 类似umask之类的命令，应当写在.profile文件中
    • 自动执行的一批命令（系统级）
      • 当bash作为注册shell被启动时:自动执行 /etc/profile文件中命令
      • 当bash作为交互式shell启动时: 自动执行 /etc/bash.bashrc
      • 当bash作为注册shell退出时:自动执行/etc/bash.bash.logout
    • 脚本文件
      • 编辑文件lsdir (格式为文本文件，文件名不必须为.sh后缀，只是个惯例)
    • 脚本文件的执行
      • 新创建子进程，并在子进程中执行脚本
        • 方法一
          • bash<lsdir
          • 无法携带命令行参数
        • 方法二
          • bash lsdir
          • bash -x lsdir
          • bash lsdir /usr/lib/gcc
        • 方法三
          • 给文件设置可执行属性x：chmod u+x lsdir
          • 然后执行 ./lsdir /usr/lib/gcc
          • 三种方法均启动程序/bin/bash，生成新进程
      • 在当前shell中执行脚本
        • . lsdir /usr/lib/gcc
        • source lsdir /usr/lib/gcc
        • 脚本执行后对当前shell状态有影响

    题目

    数字 n 代表生成括号的对数，请你设计一个函数，用于能够生成所有可能的并且 有效的 括号组合。

    示例

    示例一

    输入：n = 3
    输出：["((()))","(()())","(())()","()(())","()()()"]

    示例二

    输入：n = 1
    输出：["()"]

    提示

    • 1 <= n <= 8

    题解

    方法一

    • 分析
      • 暴力解法
      • 视作2n个字符的全排列，字符有’(’, ‘)‘两种情况
      • 合法字符串应满足的条件
        • ‘)‘不出现在’(‘之前 -> 在遍历字符串的过程中，不能出现balance < 0的情况
        • 左右括号成对出现 -> balance == 0
    • 提交结果
      • 通过
    class Solution {
public:
    bool valid(const string& str)
    {
        int balance = 0;
        for (char c : str)
        {
            if (c == '(')
            {
                ++balance;
            }
            else
            {
                --balance;
            }
            if (balance < 0)
            {
                return false;
            }
        }
        return balance == 0;
    }

    void generate_all(string& current, int n, vector<string>& result)
    {
        if (n == current.size())
        {
            if (valid(current))
            {
                result.push_back(current);
            }
            return;
        }
        current += '(';
        generate_all(current, n, result);
        current.pop_back();
        current += ')';
        generate_all(current, n, result);
        current.pop_back();
    }

    vector<string> generateParenthesis(int n) {
        vector<string> result;
        string current;
        generate_all(current, n * 2, result);
        return result;
    }
};

    方法二

    • 分析
      • 思路与方法一一致
      • 在方法一的基础上做了改进，在构造全排列的过程中判断已有序列是否合法
      • 只存储合法的字符串
    • 提交结果

    class Solution {
public:
    void backtrack(vector<string>& ans, string& cur, int open, int close, int n)
    {
        if (cur.size() == n * 2)
        {
            ans.push_back(cur);
            return;
        }

        if (open < n)
        {
            cur.push_back('(');
            backtrack(ans, cur, open + 1, close, n);
            cur.pop_back();
        }

        if (close < open)
        {
            cur.push_back(')');
            backtrack(ans, cur, open, close + 1, n);
            cur.pop_back();
        }
    }

    vector<string> generateParenthesis(int n) {
        vector<string> result;
        string current;
        backtrack(result, current, 0, 0, n);
        return result;
    }
};

    方法三

    • 分析
      • 递归
      • 将整个字符串分为两部分(a)b，递归求解a和b的所有可能性，再拼成原字符串

    class Solution {
public:
    shared_ptr<vector<string>> generate(int n)
    {
        if (cache[n] != nullptr)
        {
            return cache[n];
        }
        if (!n)
        {
            cache[0] = shared_ptr<vector<string>> (new vector<string> {""});
        }
        else
        {
            auto result = shared_ptr<vector<string>> (new vector<string>);
            for (int i = 0; i != n; i++)
            {
                auto lefts = generate(i);
                auto rights = generate(n - i - 1);
                for (const string& left : *lefts)
                {
                    for (const string& right : *rights)
                    {
                        result->push_back("(" + left + ")" + right);
                    }
                }
            }
            cache[n] = result;
        }
        return cache[n];
    }

    vector<string> generateParenthesis(int n) {
        return *generate(n);
    }
private:
    shared_ptr<vector<string>> cache[100] = {nullptr};
};

public:
    bool search(vector<vector<char>>& board, const string& word, int pos, vector<pair<int, int>>& nowposs)
    {
        // cout << pos << ", " << nowposs.size() << endl;
        if (pos >= word.size())
        {
            if (nowposs.size())
                return true;
            else
                return false;
        }

    class Solution {
public:
    bool dfs(vector<vector<char>>& board, const string& word, int pos, unordered_map<int, unordered_map<int, bool>>& visited, int i, int j)
    {
        // cout << pos << ", " << word.size() << endl;
        if (pos >= word.size())
        {
            return true;
        }
        if (i > 0 && board[i - 1][j] == word[pos])
        {
            if (!visited[i - 1][j])
            {
                visited[i - 1][j] = true;
                if (dfs(board, word, pos + 1, visited, i - 1, j)) return true;;
                visited[i - 1][j] = false;   
            }
        }
        if (i < m - 1 && board[i + 1][j] == word[pos])
        {
            if (!visited[i + 1][j])
            {
                visited[i + 1][j] = true;
                if (dfs(board, word, pos + 1, visited, i + 1, j)) return true;
                visited[i + 1][j] = false;   
            }
        }
        if (j > 0 && board[i][j - 1] == word[pos])
        {
            if (!visited[i][j - 1])
            {
                visited[i][j - 1] = true;
                if (dfs(board, word, pos + 1, visited, i, j - 1)) return true;

                visited[i][j + 1] = false;   
            }
        }

        return false;
    }

        {
            return true;
        }

        if (!m) return false;

                    visited[i][j] = true;
                    if (dfs(board, word, 1, visited, i, j)) return true;
