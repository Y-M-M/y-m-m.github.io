  • leetcode hot100 124 二叉树中的最大路径和

    leetcode hot100 124 二叉树中的最大路径和

    题目

    二叉树中的 路径 被定义为一条节点序列，序列中每对相邻节点之间都存在一条边。同一个节点在一条路径序列中 至多出现一次 。该路径 至少包含一个 节点，且不一定经过根节点。
    路径和 是路径中各节点值的总和。
    给你一个二叉树的根节点 root ，返回其 最大路径和 。

    示例

    示例一


    输入：root = [1,2,3]
    输出：6
    解释：最优路径是 2 -> 1 -> 3 ，路径和为 2 + 1 + 3 = 6

    示例二

    输入：root = [-10,9,20,null,null,15,7]
    输出：42
    解释：最优路径是 15 -> 20 -> 7 ，路径和为 15 + 20 + 7 = 42

    提示

    • 树中节点数目范围是 [1, 3 * 104]
    • -1000 <= Node.val <= 1000

    题解

    方法一

    • 分析
      • dfs
      • 经过一个节点的最大路径和，为其左右子树最大路径和+root->val
      • 注意最右子树最大路径和如果为负数的话，就无需加该负数
      • 实质是在树上的最大子段和
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int dfs(TreeNode* root)
    {
        if (root == nullptr)
        {
            return 0;
        }
        int leftmax = dfs(root->left);
        int rightmax = dfs(root->right);
        int maxpath = 0;
        if (leftmax > 0)  maxpath += leftmax;
        if (rightmax > 0) maxpath += rightmax;
        maxpath += root->val;
        if (maxpath > ans)
        {
            ans = maxpath;
        }
        int addnum = max(leftmax, rightmax);
        if (addnum > 0)
            return addnum + root->val;
        return root->val;
        
    }

    int maxPathSum(TreeNode* root) {
        if (root == nullptr)
        {
            return 0;
        }
        ans = -INT32_MAX;
        dfs(root);
        return ans;
    }
private:
    int ans;
};

    二叉树的最近公共祖先

    题目

    给定一个二叉树, 找到该树中两个指定节点的最近公共祖先。
    百度百科中最近公共祖先的定义为：“对于有根树 T 的两个节点 p、q，最近公共祖先表示为一个节点 x，满足 x 是 p、q 的祖先且 x 的深度尽可能大（一个节点也可以是它自己的祖先）。”

    示例

    示例一


    输入：root = [3,5,1,6,2,0,8,null,null,7,4], p = 5, q = 1
    输出：3
    解释：节点 5 和节点 1 的最近公共祖先是节点 3 。

    示例二


    输入：root = [3,5,1,6,2,0,8,null,null,7,4], p = 5, q = 4
    输出：5
    解释：节点 5 和节点 4 的最近公共祖先是节点 5 。因为根据定义最近公共祖先节点可以为节点本身。 '

    示例三

    输入：root = [1,2], p = 1, q = 2
    输出：1

    题解

    方法一

    • 分析
      • 找到从根节点到两个指定节点的路径，存到两个stack里面
      • 两个stack的重叠部分的最后一个节点，即为最近公共祖先节点
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void dfs(TreeNode* root, TreeNode* p, TreeNode* q)
    {
        if (root == nullptr)
        {
            return;
        }
        st.push(root);
        if (root == p || root == q)
        {
            if (st1.empty())
            {
                st1 = st;
            }
            else
            {
                st2 = st;
                return;
            }
        }
        dfs(root->left, p, q);
        dfs(root->right, p, q);
        st.pop();
    }

    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        dfs(root, p, q);
        int cnt = st1.size() - st2.size();
        if (cnt > 0)
        {
            for (int i = 0; i < cnt; i++)
            {
                st1.pop();
            }
        }
        else if (cnt < 0)
        {
            cnt = -cnt;
            for (int i = 0; i < cnt; i++)
            {
                st2.pop();
            }
        }
        while (!st1.empty() && st1.top() != st2.top())
        {
            st1.pop();
            st2.pop();
        }
        // cout << st1.top()->val << ", " << st2.top()->val << endl;
        if (!st1.empty())
        {
            return st1.top();
        }
        return nullptr;
    }
private:
    stack<TreeNode*> st;
    stack<TreeNode*> st1;
    stack<TreeNode*> st2;
};

    方法二

    • 分析
      • dfs
      • 通过返回值确定子树是否包含指定节点，向上递归
      • ans
        • 左右子树均包含指定节点(最先递归总结到的)
        • 其中一个子树包含指定节点+自己是指定节点
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool dfs(TreeNode* root, TreeNode* p, TreeNode* q)
    {
        if (ans != nullptr)
        {
            return false;
        }
        if (root == nullptr) return false;
        bool lson = dfs(root->left, p, q);
        bool rson = dfs(root->right, p, q);
        if ((lson && rson) || ((root == p || root == q) && (lson || rson)))
        {
            ans = root;
            return false;
        }
        return lson || rson || root == p || root == q;
    }
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        dfs(root, p, q);
        return ans;
    }
private:
    TreeNode* ans;
};

    leetcode hot100 437 路径总和三

    题目

    给定一个二叉树的根节点 root ，和一个整数 targetSum ，求该二叉树里节点值之和等于 targetSum 的 路径 的数目。
    路径 不需要从根节点开始，也不需要在叶子节点结束，但是路径方向必须是向下的（只能从父节点到子节点）。

    示例

    示例一


    输入：root = [10,5,-3,3,2,null,11,3,-2,null,1], targetSum = 8
    输出：3
    解释：和等于 8 的路径有 3 条，如图所示。

    示例二

    输入：root = [5,4,8,11,null,13,4,7,2,null,null,5,1], targetSum = 22
    输出：3

    提示

    • 二叉树的节点个数的范围是 [0,1000]
    • -109 <= Node.val <= 109
    • -1000 <= targetSum <= 1000

    题解

    方法一

    • 分析
      • 为每一条从根节点到叶子结点的路径，求前缀和
      • 针对每一条路径，从前往后遍历，减去前面的节点，即遍历所有l, r对，查找与targetSum相等的数字
      • 问题：求前缀和会超出int的范围
        • 判断边界条件，如果root->val==1000000000的话，就直接返回0(好偷懒哦……)
    • 提交结果
      • 执行出错
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void addtoSub(TreeNode* root, int addval) {
        if (root == nullptr)
        {
            return ;
        }
        root->val += addval;
        // cout << "root->val = " << root->val << endl;
        if (root->left != nullptr)
        {
            addtoSub(root->left, root->val);
        }
        if (root->right != nullptr)
        {
            addtoSub(root->right, root->val);
        }
    }
    void sub(TreeNode* root, const int&targetSum, int subnum)
    {
        if (root == nullptr)
        {
            return;
        }
        root->val -= subnum;
        // cout << root->val << endl;
        if (root->val == targetSum)
        {
            ans++;
        }
        if (root->left != nullptr)
        {
            sub(root->left, targetSum, subnum);
        }
        if (root->right != nullptr)
        {
            sub(root->right, targetSum, subnum);
        }
    }
    void findans(TreeNode* root, const int& targetSum, int subnum)
    {
        //cout << "root->val = " << root->val << endl;
        if (root == nullptr)
        {
            return;
        }
        sub(root, targetSum, subnum);
        if (root->left != nullptr)
        {
            findans(root->left, targetSum, root->val);
        }
        if (root->right != nullptr)
        {
            findans(root->right, targetSum, root->val);
        }
    }
    int pathSum(TreeNode* root, int targetSum) {
        ans = 0;
        if (root == nullptr || root->val == 1000000000)
        {
            return ans;
        }
        addtoSub(root, 0);
        findans(root, targetSum, 0);
        return ans;
    }
private:
    int ans;
};

    方法二

    • 分析
      • 方法一存在的问题：int无法存储前缀和；对于每一个起始位置，都需要顺序查找符合条件的结束为止
      • 改进：使用unordered_map<long long, int>存储前缀和到数量的映射关系，在遍历到每个前缀和的时候通过哈希表查找符合条件的子段和的数量
      • 核心：起始与结束的位置不重要，只有符合条件的数字的数量才重要
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int dfs(TreeNode* root, long long curr, int targetSum)
    {
        if (!root)
        {
            return 0;
        }

        int ret = 0;
        curr += root->val;
        if (prefix.count(curr - targetSum))
        {
            ret = prefix[curr - targetSum];
        }
        prefix[curr]++;
        ret += dfs(root->left, curr, targetSum);
        ret += dfs(root->right, curr, targetSum);
        prefix[curr]--;

        return ret;
    }


    int pathSum(TreeNode* root, int targetSum) {
        prefix[0] = 1;
        return dfs(root, 0, targetSum);
    }
private:
    unordered_map<long long, int> prefix;
};

    二叉树展开为链表

    题目

    给你二叉树的根结点 root ，请你将它展开为一个单链表：
    展开后的单链表应该同样使用 TreeNode ，其中 right 子指针指向链表中下一个结点，而左子指针始终为 null 。
    展开后的单链表应该与二叉树 先序遍历 顺序相同。

    示例

    示例一


    输入：root = [1,2,5,3,4,null,6]
    输出：[1,null,2,null,3,null,4,null,5,null,6]

    示例二

    输入：root = []
    输出：[]

    示例三

    输入：root = [0]
    输出：[0]

    提示

    • 树中结点数在范围 [0, 2000] 内
    • -100 <= Node.val <= 100

    题解

    方法一

    • 分析
      • 从根节点开始，将左子树插入右子树中，根节点左子树置为0
      • 将根节点向右下移动，直到遍历到的节点的左右子树均为nullptr为止
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void flatten(TreeNode* root) {
        if (root == nullptr)
        {
            return;
        }
        TreeNode* temp = nullptr;
        TreeNode* rend = nullptr;
        while (root->left != nullptr || root->right != nullptr)
        {
            if (root->left != nullptr)
            {
                temp = root->right;
                root->right = root->left;
                root->left = nullptr;
                root = root->right;
                rend = root;
                while (rend->right != nullptr)
                {
                    rend = rend->right;
                }
                rend->right = temp;
            }
            else
            {
                root = root->right;
            }
        }
    }
};

    /**

 *     TreeNode *left;
 *     TreeNode *right;

        vector<int> ans;

                    ans.push_back(nownode->val);
                }
                if (nownode->left != nullptr)
                {
                    layer.push(nownode->left);
                }
                if (nownode->right != nullptr)
                {
                    layer.push(nownode->right);
                }
            }
        }
        return ans;
    }
};

    leetcodehot105 从前序与中序遍历构造二叉树

    题目

    给定两个整数数组 preorder 和 inorder ，其中 preorder 是二叉树的先序遍历， inorder 是同一棵树的中序遍历，请构造二叉树并返回其根节点。

    示例

    示例一


    输入: preorder = [3,9,20,15,7], inorder = [9,3,15,20,7]
    输出: [3,9,20,null,null,15,7]

    示例二

    输入: preorder = [-1], inorder = [-1]
    输出: [-1]

    提示

    • 1 <= preorder.length <= 3000
    • inorder.length == preorder.length
    • -3000 <= preorder[i], inorder[i] <= 3000
    • preorder 和 inorder 均 无重复 元素
    • inorder 均出现在 preorder
    • preorder 保证 为二叉树的前序遍历序列
    • inorder 保证 为二叉树的中序遍历序列

    题解

    方法一

    • 分析
      • 递归的构造
      • 利用先序遍历确认当前子树的根节点
      • 利用中序遍历确定左右子树在先序遍历序列中的范围
    • 提交结果
      • 通过
    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int findendl(vector<int> &preorder, int rootpos, int beginpos)
    {
        int i = beginpos + 1;
        while (i < preorder.size() && inpos[preorder[i]] < rootpos)
        {
            i++;
        }
        return i - 1;
    }

    TreeNode* buidlsubTree(vector<int> &preorder, vector<int>& inorder, int l1, int r1, int l2, int r2)
    {
        if (l1 > r1 || l2 > r2)
        {
            return nullptr;
        }
        int rootval = preorder[l1];
        TreeNode* root = new TreeNode(rootval);
        if (r1 == l1)
        {
            return root;
        }
        if (inpos[rootval] == l2)
        {
            root->right = buidlsubTree(preorder, inorder, l1 + 1, r1, l2 + 1, r2);
            return root;
        }
        int preendl = findendl(preorder, inpos[rootval], l1);
        // cout << "preendl = " << preendl << endl;
        root->left = buidlsubTree(preorder, inorder, l1 + 1, preendl, l2, inpos[rootval] - 1);
        root->right = buidlsubTree(preorder, inorder, preendl + 1, r1,  inpos[rootval] + 1, r2);
        return root;
    }

    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        TreeNode* root = nullptr;
        if (preorder.size() == 0)
        {
            return root;
        }
        for (int i = 0; i < preorder.size(); i++)
        {
            prepos[preorder[i]] = i;
        }
        for (int i = 0; i < inorder.size(); i++)
        {
            inpos[inorder[i]] = i;
        }
        root = buidlsubTree(preorder, inorder, 0, preorder.size() - 1, 0, inorder.size() - 1);
        return root;
    }
private:
    unordered_map<int, int> prepos;
    unordered_map<int, int> inpos;
};

            root->right = buidlsubTree(preorder, inorder, l1 + 1, r1, l2 + 1, r2);
            return root;
        }
        int preendl = l1 + inpos[rootval] - l2;
        // cout << "preendl = " << preendl << endl;
        root->left = buidlsubTree(preorder, inorder, l1 + 1, preendl, l2, inpos[rootval] - 1);
        root->right = buidlsubTree(preorder, inorder, preendl + 1, r1,  inpos[rootval] + 1, r2);
        return root;
    }

    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        TreeNode* root = nullptr;
        if (preorder.size() == 0)
        {
            return root;
        }
        for (int i = 0; i < inorder.size(); i++)
        {
            inpos[inorder[i]] = i;
        }
        root = buidlsubTree(preorder, inorder, 0, preorder.size() - 1, 0, inorder.size() - 1);
        return root;
    }
private:
    unordered_map<int, int> inpos;
};

    • 分析
      • 迭代
      • 利用栈记录祖先节点

 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        TreeNode* root = nullptr;
        if (preorder.size() == 0)
        {
            return root;
        }
        for (int i = 0; i < inorder.size(); i++)
        {
            inpos[inorder[i]] = i;
        }
        root = new TreeNode(preorder[0]);
        stack<TreeNode*> st;
        st.push(root);
        for (int i = 1; i < preorder.size(); i++)
        {
            int preval = st.top()->val;
            int nowval = preorder[i];
            if (inpos[nowval] < inpos[preval])
            {
                TreeNode* node = new TreeNode(nowval);
                st.top()->left = node;
                st.push(node);
            }
            else
            {
                TreeNode* prenode = nullptr;
                while (!st.empty() && inpos[nowval] > inpos[preval])
                {
                    prenode = st.top();
                    st.pop();      
                    if (st.empty())
                    {
                        break;
                    }
                    preval = st.top()->val;
                }
                TreeNode* node = new TreeNode(nowval);
                st.push(node);
                prenode->right = node;
            }
        }
        return root;
    }
private:
    unordered_map<int, int> inpos;
};

        }

    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        queue<TreeNode*> q;
        vector<vector<int>> ans;
        if (root == nullptr)
        {
            return ans;
        }
        q.push(root);
        while (!q.empty())
        {
            int len = q.size();
            vector<int> nowlayer;

                nowlayer.push_back(curr->val);
                if (curr->left != nullptr)
                {
                    q.push(curr->left);
                }
                if (curr->right != nullptr)
                {
                    q.push(curr->right);
                }
            }
            ans.push_back(nowlayer);
        }
        return ans;
    }

    输出：[3,1]
    解释：[1,null,3] 和 [3,1] 都是高度平衡二叉搜索树。

    提示

    • 1 <= nums.length <= 104
    • -104 <= nums[i] <= 104
    • nums 按 严格递增 顺序排列

    题解

    方法一

    • 分析

    • 提交结果

    /**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* changetoBST(vector<int>& nums, const int& l, const int& r)
    {
        TreeNode* node = nullptr;
        if (l == r)
        {
            node = new TreeNode(nums[l]);
        }
        else if (r - l == 1)
        {
            node = new TreeNode(nums[l]);
            TreeNode* another_node = new TreeNode(nums[r]);
            node->right = another_node;
        }
        else
        {
            int mid = (l + r) / 2;
            node = new TreeNode(nums[mid]);
            node->left = changetoBST(nums, l, mid - 1);
            node->right = changetoBST(nums, mid + 1, r);
        }
        return node;
    }

        return changetoBST(nums, 0, nums.size() - 1);
