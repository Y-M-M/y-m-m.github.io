    leetcode hot100 295 数据流的中位数

    题目

    中位数是有序整数列表中的中间值。如果列表的大小是偶数，则没有中间值，中位数是两个中间值的平均值。
    例如 arr = [2,3,4] 的中位数是 3 。
    例如 arr = [2,3] 的中位数是 (2 + 3) / 2 = 2.5 。
    实现 MedianFinder 类:
    MedianFinder() 初始化 MedianFinder 对象。
    void addNum(int num) 将数据流中的整数 num 添加到数据结构中。
    double findMedian() 返回到目前为止所有元素的中位数。与实际答案相差 10-5 以内的答案将被接受。

    示例

    示例一

    输入
    [“MedianFinder”, “addNum”, “addNum”, “findMedian”, “addNum”, “findMedian”]
    [[], [1], [2], [], [3], []]
    输出
    [null, null, null, 1.5, null, 2.0]
    解释
    MedianFinder medianFinder = new MedianFinder();
    medianFinder.addNum(1); // arr = [1]
    medianFinder.addNum(2); // arr = [1, 2]
    medianFinder.findMedian(); // 返回 1.5 ((1 + 2) / 2)
    medianFinder.addNum(3); // arr[1, 2, 3]
    medianFinder.findMedian(); // return 2.0

    • 提交结果

    class Solution {
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        unordered_map<int, int> times;
        vector<int> snums;
        for (int num : nums)
        {
            if (!times[num]) snums.push_back(num);
            times[num]++;
            // cout << num << ", " << times[num] << endl;
        }
        
        auto cmp = [&times](int x, int y) -> bool 
        {
            return times[x] > times[y];
        };
        sort(snums.begin(), snums.end(), cmp);
        snums.resize(k);
        return snums;
    }
};

    leetcode hot100 215 数组中第K个最大元素

    题目

    给定整数数组 nums 和整数 k，请返回数组中第 k 个最大的元素。
    请注意，你需要找的是数组排序后的第 k 个最大的元素，而不是第 k 个不同的元素。
    你必须设计并实现时间复杂度为 O(n) 的算法解决此问题。

    示例

    示例一

    输入: [3,2,1,5,6,4], k = 2
    输出: 5

    示例二

    输入: [3,2,3,1,2,4,5,5,6], k = 4
    输出: 4

    提示

    • 1 <= k <= nums.length <= 105
    • -104 <= nums[i] <= 104

    题解

    方法一

    • 分析
      • 线性时间选择
    • 提交结果
      • 通过
    class Solution {
public:
    int quickselect(vector<int> &nums, int l, int r, int k) {
        if (l == r)
            return nums[k];
        int partition = nums[l], i = l - 1, j = r + 1;
        while (i < j) {
            do i++; while (nums[i] < partition);
            do j--; while (nums[j] > partition);
            if (i < j)
                swap(nums[i], nums[j]);
        }
        if (k <= j)return quickselect(nums, l, j, k);
        else return quickselect(nums, j + 1, r, k);
    }

    int findKthLargest(vector<int> &nums, int k) {
        int n = nums.size();
        return quickselect(nums, 0, n - 1, n - k);
    }
};

    class Solution {
public:
    void maxHeapify(vector<int>& a, int i, int heapSize) {
        int l = i * 2 + 1, r = i * 2 + 2, largest = i;
        if (l < heapSize && a[l] > a[largest]) {
            largest = l;
        } 
        if (r < heapSize && a[r] > a[largest]) {
            largest = r;
        }
        if (largest != i) {
            swap(a[i], a[largest]);
            maxHeapify(a, largest, heapSize);
        }
    }

    void buildMaxHeap(vector<int>& a, int heapSize) {
        for (int i = heapSize / 2 - 1; i >= 0; --i) {
            maxHeapify(a, i, heapSize);
        } 
    }

    int findKthLargest(vector<int>& nums, int k) {
        int heapSize = nums.size();
        buildMaxHeap(nums, heapSize);
        for (int i = nums.size() - 1; i >= nums.size() - k + 1; --i) {
            swap(nums[0], nums[i]);
            --heapSize;
            maxHeapify(nums, 0, heapSize);
        }
        return nums[0];
    }
};

  • leetcode hot100 84 柱状图中最大的矩形

    题目

    给定 n 个非负整数，用来表示柱状图中各个柱子的高度。每个柱子彼此相邻，且宽度为 1 。
    求在该柱状图中，能够勾勒出来的矩形的最大面积。

    示例

    示例一

    输入：heights = [2,1,5,6,2,3]
    输出：10
    解释：最大的矩形为图中红色区域，面积为 10

    示例二


    输入： heights = [2,4]
    输出： 4

    提示

    • 1 <= heights.length <=105
    • 0 <= heights[i] <= 104

    题解

    方法一

    • 分析
      • 遍历所有height，同时维护一个数组，记录从i开始到当前位置j的最大面积
    • 提交结果
      • 超出时间限制
      • 89/99个通过的测试样例
    • 错误分析
      • 如果当前的位置为j，j之前有两个位置i1和i2
      • 如果i2开始的面积大于i1开始的面积，且i2的最大高度大于i1开始的最大高度，则i1不用再处理了，它不可能得到最优解
    class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int n = heights.size();
        vector<int> minhi(n, INT32_MAX);
        int ans = -INT32_MAX;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j <= i; j++)
            {
                // cout << i << ", " << j << endl;
                minhi[j] = min(minhi[j], heights[i]);
                ans = max(minhi[j] * (i - j + 1), ans);
            }
        }
        return ans;
    }
};

    方法二

    • 分析
      • 使用栈维护height，其中height单调递增，栈中记录pos
      • 当出现比stack.top()对应的height更小的height的时候，stack.top()在之后就没用了，可以出栈
      • 当出栈时记录它及其所能构造的最大面积
    • 提交结果
      • 通过
    class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        stack<int> st;
        int ans = 0;
        heights.push_back(0);
        for (int r = 0; r < heights.size(); r++) {
            while (!st.empty() && heights[r] <= heights[st.top()]) {
                auto i = st.top();
                st.pop();
                int l = st.empty() ? -1 : st.top();
                ans = max(ans, (r - l - 1) * heights[i]);
            }
            st.push(r);
        }
        return ans;
    }
};

    方法一

    class Solution {
public:
    string getDigits(string &s, size_t &ptr)
    {
        string ret = "";
        while (isdigit(s[ptr]))
        {
            ret.push_back(s[ptr++]);
        }
        return ret;
    }

    string getString(vector<string> &v)
    {
        string ret;
        for (const auto &s : v)
        {
            ret += s;
        }
        return ret;
    }

    string decodeString(string s) {
        vector<string> stk;
        size_t ptr = 0;

        while (ptr < s.size())
        {
            char cur = s[ptr];
            if (isdigit(cur)) // 获取一个数字并进栈
            {
                string digits = getDigits(s, ptr);
                stk.push_back(digits);
            }
            else if (isalpha(cur) || cur == '[') // 获取一个字母并进栈
            {
                stk.push_back(string(1, s[ptr++]));
            }
            else // 右括号
            {
                ++ptr;
                vector<string> sub;
                while (stk.back() != "[")
                {
                    sub.push_back(stk.back());
                    stk.pop_back();
                }
                reverse(sub.begin(), sub.end());
                // 左括号出栈
                stk.pop_back();
                // 此时栈顶为当前sub对应的字符串应该出现的次数
                int repTime = stoi(stk.back());
                stk.pop_back();
                string t, o = getString(sub);
                // 构造字符串
                while (repTime--) t += o;
                // 将构造好的字符串入栈
                stk.push_back(t);
            }
        }

        return getString(stk);
    }
};

    leetcode hot100 739 每日温度

    题目

    给定一个整数数组 temperatures ，表示每天的温度，返回一个数组 answer ，其中 answer[i] 是指对于第 i 天，下一个更高温度出现在几天后。如果气温在这之后都不会升高，请在该位置用 0 来代替。

    示例

    示例一

    输入: temperatures = [73,74,75,71,69,72,76,73]
    输出: [1,1,4,2,1,1,0,0]

    示例二

    输入: temperatures = [30,40,50,60]
    输出: [1,1,1,0]

    示例三

    输入: temperatures = [30,60,90]
    输出: [1,1,0]

    提示

    • 1 <= temperatures.length <= 105
    • 30 <= temperatures[i] <= 100

    题解

    方法一

    • 分析
      • 使用prioity_queue记录尚未找到答案的位置，按temperature从小到达排列
      • 从前往后遍历，遍历到每一个新的temperature，就更新能更新的
    • 提交结果
      • 通过
    class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        auto cmp = [&temperatures](int left, int right)
        {
            return temperatures[left] > temperatures[right];
        };
        priority_queue<int, vector<int>, decltype(cmp)> pq(cmp);
        int n = temperatures.size();
        vector<int> ans(n, 0);
        for (int i = 0; i < n; i++)
        {
            if (!pq.empty())
            {
                int updatepos = pq.top();
                
                while (temperatures[updatepos] < temperatures[i])
                {
                    // cout << temperatures[updatepos] << ", " << temperatures[i] << endl;
                    ans[updatepos] = i - updatepos;
                    pq.pop();
                    if (pq.empty())
                    {
                        break;
                    }
                    updatepos = pq.top();
                }
            }
            pq.push(i);
        }
        return ans;
    }
};

    方法二

    • 分析
      • 使用递减栈
      • 将所有比当前temperature小的栈顶元素都出栈，并更新ans数组
    • 提交结果
      • 通过
    class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        stack<int> pos;
        int n = temperatures.size();
        vector<int> ans(n, 0);
        for (int i = 0; i < temperatures.size(); i++)
        {
            while (!pos.empty())
            {
                int nowpos = pos.top();
                if (temperatures[nowpos] < temperatures[i])
                {
                    ans[nowpos] = i - nowpos;
                    pos.pop();
                }
                else
                {
                    break;
                }
            }
            pos.push(i);
        }
        return ans;
    }
};

    leetcode hot100 155 最小栈

    题目

    设计一个支持 push ，pop ，top 操作，并能在常数时间内检索到最小元素的栈。
    实现 MinStack 类:
    MinStack() 初始化堆栈对象。
    void push(int val) 将元素val推入堆栈。
    void pop() 删除堆栈顶部的元素。
    int top() 获取堆栈顶部的元素。
    int getMin() 获取堆栈中的最小元素。

    示例

    示例一

    输入：
    [“MinStack”,“push”,“push”,“push”,“getMin”,“pop”,“top”,“getMin”]
    [[],[-2],[0],[-3],[],[],[],[]]
    输出：
    [null,null,null,null,-3,null,0,-2]
    解释：
    MinStack minStack = new MinStack();
    minStack.push(-2);
    minStack.push(0);
    minStack.push(-3);
    minStack.getMin(); –> 返回 -3.
    minStack.pop();
    minStack.top(); –> 返回 0.
    minStack.getMin(); –> 返回 -2.

    提示

    • -231 <= val <= 231 - 1
    • pop、top 和 getMin 操作总是在 非空栈 上调用
    • push, pop, top, and getMin最多被调用 3 * 104 次

    题解

    方法一

        {

        if (nums.back() == minnum)

            minnum = INT32_MAX;
            for (int i = 0; i < nums.size() - 1; i++)
            {
                if (nums[i] < minnum)
                {
                    minnum = nums[i];
                }
            }
        }
        nums.pop_back();
    }
    
    int top() {
        return nums.back();
    }
    
    int getMin() {
        return minnum;
    }
private:
    vector<int> nums;
    int minnum;
};

    leetcode hot100 20 有效的括号

    题目

    给定一个只包括 ‘(’，’)’，’{’，’}’，’[’，’]’ 的字符串 s ，判断字符串是否有效。
    有效字符串需满足：
    左括号必须用相同类型的右括号闭合。
    左括号必须以正确的顺序闭合。
    每个右括号都有一个对应的相同类型的左括号。

    示例

    示例一

    输入：s = “()”
    输出：true

    示例二

    输入：s = “()[]{}”
    输出：true

    示例三

    输入：s = “(]”
    输出：false

    示例四

    输入：s = “([])”
    输出：true

    提示

    • 1 <= s.length <= 104
    • s 仅由括号 ‘()[]{}’ 组成

    题解

    • 分析
      • 使用栈存储左括号
      • 如果匹配到合适的右括号，就将栈顶元素出栈
      • 如果当前c与栈顶的左括号不匹配，或者左右括号数目不等，说明字符串不匹配
    • 提交结果
      • 通过
    class Solution {
public:
    bool isValid(string s) {
        stack<char> left;
        for (char c : s)
        {
            if (c == '(' || c == '{' || c == '[') left.push(c);
            else 
            {
                if (left.size() == 0) return false;
                if ((c == ')' && left.top() == '(')
                    || (c == '}' && left.top() == '{')
                    || (c == ']' && left.top() == '['))
                {
                    left.pop();
                }
                else
                {
                    return false;
                }
            }
        }
        if (left.size()) return false;
        return true;
    }
};

    题目

    已知一个长度为 n 的数组，预先按照升序排列，经由 1 到 n 次 旋转 后，得到输入数组。例如，原数组 nums = [0,1,2,4,5,6,7] 在变化后可能得到：
