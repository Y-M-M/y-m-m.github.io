题目

在 LeetCode 商店中， 有 n 件在售的物品。每件物品都有对应的价格。然而，也有一些大礼包，每个大礼包以优惠的价格捆绑销售一组物品。 给你一个整数数组 price 表示物品价格，其中 price[i] 是第 i 件物品的价格。另有一个整数数组 needs 表示购物清单，其中 needs[i] 是需要购买第 i 件物品的数量。 还有一个数组 special 表示大礼包，special[i] 的长度为 n + 1 ，其中 special[i][j] 表示第 i 个大礼包中内含第 j 件物品的数量，且 special[i][n] （也就是数组中的最后一个整数）为第 i 个大礼包的价格。 返回 确切 满足购物清单所需花费的最低价格，你可以充分利用大礼包的优惠活动。你不能购买超出购物清单指定数量的物品，即使那样会降低整体价格。任意大礼包可无限次购买。

示例

示例一

输入：price = [2,5], special = [[3,0,5],[1,2,10]], needs = [3,2] 输出：14 解释：有 A 和 B 两种物品，价格分别为 ¥2 和 ¥5 。 大礼包 1 ，你可以以 ¥5 的价格购买 3A 和 0B 。 大礼包 2 ，你可以以 ¥10 的价格购买 1A 和 2B 。 需要购买 3 个 A 和 2 个 B ， 所以付 ¥10 购买 1A 和 2B（大礼包 2），以及 ¥4 购买 2A 。

示例二

输入：price = [2,3,4], special = [[1,1,0,4],[2,2,1,9]], needs = [1,2,1] 输出：11 解释：A ，B ，C 的价格分别为 ¥2 ，¥3 ，¥4 。 可以用 ¥4 购买 1A 和 1B ，也可以用 ¥9 购买 2A ，2B 和 1C 。 需要买 1A ，2B 和 1C ，所以付 ¥4 买 1A 和 1B（大礼包 1），以及 ¥3 购买 1B ， ¥4 购买 1C 。 不可以购买超出待购清单的物品，尽管购买大礼包 2 更加便宜。

提示

  • n == price.length == needs.length
  • 1 <= n <= 6
  • 0 <= price[i], needs[i] <= 10
  • 1 <= special.length <= 100
  • special[i].length == n + 1
  • 0 <= special[i][j] <= 50
  • 生成的输入对于 0 <= j <= n - 1 至少有一个 special[i][j] 非零。

题解

方法一

class Solution {
public:
    map<vector<int>, int> memory; // 记忆化数组
    int shoppingOffers(vector<int>& price, vector<vector<int>>& special, vector<int>& needs) {
        vector<vector<int>> useful; // 只保留有用的大礼包
        int n = price.size();
        for (int i = 0; i < special.size(); i++)
        {
            int origin_price = 0;
            int out_needs = 0;
            for (int j = 0; j < n; j++) { 
                if (special[i][j] > needs[j]) {
                    out_needs = 1;
                    break;
                }
                origin_price += special[i][j] * price[j];
            }
            // 去掉会超出needs中的物品数目的大礼包和没没有优惠的大礼包
            if (out_needs || origin_price <= special[i][n]) continue; 
            else {
                useful.emplace_back(special[i]);
            }
        }
        return dfs(price, useful, needs); // 记忆化搜索
    }

    int dfs(vector<int>& price, vector<vector<int>>& useful, vector<int>& needs) {
        if (memory.count(needs) == 0) { // 如果之前没有经历过现在这种情况(没有记忆)
            int min_price = 0;
            for (int i = 0; i < needs.size(); i++) // 不购买大礼包
            {
                min_price += needs[i] * price[i];
            }
            for (int i = 0; i < useful.size(); i++) // 遍历所有大礼包，购买能购买的大礼包，再递归分析剩下的
            {
                int special_price = useful[i][price.size()];
                vector<int> next_needs;
                int out_needs = 0;
                for (int j = 0; j < needs.size(); j++)
                {
                    if (needs[j] < useful[i][j]) {
                        out_needs = 1;
                        break;
                    }
                    next_needs.emplace_back(needs[j] - useful[i][j]);
                }
                if (!out_needs)
                {
                    min_price = min(min_price, dfs(price, useful, next_needs) + special_price);
                }
            }
            memory[needs] = min_price;
        }
        return memory[needs];
    }
};

    • 通过
  • 分析
    • 记忆化搜索，记忆数组memory
    • 以vector needs，当前需要购买的物品的数目作为当前的状态
    • 对每个状态，考虑不购买大礼包和购买大礼包两种情形，取price的最小值
    • 对于购买大礼包的情形
      • 遍历所有useful大礼包
      • 对于所有能购买的大礼包，选择购买它，再递归分析剩下的needs
    • 递归的方法能简化思路
    • 自顶向下，不必遍历所有状态

方法二

class Solution {
public:
    unordered_map<int, int> memory;
    const int bits = 4;
    int shoppingOffers(vector<int>& price, vector<vector<int>>& special, vector<int>& needs) {
        vector<vector<int>> useful;
        int n = price.size();
        int mask = 0;
        for (int i = 0; i < n; i++) { // 状态压缩，用int存储needs
            mask |= needs[i] << (i * bits);
        }
        for (int i = 0; i < special.size(); i++)
        {
            int origin_price = 0;
            int out_needs = 0;
            for (int j = 0; j < n; j++) {
                if (special[i][j] > (mask >> (j * bits) & 0xf)) {
                    out_needs = 1;
                    break;
                }
                origin_price += special[i][j] * price[j];
            }
            if (out_needs || origin_price <= special[i][n]) continue;
            else {
                useful.emplace_back(special[i]);
            }
        }
        return dfs(price, special, mask);
    }

    int dfs(vector<int>& price, vector<vector<int>>& useful, int& needs) {
        if (memory.count(needs) == 0) {
            int min_price = 0;
            for (int i = 0; i < price.size(); i++)
            {
                min_price += ((needs >> i * bits) & 0xf) * price[i];
            }
            for (int i = 0; i < useful.size(); i++)
            {
                int next_needs = needs;
                int special_price = useful[i][price.size()];
                int out_needs = 0;
                for (int j = 0; j < price.size(); j++)
                {
                    if ((needs >> bits * j & 0xf) < useful[i][j]) {
                        out_needs = 1;
                        break;
                    }
                    next_needs -= useful[i][j] << (j * bits);
                }
                if (!out_needs)
                {
                    min_price = min(min_price, dfs(price, useful, next_needs) + special_price);
                }
            }
            memory[needs] = min_price;
        }
        return memory[needs];
    }
};

    • 通过
  • 分析
    • 状态压缩+位运算
    • 建立在方法一的基础上
    • 根据题目所给的范围, needs[i] < 10，不会超过4位，1 ≤ needs.size() ≤ 6
      • 如果用4位表示一个need，则总位数不会超过24，在int的取值范围内
      • 故可以用一个int来压缩表示整个needs数组，即状态
      • 更新状态：在选择购买大礼包的时候，只需要在int中对应位减去购买的大礼包的对应位，即可更新状态
