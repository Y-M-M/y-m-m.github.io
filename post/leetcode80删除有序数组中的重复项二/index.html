leetcode 80 删除有序数组中的重复项二

题目

给你一个有序数组 nums ，请你 原地 删除重复出现的元素，使得出现次数超过两次的元素只出现两次 ，返回删除后数组的新长度。
不要使用额外的数组空间，你必须在 原地 修改输入数组 并在使用 O(1) 额外空间的条件下完成。

示例

示例一

输入：nums = [1,1,1,2,2,3]
输出：5, nums = [1,1,2,2,3]
解释：函数应返回新长度 length = 5, 并且原数组的前五个元素被修改为 1, 1, 2, 2, 3。 不需要考虑数组中超出新长度后面的元素。

示例二

输入：nums = [0,0,1,1,1,1,2,3,3]
输出：7, nums = [0,0,1,1,2,3,3]
解释：函数应返回新长度 length = 7, 并且原数组的前七个元素被修改为 0, 0, 1, 1, 2, 3, 3。不需要考虑数组中超出新长度后面的元素。

提示

  • 1 <= nums.length <= 3 * 104
  • -104 <= nums[i] <= 104
  • nums 已按升序排列

题解

方法一

class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int ans = 0;
        int len = nums.size();
        int pos = 0;
        while (len > pos)
        {
            ans++;
            int nownum = nums[pos];
            if (pos + 1 < len && nums[pos + 1] == nownum)
            {
                ans++;
                pos += 2;
                int newpos = pos;
                int nowpos = pos;
                while (newpos < len && nums[newpos] == nownum)
                {
                    newpos++;
                }
                if (newpos > nowpos) // 超过两个，移动
                {
                    for (int i = newpos; i < len; i++)
                    {
                        nums[nowpos++] = nums[newpos++];
                    }
                }
                len -= (newpos - nowpos);
            }
            else
            {
                pos++;
            }
        }
        return ans;
    }
};

方法二

class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        if (nums.size() <= 2)
        {
            return nums.size();
        }
        int ans = 1;
        int l = 1, r = 1;
        int havetwo = 0;
        while (r < nums.size())
        {
            if (nums[r] != nums[r - 1])
            {
                havetwo = 0;
            }
            else
            {
                if (havetwo)
                {
                    r++;
                    while (r < nums.size() && nums[r] == nums[r - 1])
                    {
                        r++;
                    }
                    continue;
                }
                else
                {
                    havetwo = 1;
                }
            }
            ans++;
            if (l != r)
            {
                nums[l] = nums[r];
            }
            l++;
            r++;
        }
        return l;
    }
};
